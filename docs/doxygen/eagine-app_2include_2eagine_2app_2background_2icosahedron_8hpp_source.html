54 //------------------------------------------------------------------------------
55 inline background_icosahedron::background_icosahedron(
56  video_context& vc,
57  const oglplus::vec4 edge_color,
58  const oglplus::vec4 face_color,
59  const float depth) noexcept
60  : _ecolor{edge_color}
61  , _fcolor{face_color}
62  , _depth{depth} {
63  const auto& glapi = vc.gl_api();
64  const auto& [gl, GL] = glapi;
65  memory::buffer temp;
66 
67  gl.clear_depth(_depth);
68 
69  gl.create_program() >> _prog;
70 
71  oglplus::owned_shader_name fs;
72  if(gl.create_shader(GL.fragment_shader) >> fs) {
73  const auto cleanup_fs = gl.delete_shader.raii(fs);
74 
75  const string_view fs_source = R"(
76  #version 140
77  in vec4 geomColor;
78  out vec4 fragColor;
79 
80  void main() {
81  fragColor = geomColor;
82  })";
83 
84  gl.shader_source(fs, oglplus::glsl_string_ref(fs_source));
85  gl.compile_shader(fs);
86  gl.attach_shader(_prog, fs);
87  }
88 
89  oglplus::owned_shader_name vs;
90  if(gl.create_shader(GL.vertex_shader) >> vs) {
91  const auto cleanup_vs = gl.delete_shader.raii(vs);
92 
93  oglplus::owned_shader_name gs;
94  if(gl.create_shader(GL.geometry_shader) >> gs) {
95  const auto cleanup_gs = gl.delete_shader.raii(gs);
96 
97  const string_view gs_source = R"(
98  #version 330
99 
100  layout(triangles) in;
101  layout(triangle_strip, max_vertices = 3) out;
102  out vec4 geomColor;
103  uniform mat4 Camera;
104  uniform vec4 Color;
105 
106  void main() {
107  float m = mod(gl_PrimitiveIDIn / 1.618, 1.0);
108  geomColor = vec4(mix(0.96, 1.04, m) * Color.rgb, Color.a);
109 
110  for(int i=0; i<3; ++i) {
111  gl_Position = Camera * gl_in[i].gl_Position;
112  EmitVertex();
113  }
114  })";
115  gl.shader_source(gs, oglplus::glsl_string_ref(gs_source));
116  gl.compile_shader(gs);
117  gl.attach_shader(_prog, gs);
118 
119  const string_view vs_source = R"(
120  #version 140
121  in vec3 Position;
122  uniform float Scale = 1.0;
123 
124  void main() {
125  gl_Position = vec4(Position * Scale, 1.0);
126  })";
127  gl.shader_source(vs, oglplus::glsl_string_ref(vs_source));
128  } else {
129  const string_view vs_source = R"(
130  #version 140
131  in vec3 Position;
132  out vec4 geomColor;
133  uniform mat4 Camera;
134  uniform vec4 Color;
135  uniform float Scale = 1.0;
136 
137  void main() {
138  gl_Position = Camera * vec4(Position * Scale, 1.0);
139  geomColor = Color;
140  })";
141  gl.shader_source(vs, oglplus::glsl_string_ref(vs_source));
142  }
143  gl.compile_shader(vs);
144  gl.attach_shader(_prog, vs);
145  }
146 
147  gl.link_program(_prog);
148  gl.use_program(_prog);
149 
150  gl.get_uniform_location(_prog, "Camera") >> _camera_loc;
151  gl.get_uniform_location(_prog, "Scale") >> _scale_loc;
152  gl.get_uniform_location(_prog, "Color") >> _color_loc;
153 
154  oglplus::shape_generator shape{
155  glapi, shapes::unit_icosahedron(shapes::vertex_attrib_kind::position)};
156 
157  _ops.resize(integer(shape.operation_count()));
158  shape.instructions(glapi, cover(_ops));
159 
160  gl.gen_vertex_arrays() >> _vao;
161  gl.bind_vertex_array(_vao);
162 
163  gl.gen_buffers() >> _positions;
164  shape.attrib_setup(
165  glapi,
166  _vao,
167  _positions,
168  oglplus::vertex_attrib_location{0},
169  eagine::shapes::vertex_attrib_kind::position,
170  temp);
171 
172  gl.gen_buffers() >> _indices;
173  shape.index_setup(glapi, _indices, temp);
174 }
175 //------------------------------------------------------------------------------
176 inline auto background_icosahedron::clean_up(video_context& vc) noexcept
177  -> background_icosahedron& {
178  const auto& gl = vc.gl_api();
179 
180  gl.delete_program(std::move(_prog));
181  gl.delete_buffers(std::move(_indices));
182  gl.delete_buffers(std::move(_positions));
183  gl.delete_vertex_arrays(std::move(_vao));
184  return *this;
185 }
186 //------------------------------------------------------------------------------
187 inline auto background_icosahedron::clear(
188  video_context& vc,
189  const orbiting_camera& camera) noexcept -> background_icosahedron& {
190  const auto glapi = vc.gl_api();
191  const auto& [gl, GL] = glapi;
192  const auto radius{(camera.skybox_distance())};
193 
194  gl.clear(GL.color_buffer_bit);
195  gl.use_program(_prog);
196  glapi.set_uniform(_prog, _camera_loc, camera.matrix(vc));
197  glapi.set_uniform(_prog, _scale_loc, radius);
198  gl.disable(GL.depth_test);
199  gl.disable(GL.cull_face);
200  gl.bind_vertex_array(_vao);
201  gl.polygon_mode(GL.front_and_back, GL.fill);
202  glapi.set_uniform(_prog, _color_loc, _fcolor);
203  draw_using_instructions(glapi, view(_ops));
204  gl.polygon_mode(GL.front_and_back, GL.line);
205  glapi.set_uniform(_prog, _color_loc, _ecolor);
206  draw_using_instructions(glapi, view(_ops));
207  gl.polygon_mode(GL.front_and_back, GL.fill);
208  gl.enable(GL.cull_face);
209  gl.enable(GL.depth_test);
210  gl.clear(GL.depth_buffer_bit);
211  return *this;
212 }
