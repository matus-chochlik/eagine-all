127  /// @brief Updates the internal state of the connection (called repeatedly).

131  return {};
132  }
133 
134  /// @brief Cleans up the connection before destroying it.
135  virtual void cleanup() noexcept {}
136 
137  /// @brief Checks if the connection is in usable state.
138  virtual auto is_usable() noexcept -> bool {
139  return true;
140  }
141 
142  /// @brief Returns the maximum data block size in bytes that can be sent.
143  virtual auto max_data_size() noexcept -> valid_if_positive<span_size_t> {
144  return {0};
145  }
146 
147  /// @brief Sent a message with the specified id.
148  /// @see fetch_messages

150  virtual auto send(const message_id msg_id, const message_view&) noexcept
151  -> bool = 0;
152

157  -> work_done = 0;
158

161 };
