19 /// @brief Class executing a specified action on scope exit.
20 /// @ingroup functional
21 /// @see func_on_scope_exit
22 /// @see finally
23 template <typename OnException = nothing_t>
24 class on_scope_exit {
25 public:
26  /// @brief The callable action type.
27  using action_type = callable_ref<void()>;
28 
29  /// @brief Construction intializing with the specified action.
30  on_scope_exit(action_type action) noexcept
31  : _action{std::move(action)} {}
32 
33  /// @brief Construction intializing with the specified action.
34  template <typename Func>
35  on_scope_exit(Func& action) noexcept
36  : _action{construct_from, action} {}
37 
38  /// @brief Move constructor.
39  on_scope_exit(on_scope_exit&& temp) noexcept
40  : on_scope_exit{temp.release()} {}
41 
42  /// @brief Not copy constructible.
43  on_scope_exit(const on_scope_exit&) = delete;
44 
45  /// @brief Not move constructible.
46  auto operator=(on_scope_exit&&) = delete;
47 
48  /// @brief Not copy constructible.
49  auto operator=(const on_scope_exit&) = delete;
50 
51  /// @brief Invokes the stored action, unless it was released or cancelled.
52  /// @see release
53  /// @see cancel
54  ~on_scope_exit() noexcept(false) {
55  _invoke(OnException());
56  }
57 
58  /// @brief Indicates if there is any on-scope-exit action.
59  /// @see release
60  /// @see cancel
61  auto is_active() const noexcept -> bool {
62  return _action.is_valid();
63  }
64 
65  /// @brief Moves out and returns the stored action.
66  /// @post !is_active()
67  /// @see cancel
68  /// @see is_active
69  auto release() noexcept -> action_type {
70  return std::move(_action);
71  }
72 
73  /// @brief Resets the stored action to empty default.
74  /// @post !is_active()
75  /// @see release
76  /// @see is_active
77  void cancel() noexcept {
78  _action = {};
79  }
80 
81 private:
82  action_type _action{};
83 
84  void _invoke(const std::true_type) const {
85  if(_action) {
86  if(std::uncaught_exceptions()) {
87  try {
88  _action();
89  } catch(...) {
90  }
91  }
92  }
93  }
94 
95  void _invoke(const std::false_type) const {
96  if(_action) {
97  if(!std::uncaught_exceptions()) {
98  _action();
99  }
100  }
101  }
102 
103  void _invoke(const nothing_t) const {
104  if(_action) {
105  _action();
106  }
107  }
108 };

110 /// @brief Class storing a callable object and an instance of on_scope_exit.
111 /// @ingroup functional
112 /// @tparam Func type of the callable function object.
113 /// @see on_scope_exit
114 /// @see finally
115 ///
116 /// This class stores a copy of the specified callable object and an instance
117 /// of on_scope_exit referencing and invoking the callable when desctoyed.
118 ///
119 /// @note Do not use directly, use the finally function to construct.
120 template <typename Func, typename OnException = nothing_t>
121 class func_on_scope_exit {
122 public:
123  /// @brief Initialization from the specified callable object.
124  func_on_scope_exit(Func func) noexcept
125  : _func{std::move(func)}
126  , _ose{_func} {}
127 
128  /// @brief Cancels this on scope exit action.
129  void cancel() noexcept {
130  _ose.cancel();
131  }
132 
133 private:
134  Func _func;
135  on_scope_exit<OnException> _ose;
136 };

138 /// @brief Function constructing on-scope-exit actions.
139 /// @ingroup functional
140 ///
141 /// This function takes the specified callable argument and creates an object
142 /// which when destroyed, typically when leaving scope block, invokes the callable.
143 template <typename Func>
144 [[nodiscard]] static inline auto finally(Func func) noexcept
145  -> func_on_scope_exit<Func> {
146  return func;
147 }
