16 /// @brief Basic template for lines in N-dimensional space.
17 /// @ingroup math
18 template <typename T, int N, bool V>
19 class basic_line {
20 public:
21  /// @brief Default constructor.
22  constexpr basic_line() noexcept = default;
23 
24  /// @brief Construction from origin point and direction vector.
25  constexpr basic_line(
26  const tvec<T, N, V> orig,
27  const tvec<T, N, V> dir) noexcept
28  : _origin{orig}
29  , _direction{dir} {}
30 
31  /// @brief Returns the line origin.
32  constexpr auto origin() const noexcept -> vector<T, N, V> {
33  return _origin;
34  }
35 
36  /// @brief Returns the line direction.
37  constexpr auto direction() const noexcept -> vector<T, N, V> {
38  return _direction;
39  }
40 
41  /// @brief Returns a point on the line at the specified parameter @p t.
42  constexpr auto point_at(T t) const noexcept -> vector<T, N, V> {
43  return _origin + _direction * t;
44  }
45 
46 private:
47  vector<T, N, V> _origin{};
48  vector<T, N, V> _direction{};
49 };

56 /// @brief Basic template for triangles in N-dimensional space.
57 /// @ingroup math
58 template <typename T, int N, bool V>
59 class basic_triangle {
60 public:
61  /// @brief Default constructor.
62  constexpr basic_triangle() noexcept = default;
63 
64  /// @brief Construction from three points making up the triangle.
65  constexpr basic_triangle(
66  const tvec<T, N, V> a,
67  const tvec<T, N, V> b,
68  const tvec<T, N, V> c) noexcept
69  : _vertices{{a, b, c}} {}
70 
71  /// @brief Returns the point at the specified index.
72  /// @pre index < 3
73  constexpr auto vertex(const span_size_t index) const noexcept
74  -> vector<T, N, V> {
75  return _vertices[index];
76  }
77 
78  /// @brief Returns the first point of the triangle.
79  constexpr auto a() const noexcept -> vector<T, N, V> {
80  return _vertices[0];
81  }
82 
83  /// @brief Returns the second point of the triangle.
84  constexpr auto b() const noexcept -> vector<T, N, V> {
85  return _vertices[1];
86  }
87 
88  /// @brief Returns the third point of the triangle.
89  constexpr auto c() const noexcept -> vector<T, N, V> {
90  return _vertices[2];
91  }
92 
93  /// @brief Returns the ab edge vector of the triangle.
94  constexpr auto ab() const noexcept -> vector<T, N, V> {
95  return b() - a();
96  }
97 
98  /// @brief Returns the ac edge vector of the triangle.
99  constexpr auto ac() const noexcept -> vector<T, N, V> {
100  return c() - a();
101  }
102 
103  /// @brief Returns the center of the triangle.
104  constexpr auto center() const noexcept -> vector<T, N, V> {
105  return (a() + b() + c()) / T(3);
106  }
107 
108  /// @brief Returns the normal vector of the triangle, in specified direction.
109  constexpr auto normal(bool cw) const noexcept -> vector<T, N, V> {
110  return cw ? cross(ac(), ab()) : cross(ab(), ac());
111  }
112 
113  /// @brief Returns the area of the triangle.
114  constexpr auto area() const noexcept -> T {
115  return length(cross(ab(), ac())) / T(2);
116  }
117 
118 private:
119  std::array<vector<T, N, V>, 3> _vertices{};
120 };

128 /// @ingroup math
129 template <typename T, int N, bool V>
130 class basic_sphere {
131 public:
132  /// @brief Default constructor.
133  constexpr basic_sphere() noexcept = default;
134 
135  /// @brief Construction from center point and radius value.
136  constexpr basic_sphere(const tvec<T, N, V> cntr, const T rad) noexcept
137  : _params{vector<T, N + 1, V>::from(cntr, rad)} {}
138 
139  /// @brief Returns the sphere center.
140  constexpr auto center() const noexcept -> vector<T, N, V> {
141  return vector<T, N, V>::from(_params);
142  }
143 
144  /// @brief Returns the sphere radius.
145  constexpr auto radius() const noexcept -> T {
146  return _params[N];
147  }
148 
149 private:
150  vector<T, N + 1, V> _params{};
151 };
