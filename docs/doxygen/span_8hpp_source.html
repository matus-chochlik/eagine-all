26 /// @brief Helper function for pretty-printing spans as lists into output streams.
27 /// @ingroup memory
28 /// @see write_to_stream
29 /// @relates memory::basic_span
30 template <typename T, typename P, typename S, typename Output>
31 static inline auto list_to_stream(
32  Output& out,
33  const memory::basic_span<T, P, S> s) -> Output& {
34  out << '[';
35  bool first = true;
36  for(const auto& e : s) {
37  if(first) {
38  first = false;
39  } else {
40  out << ',';
41  }
42  out << e;
43  }
44  return out << ']';
45 }

96 /// @brief Makes a callable that returns consecutive span elements starting at i.
97 /// @ingroup memory
98 /// @see make_span_putter
99 /// @relates memory::basic_span
100 ///
101 /// The constructed callable object does not take any arguments in the call
102 /// operator and returns optional values of T.
103 template <typename T, typename P, typename S>
104 static inline auto make_span_getter(
105  span_size_t& i,
106  const memory::basic_span<T, P, S> spn) {
107  return [&i, spn]() -> optionally_valid<std::remove_const_t<T>> {
108  if(i < spn.size()) {
109  return {spn[i++], true};
110  }
111  return {};
112  };
113 }
114 //------------------------------------------------------------------------------
115 template <typename Src>
116 static inline auto make_span_getter(span_size_t& i, const Src& src) {
117  return make_span_getter(i, view(src));
118 }
119 //------------------------------------------------------------------------------
120 /// @brief Makes a callable getting consecutive, transformed span elements starting at i.
121 /// @ingroup memory
122 /// @see make_span_putter
123 /// @relates memory::basic_span
124 ///
125 /// The constructed callable object does not take any arguments in the call
126 /// operator and returns the result of the transform function.
127 /// The transform function takes a single optional value of T.
128 template <typename T, typename P, typename S, typename Transform>
129 static inline auto make_span_getter(
130  span_size_t& i,
131  const memory::basic_span<T, P, S> spn,
132  const Transform transform) {
133  return [&i, spn, transform]() -> decltype(transform(std::declval<T>())) {
134  if(i < spn.size()) {
135  return transform(spn[i++]);
136  }
137  return {};
138  };
139 }
140 //------------------------------------------------------------------------------
141 template <typename Src, typename Transform>
142 static inline auto make_span_getter(
143  span_size_t& i,
144  const Src& src,
145  Transform transform) {
146  return make_span_getter(i, view(src), std::move(transform));
147 }
148 //------------------------------------------------------------------------------
149 /// @brief Makes a callable setting consecutive elements of a span starting at i.
150 /// @ingroup memory
151 /// @see make_span_getter
152 /// @relates memory::basic_span
153 ///
154 /// The constructed callable takes a single value explicitly convertible to T.
155 template <typename T, typename P, typename S>
156 static inline auto make_span_putter(
157  span_size_t& i,
158  memory::basic_span<T, P, S> spn) {
159  return [&i, spn](auto value) mutable -> bool {
160  if(i < spn.size()) {
161  spn[i++] = T(std::move(value));
162  return true;
163  }
164  return false;
165  };
166 }
167 //------------------------------------------------------------------------------
168 template <typename Dst>
169 static inline auto make_span_putter(span_size_t& o, Dst& dst) {
170  return make_span_putter(o, cover(dst));
171 }
172 //------------------------------------------------------------------------------
173 /// @brief Makes a callable setting consecutive elements of a span starting at i.
174 /// @ingroup memory
175 /// @param transform transformation operation to be applied
176 /// @see make_span_getter
177 /// @relates memory::basic_span
178 /// The constructed callable takes a single value explicitly convertible to the
179 /// argument of the transform function.
180 template <typename T, typename P, typename S, typename Transform>
181 static inline auto make_span_putter(
182  span_size_t& i,
183  memory::basic_span<T, P, S> spn,
184  Transform transform) {
185  return [&i, spn, transform](auto value) mutable -> bool {
186  if(i < spn.size()) {
187  if(auto transformed{transform(value)}) {
188  spn[i++] = T(std::move(extract(transformed)));
189  return true;
190  }
191  }
192  return false;
193  };
194 }
195 //------------------------------------------------------------------------------
196 template <typename Dst, typename Transform>
197 static inline auto make_span_putter(
198  span_size_t& o,
199  Dst& dst,
200  Transform transform) {
201  return make_span_putter(o, cover(dst), std::move(transform));
202 }
