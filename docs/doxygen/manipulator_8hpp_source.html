114 template <typename Component>
115 class manipulator

117  std::remove_const_t<Component>,
118  std::is_const_v<Component>> {

141  manipulator(std::nullptr_t, _nonconstC& add, const bool can_rem)
142  : _add_place{&add}
143  , _can_rem{can_rem} {}
144 
145  auto can_add_component() const noexcept -> bool {
146  return _add_place != nullptr;
147  }
148 
149  void add_component(std::remove_const_t<Component>&& cmp) {
150  EAGINE_ASSERT(can_add_component());
151  EAGINE_ASSERT(_add_place);
152  *_add_place = std::move(cmp);
153  this->_reset_cmp(*_add_place);
154  _added = true;
155  }
156 
157  auto can_remove() const noexcept -> bool {
158  return _can_rem && this->is_valid();
159  }
160 
161  void remove() {
162  _removed = true;
163  }
164 
165 protected:
166  const bool _can_rem{false};
167  bool _removed{false};
168  bool _added{false};
169 
170 private:
171  using _base = get_manipulator_t<
172  std::remove_const_t<Component>,
173  std::is_const_v<Component>>;
174 };
