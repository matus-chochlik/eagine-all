21 using shutdown_service_duration =
22  std::chrono::duration<std::int64_t, std::milli>;
23 //------------------------------------------------------------------------------
24 /// @brief Service allowing an endpoint to be shut down over the message bus.
25 /// @ingroup msgbus
26 /// @see service_composition
27 /// @see shutdown_invoker
28 template <typename Base = subscriber>
29 class shutdown_target
30  : public Base
31  , protected shutdown_service_clock {
32  using This = shutdown_target;
33 
34 public:
35  /// @brief Triggered when a shutdown request is received.
36  signal<void(
37  const std::chrono::milliseconds age,
38  const identifier_t source_id,
39  const verification_bits verified) noexcept>
40  shutdown_requested;
41 
42 protected:
43  using Base::Base;
44 
45  void add_methods() noexcept {
46  Base::add_methods();
47  Base::add_method(
48  this, EAGINE_MSG_MAP(Shutdown, shutdown, This, _handle_shutdown));
49  }
50 
51 private:
52  auto _handle_shutdown(
53  const message_context&,
54  const stored_message& message) noexcept -> bool {
55  typename shutdown_service_duration::rep count{0};
56  if(default_deserialize(count, message.content())) {
57  const shutdown_service_duration ticks{count};
58  const typename shutdown_service_clock::time_point ts{ticks};
59  const auto age{this->now() - ts};
60  shutdown_requested(
61  std::chrono::duration_cast<std::chrono::milliseconds>(age),
62  message.source_id,
63  this->verify_bits(message));
64  }
65  return true;
66  }
67 };
68 //------------------------------------------------------------------------------
69 /// @brief Service allowing to shut down other endpoints over the message bus.
70 /// @ingroup msgbus
71 /// @see service_composition
72 /// @see shutdown_target
73 template <typename Base = subscriber>
74 class shutdown_invoker
75  : public Base
76  , protected shutdown_service_clock {
77 
78  using This = shutdown_invoker;
79 
80 public:
81  /// @brief Sends shutdown request to the specified target endpoint.
82  void shutdown_one(const identifier_t target_id) noexcept {
83  std::array<byte, 32> temp{};
84  const auto ts{this->now()};
85  const auto ticks{std::chrono::duration_cast<shutdown_service_duration>(
86  ts.time_since_epoch())};
87  const auto count{ticks.count()};
88  auto serialized{default_serialize(count, cover(temp))};
89  EAGINE_ASSERT(serialized);
90 
91  message_view message{extract(serialized)};
92  message.set_target_id(target_id);
93  this->bus_node().post_signed(
94  EAGINE_MSG_ID(Shutdown, shutdown), message);
95  }
96 
97 protected:
98  using Base::Base;
99 };
