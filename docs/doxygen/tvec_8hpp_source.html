18 /// @brief Generic template for N-dimensional vectors.
19 /// @ingroup math
20 template <typename T, int N, bool V>
21 struct tvec : vector<T, N, V> {
22  /// @brief The base vector type.
23  using base = vector<T, N, V>;
24 
25  /// @brief Default constructor. Constructs a zero vector.
26  constexpr tvec() noexcept
27  : base{base::zero()} {}
28 
29  /// @brief Constructor initializing all coordinates to @p v.
30  constexpr tvec(T v) noexcept
31  : base{base::fill(v)} {}
32 
33  /// @brief Construction from base vector.
34  constexpr tvec(const base& v) noexcept
35  : base{v} {}
36 
37  /// @brief Construction from native array.
38  tvec(const T (&d)[N]) noexcept
39  : base{base::from(static_cast<const T*>(d), N)} {}
40 
41  /// @brief Construction from coordinates.
42  template <typename... P>
43  constexpr tvec(P&&... p) noexcept
44  requires((sizeof...(P) == N) && all_are_convertible_to<T, P...>::value)
45  : base{base::make(std::forward<P>(p)...)} {}
46 
47  /// @brief Construction from vector of different dimensionality.
48  template <typename P, int M, bool W>
49  constexpr tvec(const vector<P, M, W>& v) noexcept
50  requires(!std::is_same_v<P, T> || !(M == N))
51  : base{base::from(v)} {}
52 
53  /// @brief Construction from vector of different dimensionality.
54  template <typename P, int M, bool W>
55  constexpr tvec(const vector<P, M, W>& v, const T d) noexcept
56  : base{base::from(v, d)} {}
57 
58  /// @brief Construction from vector of different dimensionality.
59  template <typename P, int M, bool W, typename... R>
60  constexpr tvec(const vector<P, M, W>& v, R&&... r) noexcept requires(
61  (sizeof...(R) > 1) && (M + sizeof...(R) == N) &&
62  all_are_convertible_to<T, R...>::value)
63  : base{base::from(v, vector<T, N - M, W>::make(std::forward<R>(r)...))} {}
64 
65  /// @brief Construction from a pair of vectors of different dimensionality.
66  template <typename P, int M, bool W>
67  constexpr tvec(
68  const vector<P, M, W>& v,
69  const vector<T, N - M, W>& w) noexcept
70  : base{base::from(v, w)} {}
71 };

99  template <typename Pd, typename Sd>
100  static auto apply(
101  const math::tvec<T, N, V>& src,
102  memory::basic_span<T, Pd, Sd> dst) noexcept {
103  EAGINE_ASSERT(N <= dst.size());
104  _do_apply(src._v, dst, std::make_index_sequence<std::size_t(N)>{});
105  return skip(dst, N);
106  }
107 
108 private:
109  template <typename Pd, typename Sd, std::size_t... I>
110  static void _do_apply(
111  const vect::data_t<T, N, V> src,
112  memory::basic_span<T, Pd, Sd> dst,
113  std::index_sequence<I...>) noexcept {
114  ((dst[I] = src[I]), ...);
115  }
116 };
