16 /// @brief Policy for values valid if between Min and Max.
17 /// @tparam T the checked value type must be convertible to @p C.
18 /// @tparam C the Min and Max constant type.
19 /// @tparam Min the minimal valid value.
20 /// @tparam Max the maximal valid value.
21 /// @ingroup valid_if
22 template <typename T, typename C, C Min, C Max>
23 struct valid_if_btwn_policy {
24 
25  /// @brief Indicates @p value validity, true if between Min and Max.
26  constexpr auto operator()(const T value) const noexcept {
27  return (T(Min) <= value) && (value <= T(Max));
28  }
