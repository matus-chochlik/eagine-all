55  /// @brief Returns the set of vertex attributes supported by this generator.

59  /// @brief Tests if the specified attribute is supported by this generator.
60  /// @see attrib_kinds
61  /// @see attrib_count
62  auto has(const vertex_attrib_kind attrib) noexcept {
63  return bool(attrib_kinds() | attrib);
64  }
65 
66  /// @brief Returns the count of vertex attributes supported by this generator.
67  /// @see attrib_kinds
68  /// @see for_each_attrib
69  auto supported_attrib_count() noexcept -> span_size_t {
70  return count_set_bits(attrib_kinds());
71  }
72 
73  /// @brief Calls the specified function for each attribute kind.
74  /// @see attrib_kinds
75  /// @see supported_attrib_count
76  void for_each_attrib(
77  callable_ref<void(vertex_attrib_kinds, vertex_attrib_name_and_kind)> func) {
78  for_each_bit(func, attrib_kinds());
79  }

277  /// @brief Checks if the structure for random values is consistent.
278  /// @see random_attribute_values
279  auto are_consistent(const random_attribute_values& values) noexcept
280  -> bool {
281  for(const auto& [vav, dest] : values.float_values) {
282  if(dest.size() < values.count * values_per_vertex(vav)) {
283  return false;
284  }
285  }
286  return true;
287  }

325 /// @brief Common base implementation of the shape generator interface.
326 /// @ingroup shapes
327 class generator_base : public generator {
328 public:
329  auto attrib_kinds() noexcept -> vertex_attrib_kinds final {
330  return _attr_kinds;
331  }
332 
333  auto enable(const generator_capability cap, const bool value) noexcept
334  -> bool final {
335  bool result{true};
336  if(value) {
337  if(_supported_caps.has(cap)) {
338  _enabled_caps.set(cap);
339  } else {
340  result = false;
341  }
342  } else {
343  _enabled_caps.clear(cap);
344  }
345  return result;
346  }
347 
348  auto is_enabled(const generator_capability cap) noexcept -> bool final {
349  return _enabled_caps.has(cap);
350  }
351 
352  auto attribute_variants(const vertex_attrib_kind attrib)
353  -> span_size_t override {
354  return has(attrib) ? 1 : 0;
355  }
356 
357  auto variant_name(const vertex_attrib_variant) -> string_view override {
358  return {};
359  }
360 
361  auto values_per_vertex(const vertex_attrib_variant vav)
362  -> span_size_t override {
363  return has_variant(vav) ? attrib_values_per_vertex(vav) : 0U;
364  }
365 
366  auto attrib_type(const vertex_attrib_variant) -> attrib_data_type override {
367  return attrib_data_type::float_;
368  }
369 
370  auto is_attrib_integral(const vertex_attrib_variant) -> bool override {
371  return false;
372  }
373 
374  auto is_attrib_normalized(const vertex_attrib_variant) -> bool override {
375  return false;
376  }
377 
378  void attrib_values(const vertex_attrib_variant, span<byte>) override {
379  EAGINE_UNREACHABLE("Generator failed to get byte attribute values.");
380  }
381 
382  void attrib_values(const vertex_attrib_variant, span<std::int16_t>)
383  override {
384  EAGINE_UNREACHABLE("Generator failed to get int16 attribute values.");
385  }
386 
387  void attrib_values(const vertex_attrib_variant, span<std::int32_t>)
388  override {
389  EAGINE_UNREACHABLE("Generator failed to get int32 attribute values.");
390  }
391 
392  void attrib_values(const vertex_attrib_variant, span<std::uint16_t>)
393  override {
394  EAGINE_UNREACHABLE("Generator failed to get uint16 attribute values.");
395  }
396 
397  void attrib_values(const vertex_attrib_variant, span<std::uint32_t>)
398  override {
399  EAGINE_UNREACHABLE("Generator failed to get uint32 attribute values.");
400  }
401 
402  void attrib_values(const vertex_attrib_variant, span<float>) override {
403  EAGINE_UNREACHABLE("Generator failed to get float attribute values.");
404  }
405 
406  auto draw_variant_count() -> span_size_t override {
407  return 1;
408  }
409 
410  auto index_type(const drawing_variant) -> index_data_type override;
411 
412  auto index_count(const drawing_variant) -> span_size_t override;
413 
414  void indices(const drawing_variant, span<std::uint8_t> dest) override;
415 
416  void indices(const drawing_variant, span<std::uint16_t> dest) override;
417 
418  void indices(const drawing_variant, span<std::uint32_t> dest) override;
419 
420 protected:
421  generator_base(
422  const vertex_attrib_kinds attr_kinds,
423  const generator_capabilities supported_caps) noexcept
424  : _attr_kinds{attr_kinds}
425  , _supported_caps{supported_caps} {}
426 
427 private:
428  vertex_attrib_kinds _attr_kinds;
429  const generator_capabilities _supported_caps;
430  generator_capabilities _enabled_caps{_supported_caps};
431 };
