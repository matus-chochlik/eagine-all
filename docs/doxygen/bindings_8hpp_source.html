19 /// @brief Interface for bindings between attribute variant and vertex array index.
20 /// @ingroup shapes
21 /// @see vertex_attrib_bindings
22 /// @note Do not use directly. Use vertex_attrib_bindings instead.
23 struct vertex_attrib_binding_intf : interface<vertex_attrib_binding_intf> {
24 
25  /// @brief Returns the number of attributes in the binding.
26  /// @see attrib_variant
27  virtual auto attrib_count() -> span_size_t = 0;
28 
29  /// @brief Returns the attribute variant bound to specified index.
30  /// @see attrib_count
31  /// @see location
32  /// @pre index < attrib_count()
33  virtual auto attrib_variant(span_size_t index)
34  -> shapes::vertex_attrib_variant = 0;
35 
36  /// @brief Returns the index at which the specified attribute variant is bound.
37  /// @see attrib_variant
38  virtual auto location(shapes::vertex_attrib_variant vav)
39  -> vertex_attrib_location = 0;
40 };

42 class default_vertex_attrib_bindings : public vertex_attrib_binding_intf {
43 public:
44  default_vertex_attrib_bindings() noexcept = default;
45 
46  default_vertex_attrib_bindings(const shape_generator& shape) {
47  shape.for_each_attrib(
48  {construct_from, [this](const auto attribs, const auto info) {
49  if(attribs.has(info.enumerator)) {
50  _bindings.emplace_back(info.enumerator, 0);
51  }
52  }});
53  }
54 
55  default_vertex_attrib_bindings(
56  std::initializer_list<shapes::vertex_attrib_variant> vavs)
57  : _bindings{vavs.begin(), vavs.end()} {}
58 
59  auto add(shapes::vertex_attrib_variant vav) -> auto& {
60  _bindings.push_back(vav);
61  return *this;
62  }
63 
64  auto attrib_count() -> span_size_t final {
65  return span_size(_bindings.size());
66  }
67 
68  auto attrib_variant(span_size_t index)
69  -> shapes::vertex_attrib_variant final {
70  return _bindings[integer(index)];
71  }
72 
73  auto location(shapes::vertex_attrib_variant vav)
74  -> vertex_attrib_location final {
75  for(const integer i : integer_range(_bindings.size())) {
76  const auto& entry = _bindings[i];
77  if(entry == vav) {
78  return vertex_attrib_location{i};
79  }
80  }
81  return vertex_attrib_location{0};
82  }
83 
84 private:
85  std::vector<shapes::vertex_attrib_variant> _bindings;
86 };

129  /// @brief Returns the number of attributes in the binding.
130  /// @see attrib_variant
131  auto attrib_count() const -> span_size_t {
132  if(_pimpl) {
133  return _pimpl->attrib_count();
134  }
135  return 0;
136  }

138  /// @brief Returns the attribute variant bound to specified index.
139  /// @see attrib_count
140  /// @see location
141  /// @pre index < attrib_count()
142  auto attrib_variant(span_size_t index) const
143  -> shapes::vertex_attrib_variant {
144  if(_pimpl) {
145  EAGINE_ASSERT(index < attrib_count());
146  return _pimpl->attrib_variant(index);
147  }
148  return {};
149  }
150 
151  /// @brief Returns the attribute kind bits for all attributes in this binding.
152  auto attrib_kinds() const -> shapes::vertex_attrib_kinds {
153  shapes::vertex_attrib_kinds result;
154  for(const auto i : integer_range(attrib_count())) {
155  result.set(attrib_variant(i).attribute());
156  }
157  return result;
158  }
