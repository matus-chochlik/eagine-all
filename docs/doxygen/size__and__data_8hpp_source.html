18 /// @brief Encodes the size of the source block into destination, copies data afterwards.
19 /// @ingroup serialization
20 /// @see get_data_with_size
21 static inline auto store_data_with_size(
22  const memory::const_block src,
23  memory::block dst) noexcept -> memory::block {
24 
25  const auto opt_size_cp = limit_cast<mbs::code_point>(src.size());
26  if(opt_size_cp) [[likely]] {
27  const auto size_cp = extract(opt_size_cp);
28  const auto opt_size_len = mbs::required_sequence_length(size_cp);
29  if(opt_size_len) [[likely]] {
30  if(extract(opt_size_len) + src.size() <= dst.size()) {
31  const auto opt_skip_len = mbs::encode_code_point(size_cp, dst);
32  if(opt_skip_len) [[likely]] {
33  const auto skip_len =
34  limit_cast<span_size_t>(extract(opt_skip_len));
35  copy(src, skip(dst, skip_len));
36  return head(dst, skip_len + src.size());
37  }
38  }
39  }
40  }
41 
42  return {};
43 }
44 //------------------------------------------------------------------------------
45 /// @brief In a block starting with sub-block with size returns the size of the sub-block.
46 /// @ingroup serialization
47 /// @see store_data_with_size
48 /// @see get_data_with_size
49 static inline auto skip_data_with_size(const memory::const_block src) noexcept
50  -> span_size_t {
51  const auto opt_skip_len = mbs::decode_sequence_length(src);
52  if(const auto opt_data_len = mbs::do_decode_code_point(src, opt_skip_len)) {
53  return extract(opt_skip_len) + extract(opt_data_len);
54  }
55  return 0;
56 }

58 /// @brief Extracts a sub-block from a larger mutable block with encoded sub-block size.
59 /// @ingroup serialization
60 /// @see store_data_with_size
61 static inline auto get_data_with_size(const memory::block src) noexcept
62  -> memory::block {
63  const memory::const_block tmp{src};
64  const auto opt_skip_len = mbs::decode_sequence_length(tmp);
65  if(const auto opt_data_len = mbs::do_decode_code_point(tmp, opt_skip_len)) {
66  return head(
67  skip(src, extract(opt_skip_len)),
68  limit_cast<span_size_t>(extract(opt_data_len)));
69  }
70  return {};
71 }

77  -> memory::const_block {

86 //------------------------------------------------------------------------------
87 /// @brief In a larger block with sub-blocks with size, calls function on each sub-block.
88 /// @ingroup serialization
89 /// @see store_data_with_size
90 /// @see get_data_with_size
91 template <typename Function>
92 static inline void for_each_data_with_size(
93  memory::const_block src,
94  Function function) noexcept {
95  while(src) {
96  const auto opt_skip_len = mbs::decode_sequence_length(src);
97  const auto opt_data_len = mbs::do_decode_code_point(src, opt_skip_len);
98  if(opt_data_len) {
99  const auto skip_len = extract(opt_skip_len);
100  const auto data_len = extract(opt_data_len);
101  if(data_len > 0) {
102  function(
103  head(skip(src, skip_len), limit_cast<span_size_t>(data_len)));
104  src = skip(src, skip_len + data_len);
105  } else {
106  break;
107  }
108  } else {
109  break;
110  }
111  }
112 }
