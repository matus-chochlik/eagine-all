36 //------------------------------------------------------------------------------
37 /// @brief Class wrapping a constant with missing or unknown value.
38 /// @ingroup c_api_wrap
39 template <typename T, typename Tag = nothing_t, bool isIndexed = false>
40 struct no_constant
41  : constant_base
42  , no_enum_value<T, Tag> {
43 public:
44  template <typename ApiTraits, typename Api>
45  constexpr no_constant(const string_view name, ApiTraits&, Api&) noexcept
46  : constant_base{name} {}
47 
48  /// @brief Adds the specified value to the constant (if it isIndexed).
49  template <std::integral I>
50  constexpr auto operator+(const I) const noexcept
51  -> no_enum_value<T, Tag> requires(isIndexed) {
52  return {};
53  }
54 };
55 //------------------------------------------------------------------------------
56 /// @brief Class wrapping a constant with value known at compile-time.
57 /// @ingroup c_api_wrap
58 template <
59  typename ClassList,
60  typename T,
61  T value,
62  typename Tag = nothing_t,
63  bool isIndexed = false>
64 struct static_constant
65  : constant_base
66  , enum_value<T, ClassList, Tag> {
67 public:
68  template <typename ApiTraits, typename Api>
69  constexpr static_constant(string_view name, ApiTraits&, Api&) noexcept
70  : constant_base{name}
71  , enum_value<T, ClassList, Tag>{value} {}
72 
73  /// @brief Adds the specified value to the constant (if it isIndexed).
74  template <typename I>
75  constexpr auto operator+(const I index) const noexcept
76  -> enum_value<T, ClassList, Tag> requires(
77  isIndexed&& std::is_integral_v<I>) {
78  using O = std::conditional_t<
79  std::is_signed_v<T>,
80  std::make_signed_t<I>,
81  std::make_unsigned_t<I>>;
82  using eagine::limit_cast;
83  return {limit_cast<T>(value + limit_cast<O>(index))};
84  }
85 };
86 //------------------------------------------------------------------------------
87 /// @brief Class wrapping a constant with value loaded at run-time.
88 /// @ingroup c_api_wrap
89 template <
90  typename ClassList,
91  typename T,
92  typename Tag = nothing_t,
93  bool isIndexed = false>
94 struct dynamic_constant
95  : constant_base
96  , opt_enum_value<T, ClassList, Tag> {
97 public:
98  using tag_type = Tag;
99 
100  template <typename ApiTraits, typename Api>
101  constexpr dynamic_constant(
102  const string_view name,
103  ApiTraits& traits,
104  Api& api) noexcept
105  : constant_base{name}
106  , opt_enum_value<T, ClassList, Tag>{
107  traits.load_constant(api, name, type_identity<T>())} {}
108 
109  /// @brief Adds the specified value to the constant (it it isIndexed).
110  template <typename I>
111  constexpr auto operator+(const I index) const noexcept
112  -> opt_enum_value<T, ClassList, Tag> requires(
113  isIndexed&& std::is_integral_v<I>) {
114  using O = std::conditional_t<
115  std::is_signed_v<T>,
116  std::make_signed_t<I>,
117  std::make_unsigned_t<I>>;
118  return {limit_cast<T>(this->value + limit_cast<O>(index))};
119  }
120 };

122 template <typename ClassList, typename Constant, typename Tag, bool isIndexed>
123 struct get_opt_constant;

125 template <typename ClassList, typename T, T value, typename Tag, bool isIndexed>
126 struct get_opt_constant<ClassList, std::integral_constant<T, value>, Tag, isIndexed>
127  : type_identity<static_constant<ClassList, T, value, Tag, isIndexed>> {};
128 
129 template <typename ClassList, typename T, typename Tag, bool isIndexed>
130 struct get_opt_constant<ClassList, type_identity<T>, Tag, isIndexed>
131  : type_identity<dynamic_constant<ClassList, T, Tag, isIndexed>> {};

133 /// @brief Template alias used for switching between static and dynamic constants.
134 /// @tparam ClassList a list of enum_class types into which the constant can
135 /// be converted.
136 /// @tparam Constant integral constant specifying the value or a placeholder.
137 /// @tparam Tag a tag type that can be used in custiomization of some operations
138 /// on the constant.
139 /// @tparam isIndexed indicates if the constant is indexed, which enables
140 /// additional operations on the constants.
141 /// @ingroup c_api_wrap
142 ///
143 /// @see no_constant
144 /// @see static_constant
145 /// @see dynamic_constant
146 /// @see enum_class
147 /// @see enum_value
148 /// @see mp_list
149 ///
150 /// C-API constants can be either known at compile time or loaded dynamically
151 /// by using some API function(s).
152 template <
153  typename ClassList,
154  typename Constant,
155  typename Tag = nothing_t,
