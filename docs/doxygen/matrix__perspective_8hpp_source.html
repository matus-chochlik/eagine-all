25 /// @brief Implements constructor of perspective projection matrix.
26 /// @ingroup math
27 /// @see matrix_perspective
28 ///
29 /// @note Do not use directly, use matrix_perspective.
30 template <typename T, bool RM, bool V>
31 class perspective<matrix<T, 4, 4, RM, V>> {
32 public:
33  constexpr perspective(const vect::data_t<T, 6, V>& v) noexcept
34  : _v{v} {}
35 
36  /// @brief Initialized the matrix constructor.
37  constexpr perspective(
38  const T x_left,
39  const T x_right,
40  const T y_bottom,
41  const T y_top,
42  const T z_near,
43  const T z_far) noexcept
44  : _v{x_left, x_right, y_bottom, y_top, z_near, z_far} {}
45 
46  /// @brief Returns the constructed matrix.
47  constexpr auto operator()() const noexcept {
48  return _make(bool_constant<RM>());
49  }
50 
51  /// @brief Constructs perspective matrix with x-FOV angle and aspect ratio.
52  static auto x(
53  const radians_t<T> xfov,
54  const T aspect,
55  const T z_near,
56  const T z_far) noexcept {
57  EAGINE_ASSERT(aspect > T(0));
58  EAGINE_ASSERT(T(xfov) > T(0));
59 
60  const T x_right = z_near * tan(xfov * T(0.5));
61  const T x_left = -x_right;
62 
63  const T y_bottom = x_left / aspect;
64  const T y_top = x_right / aspect;
65 
66  return perspective(x_left, x_right, y_bottom, y_top, z_near, z_far);
67  }
68 
69  /// @brief Constructs perspective matrix with y-FOV angle and aspect ratio.
70  static auto y(
71  const radians_t<T> yfov,
72  const T aspect,
73  const T z_near,
74  const T z_far) noexcept {
75  EAGINE_ASSERT(aspect > T(0));
76  EAGINE_ASSERT(T(yfov) > T(0));
77 
78  const T y_top = z_near * tan(yfov * T(0.5));
79  const T y_bottom = -y_top;
80 
81  const T x_left = y_bottom * aspect;
82  const T x_right = y_top * aspect;
83 
84  return perspective(x_left, x_right, y_bottom, y_top, z_near, z_far);
85  }
86 
87  /// @brief Constructs perspective matrix with FOV angle and aspect ratio of 1.
88  static auto square(
89  const radians_t<T> fov,
90  const T z_near,
91  const T z_far) noexcept {
92  EAGINE_ASSERT(T(fov) > T(0));
93 
94  const T x_right = z_near * tan(fov * T(0.5));
95  const T x_left = -x_right;
96 
97  const T y_bottom = x_left;
98  const T y_top = x_right;
99 
100  return perspective(x_left, x_right, y_bottom, y_top, z_near, z_far);
101  }
102 
103 private:
104  constexpr auto _x_left() const noexcept {
105  return _v[0];
106  }
107 
108  constexpr auto _x_right() const noexcept {
109  return _v[1];
110  }
111 
112  constexpr auto _y_bottom() const noexcept {
113  return _v[2];
114  }
115 
116  constexpr auto _y_top() const noexcept {
117  return _v[3];
118  }
119 
120  constexpr auto _z_near() const noexcept {
121  return _v[4];
122  }
123 
124  constexpr auto _z_far() const noexcept {
125  return _v[5];
126  }
127 
128  constexpr auto _m00() const noexcept {
129  return (T(2) * _z_near()) / (_x_right() - _x_left());
130  }
131 
132  constexpr auto _m11() const noexcept {
133  return (T(2) * _z_near()) / (_y_top() - _y_bottom());
134  }
135 
136  constexpr auto _m22() const noexcept {
137  return -(_z_far() + _z_near()) / (_z_far() - _z_near());
138  }
139 
140  constexpr auto _m20() const noexcept {
141  return (_x_right() + _x_left()) / (_x_right() - _x_left());
142  }
143 
144  constexpr auto _m21() const noexcept {
145  return (_y_top() + _y_bottom()) / (_y_top() - _y_bottom());
146  }
147 
148  constexpr auto _m23() const noexcept {
149  return -T(1);
150  }
151 
152  constexpr auto _m32() const noexcept {
153  return -(T(2) * _z_far() * _z_near()) / (_z_far() - _z_near());
154  }
155 
156  constexpr auto _make(const std::true_type) const noexcept {
157  return matrix<T, 4, 4, true, V>{
158  {{_m00(), T(0), _m20(), T(0)},
159  {T(0), _m11(), _m21(), T(0)},
160  {T(0), T(0), _m22(), _m32()},
161  {T(0), T(0), _m23(), T(0)}}};
162  }
163 
164  constexpr auto _make(const std::false_type) const noexcept {
165  return matrix<T, 4, 4, false, V>{
166  {{_m00(), T(0), T(0), T(0)},
167  {T(0), _m11(), T(0), T(0)},
168  {_m20(), _m21(), _m22(), _m23()},
169  {T(0), T(0), _m32(), T(0)}}};
170  }
171 
172  using _dT = vect::data_t<T, 6, V>;
173  _dT _v;
174 };
