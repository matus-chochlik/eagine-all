15 namespace eagine {

17 /// @brief Class tracking the progress of a long-running activity.
18 /// @ingroup progress
19 class activity_progress {
20 public:
21  activity_progress(
22  const activity_progress& parent,
23  const string_view title,
24  const span_size_t total_steps = 0) noexcept
25  : _backend{parent._backend}
26  , _activity_id{
27  _backend
28  ? _backend->begin_activity(parent._activity_id, title, total_steps)
29  : 0U} {}
30 
31  activity_progress(activity_progress&& temp) noexcept
32  : _backend{std::move(temp._backend)}
33  , _activity_id{temp._activity_id} {}
34  activity_progress(const activity_progress&) = delete;
35  auto operator=(activity_progress&&) = delete;
36  auto operator=(const activity_progress&) = delete;
37 
38  /// @brief Marks the activity as finished.
39  /// @see finish
40  ~activity_progress() noexcept {
41  finish();
42  }
43 
44  /// @brief Creates a sub-activity progress tracker with a title.
45  auto activity(const string_view title, const span_size_t total_steps = 0)
46  const noexcept -> activity_progress {
47  return {*this, title, total_steps};
48  }
49 
50  /// @brief Returns the unique id of this logger instance.
51  auto activity_id() const noexcept -> activity_progress_id_t {
52  return _activity_id;
53  }
54 
55  auto configure(application_config& config) const -> bool {
56  if(auto pbe{backend()}) {
57  extract(pbe).configure(config);
58  }
59  return false;
60  }
61 
62  /// @brief Updates the activity progress to the specified current value.
63  /// @see finish
64  /// @see advance_progress
65  ///
66  /// The return value indicates if the activity was canceled, true values
67  /// means that the activity should continue, false means that the activity
68  /// was canceled and should discontinue.
69  auto update_progress(span_size_t current) const noexcept -> bool {
70  if(auto pbe{backend()}) {
71  return extract(pbe).update_progress(_activity_id, current);
72  }
73  return true;
74  }
75 
76  /// @brief Advances the activity progress by the specified increment.
77  /// @see finish
78  /// @see update_progress
79  ///
80  /// The return value indicates if the activity was canceled, true values
81  /// means that the activity should continue, false means that the activity
82  /// was canceled and should discontinue.
83  auto advance_progress(span_size_t increment = 1) const noexcept -> bool {
84  if(auto pbe{backend()}) {
85  return extract(pbe).advance_progress(_activity_id, increment);
86  }
87  return true;
88  }
89 
90  /// @brief Explicitly marks the activity as finished.
91  /// @see advance_progress
92  /// @see update_progress
93  void finish() noexcept {
94  if(_backend) {
95  _backend->finish_activity(_activity_id);
96  _backend.reset();
97  }
98  }
99 
100 protected:
101  activity_progress(std::shared_ptr<progress_tracker_backend> backend) noexcept
102  : _backend{std::move(backend)} {}
103 
104  auto backend() const noexcept
105  -> const std::shared_ptr<progress_tracker_backend>& {
106  return _backend;
107  }
108 
109 private:
110  std::shared_ptr<progress_tracker_backend> _backend;
111  const activity_progress_id_t _activity_id{0U};
112 };

114 } // namespace eagine
