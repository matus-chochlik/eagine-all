56  /// @brief Construction from a pack of integer values.

198  /// @brief Converts the byte sequence into an unsigned integer value.
199  template <typename UInt>
200  constexpr auto as(const UInt i = 0) const noexcept requires(
201  (sizeof(UInt) >= N) &&
202  (
203 #if __SIZEOF_INT128__
204  std::is_same_v<UInt, __uint128_t> || std::is_same_v<UInt, __int128_t> ||
205 #endif
206  std::is_integral_v<UInt>)) {
207  return _push_back_to(i, 0);
208  }

210 private:
211  value_type _bytes[N]{};
212 
213  template <typename UInt>
214  constexpr auto _push_back_to(const UInt state, const std::size_t i)
215  const noexcept -> UInt {
216  // NOLINTNEXTLINE(hicpp-signed-bitwise)
217  return (i < N) ? _push_back_to((state << CHAR_BIT) | _bytes[i], i + 1)
218  : state;
219  }

221  static constexpr auto _cmp_byte(

224  return (a == b) ? 0 : (a < b) ? -1 : 1;

227  static constexpr auto _do_cmp(

234  template <std::size_t I, std::size_t... In>
235  static constexpr auto _do_cmp(
236  const byteset& a,
237  const byteset& b,
238  const std::index_sequence<I, In...>) noexcept -> int {
239  return (a._bytes[I] == b._bytes[I])
240  ? _do_cmp(a, b, std::index_sequence<In...>{})
