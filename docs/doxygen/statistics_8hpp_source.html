19 /// @brief Service observing message bus node network statistics.
20 /// @ingroup msgbus
21 /// @see service_composition
22 template <typename Base = subscriber>
23 class statistics_consumer : public Base {
24  using This = statistics_consumer;
25 
26 public:
27  /// @brief Queries the statistics information of the specified bus node.
28  /// @see discover_statistics
29  void query_statistics(identifier_t node_id) noexcept {
30  message_view message{};
31  message.set_target_id(node_id);
32  const auto msg_id{EAGINE_MSGBUS_ID(statsQuery)};
33  this->bus_node().post(msg_id, message);
34  }
35 
36  /// @brief Broadcasts network statistics query to all message bus nodes.
37  /// @see query_statistics
38  void discover_statistics() noexcept {
39  query_statistics(broadcast_endpoint_id());
40  }
41 
42  /// @brief Triggered on receipt of router node statistics information.
43  /// @see router_disappeared
44  /// @see bridge_stats_received
45  /// @see endpoint_stats_received
46  /// @see connection_stats_received
47  signal<void(const identifier_t, const router_statistics&) noexcept>
48  router_stats_received;
49 
50  /// @brief Triggered on receipt of bridge node statistics information.
51  /// @see bridge_disappeared
52  /// @see router_stats_received
53  /// @see endpoint_stats_received
54  /// @see connection_stats_received
55  signal<void(const identifier_t, const bridge_statistics&) noexcept>
56  bridge_stats_received;
57 
58  /// @brief Triggered on receipt of endpoint node statistics information.
59  /// @see endpoint_disappeared
60  /// @see router_stats_received
61  /// @see bridge_stats_received
62  /// @see connection_stats_received
63  signal<void(const identifier_t, const endpoint_statistics&) noexcept>
64  endpoint_stats_received;
65 
66  /// @brief Triggered on receipt of connection statistics information.
67  /// @see router_stats_received
68  /// @see bridge_stats_received
69  /// @see endpoint_stats_received
70  signal<void(const connection_statistics&) noexcept>
71  connection_stats_received;
72 
73 protected:
74  using Base::Base;
75 
76  void add_methods() noexcept {
77  Base::add_methods();
78  Base::add_method(
79  this, EAGINE_MSG_MAP(eagiMsgBus, statsRutr, This, _handle_router));
80  Base::add_method(
81  this, EAGINE_MSG_MAP(eagiMsgBus, statsBrdg, This, _handle_bridge));
82  Base::add_method(
83  this, EAGINE_MSG_MAP(eagiMsgBus, statsEndpt, This, _handle_endpoint));
84  }
85 
86 private:
87  auto _handle_router(
88  const message_context&,
89  const stored_message& message) noexcept -> bool {
90  router_statistics stats{};
91  if(default_deserialize(stats, message.content())) {
92  router_stats_received(message.source_id, stats);
93  }
94  return true;
95  }
96 
97  auto _handle_bridge(
98  const message_context&,

100  bridge_statistics stats{};

102  bridge_stats_received(message.source_id, stats);
103  }
104  return true;
105  }
106 
107  auto _handle_endpoint(
108  const message_context&,
109  const stored_message& message) noexcept -> bool {
110  endpoint_statistics stats{};

112  endpoint_stats_received(message.source_id, stats);
113  }
114  return true;
115  }
116 
117  auto _handle_connection(
118  const message_context&,
119  const stored_message& message) noexcept -> bool {
120  connection_statistics stats{};
121  if(default_deserialize(stats, message.content())) {
122  connection_stats_received(stats);
123  }
124  return true;
125  }
