17 /// @brief Shape generator capability bit enumeration.
18 /// @ingroup shapes
19 /// @see generator_capabilities
20 enum class generator_capability : unsigned {
21  /// @brief Indexed drawing of elements should be used if possible.
22  indexed_drawing = 1U << 0U,
23  /// @brief Line or triangle strips should be generated if possible.
24  element_strips = 1U << 1U,
25  /// @brief Line or triangle fanst should be generated if possible.
26  element_fans = 1U << 2U,
27  /// @brief Primitive restart functionality should be used if possible.
28  primitive_restart = 1U << 3U
29 };
30 //------------------------------------------------------------------------------
31 /// @brief Alias for generator_capability bitfield type.
32 /// @ingroup shapes
33 using generator_capabilities = bitfield<generator_capability>;
34 //------------------------------------------------------------------------------
35 /// @brief Returns vertex_attrib_kinds value with all bits set.
36 /// @ingroup shapes
37 static constexpr auto all_generator_capabilities() noexcept
38  -> generator_capabilities {
39  return generator_capabilities{(1U << 4U) - 1U};
40 }
41 //------------------------------------------------------------------------------

43 /// @ingroup shapes
44 static constexpr auto operator|(
