16 namespace eagine::msgbus {

18 /// @brief Service observing message bus node network topology.
19 /// @ingroup msgbus
20 /// @see service_composition
21 template <typename Base = subscriber>
22 class network_topology : public Base {
23  using This = network_topology;
24 
25 public:
26  /// @brief Queries the topology information of the specified bus node.
27  /// @see discover_topology
28  void query_topology(const identifier_t node_id) noexcept {
29  message_view message{};
30  message.set_target_id(node_id);
31  const auto msg_id{EAGINE_MSGBUS_ID(topoQuery)};
32  this->bus_node().post(msg_id, message);
33  }
34 
35  /// @brief Broadcasts network topology query to all message bus nodes.
36  /// @see query_topology
37  void discover_topology() noexcept {
38  query_topology(broadcast_endpoint_id());
39  }
40 
41  /// @brief Triggered on receipt of router node topology information.
42  /// @see router_disappeared
43  /// @see bridge_appeared
44  /// @see endpoint_appeared
45  signal<void(const router_topology_info&) noexcept> router_appeared;
46 
47  /// @brief Triggered on receipt of bridge node topology information.
48  /// @see bridge_disappeared
49  /// @see router_appeared
50  /// @see endpoint_appeared
51  signal<void(const bridge_topology_info&) noexcept> bridge_appeared;
52 
53  /// @brief Triggered on receipt of endpoint node topology information.
54  /// @see endpoint_disappeared
55  /// @see router_appeared
56  /// @see bridge_appeared
57  signal<void(const endpoint_topology_info&) noexcept> endpoint_appeared;
58 
59  /// @brief Triggered on receipt of bye-bye message from a router node.
60  /// @see router_appeared
61  /// @see bridge_disappeared
62  /// @see endpoint_disappeared
63  signal<void(const identifier_t) noexcept> router_disappeared;
64 
65  /// @brief Triggered on receipt of bye-bye message from a bridge node.
66  /// @see bridge_appeared
67  /// @see router_disappeared
68  /// @see endpoint_disappeared
69  signal<void(const identifier_t) noexcept> bridge_disappeared;
70 
71  /// @brief Triggered on receipt of bye-bye message from an endpoint node.
72  /// @see endpoint_appeared
73  /// @see router_disappeared
74  /// @see bridge_disappeared
75  signal<void(const identifier_t) noexcept> endpoint_disappeared;
76 
77 protected:
78  using Base::Base;
79 
80  void add_methods() noexcept {
81  Base::add_methods();
82  Base::add_method(
83  this, EAGINE_MSG_MAP(eagiMsgBus, topoRutrCn, This, _handle_router));
84  Base::add_method(
85  this, EAGINE_MSG_MAP(eagiMsgBus, topoBrdgCn, This, _handle_bridge));
86  Base::add_method(
87  this, EAGINE_MSG_MAP(eagiMsgBus, topoEndpt, This, _handle_endpoint));
88  Base::add_method(
89  this,
90  EAGINE_MSG_MAP(eagiMsgBus, byeByeRutr, This, _handle_router_bye));
91  Base::add_method(
92  this,
93  EAGINE_MSG_MAP(eagiMsgBus, byeByeBrdg, This, _handle_bridge_bye));
94  Base::add_method(
95  this,
96  EAGINE_MSG_MAP(eagiMsgBus, byeByeEndp, This, _handle_endpoint_bye));
97  }
98 
99 private:
100  auto _handle_router(
101  const message_context&,
102  const stored_message& message) noexcept -> bool {
103  router_topology_info info{};
104  if(default_deserialize(info, message.content())) {
105  router_appeared(info);
106  }
107  return true;
108  }
109 
110  auto _handle_bridge(
111  const message_context&,
112  const stored_message& message) noexcept -> bool {
113  bridge_topology_info info{};
114  if(default_deserialize(info, message.content())) {
115  bridge_appeared(info);
116  }
117  return true;
118  }
119 
120  auto _handle_endpoint(
121  const message_context&,
122  const stored_message& message) noexcept -> bool {
123  endpoint_topology_info info{};
124  if(default_deserialize(info, message.content())) {
125  endpoint_appeared(info);
126  }
127  return true;
128  }
129 
130  auto _handle_router_bye(
131  const message_context&,
132  const stored_message& message) noexcept -> bool {
133  router_disappeared(message.source_id);
134  return true;
135  }
136 
137  auto _handle_bridge_bye(
138  const message_context&,
139  const stored_message& message) noexcept -> bool {
140  bridge_disappeared(message.source_id);
141  return true;
142  }
143 
144  auto _handle_endpoint_bye(
145  const message_context&,
146  const stored_message& message) noexcept -> bool {
147  endpoint_disappeared(message.source_id);
148  return true;
149  }
150 };
