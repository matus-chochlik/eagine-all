14 namespace eagine::oglplus {
15 //------------------------------------------------------------------------------
16 /// @brief Reference to a GLSL source code string.
17 /// @ingroup glsl_utils
18 class glsl_string_ref {
19 public:
20  /// @brief Alias for string character type.
21  using char_type = gl_types::char_type;
22 
23  /// @brief Alias for signed intever type.
24  using int_type = gl_types::int_type;
25 
26  /// @brief Construction from a C-string and a length value.
27  glsl_string_ref(const char* src_str, const span_size_t n) noexcept
28  : _src_str{static_cast<const char_type*>(src_str)}
29  , _length{int_type(n == 0 ? 0 : (src_str[n - 1] == '\0' ? n - 1 : n))} {}
30 
31  /// @brief Explicit construction from a string_view.
32  explicit glsl_string_ref(const string_view str) noexcept
33  : glsl_string_ref(str.data(), str.size()) {}
34 
35  /// @brief Explicit construction from a memory const_block.
36  explicit glsl_string_ref(const memory::const_block blk) noexcept
37  : glsl_string_ref(as_chars(blk)) {}
38 
39  /// @brief Conversion to glsl_source_ref
40  operator glsl_source_ref() const noexcept {
41  return {1, &_src_str, &_length};
42  }
43 
44 private:
45  mutable const char_type* _src_str{nullptr};
46  int_type _length{0};
47 };
48 //------------------------------------------------------------------------------
49 /// @brief Wrapper class for a GLSL string literal, convertible to glsl_source_ref.
50 /// @ingroup glsl_utils
51 /// @see glsl_source_ref
52 class glsl_literal : public glsl_string_ref {
53 public:
54  /// @brief Construction from a zero-terminated string literal.
55  template <span_size_t N>
56  glsl_literal(const char (&src_str)[N]) noexcept
57  : glsl_string_ref(static_cast<const char*>(src_str), N) {}
58 };
59 //------------------------------------------------------------------------------
60 /// @brief User defined literal for GLSL source code strings.
61 /// @ingroup glsl_utils
62 static inline auto operator"" _glsl(
63  const char* src_str,
64  const std::size_t n) noexcept -> glsl_string_ref {
65  return {src_str, span_size(n)};
