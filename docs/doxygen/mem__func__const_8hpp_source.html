21 /// @brief Implementation of compile-time member function pointer wrappers.
22 /// @ingroup functional
23 template <typename RV, typename C, typename... P, RV (C::*Ptr)(P...)>
24 struct member_function_constant<RV (C::*)(P...), Ptr> {
25 
26  /// @brief Alias for the member function pointer type.
27  using pointer = RV (C::*)(P...);
28 
29  /// @brief Alias for corresponding free function pointer type.
30  using free_pointer = RV (*)(C*, P...);
31 
32  /// @brief Alias for the function result type.
33  using result_type = RV;
34 
35  /// @brief Alias for the class to which the member function belongs.
36  using scope = C;
37 
38  /// @brief Alias for type indicating if the member function is const.
39  using is_const = std::false_type;
40 
41  /// @brief Returns the member function pointer.
42  /// @see free_func
43  static constexpr auto get() noexcept -> pointer {
44  return Ptr;
45  }
46 
47  /// @brief A function that calls the member function on an instance of @p C.
48  /// @see make_free
49  static auto free_func(C* c, P... a) -> RV {
50  return (c->*Ptr)(std::forward<P>(a)...);
51  }
52 
53  /// @brief Returns pointer to a free function corresponding to the member function.
54  /// @see get
55  /// @see free_func
56  static auto make_free() noexcept -> free_pointer {
57  return &free_func;
58  }
59 };
