118 template <typename Derived, typename Node>
119 class compound_with_refcounted_node : public compound_implementation<Derived> {
120 private:
121  std::vector<std::tuple<span_size_t, std::unique_ptr<Node>>> _nodes{};
122 
123  auto _do_make_new(Node&& temp) -> Node* {
124  for(auto& [ref_count, node_ptr] : _nodes) {
125  if(temp == *node_ptr) {
126  ++ref_count;
127  return node_ptr.get();
128  }
129  }
130  _nodes.emplace_back(1, std::make_unique<Node>(std::move(temp)));
131  return std::get<1>(_nodes.back()).get();
132  }
133 
134 protected:
135  inline auto _unwrap(attribute_interface& attrib) const noexcept -> auto& {
136  EAGINE_ASSERT(attrib.type_id() == this->type_id());
137  EAGINE_ASSERT(dynamic_cast<Node*>(&attrib));
138  return static_cast<Node&>(attrib);
139  }
140 
141 public:
142  template <typename... Args>
143  auto make_node(Args&&... args) {
144  return _do_make_new(Node{std::forward<Args>(args)...});
145  }
146 
147  void add_ref(attribute_interface& attrib) noexcept final {
148  auto& that = _unwrap(attrib);
149  for(auto& [ref_count, node_ptr] : _nodes) {
150  if(&that == node_ptr.get()) {
151  ++ref_count;
152  }
153  }
154  }
155 
156  void release(attribute_interface& attrib) noexcept final {
157  auto& that = _unwrap(attrib);
158  for(auto pos = _nodes.begin(); pos != _nodes.end(); ++pos) {
159  auto& [ref_count, node_ptr] = *pos;
160  if(&that == node_ptr.get()) {
161  if(--ref_count <= 0) {
162  _nodes.erase(pos);
163  break;
164  }
165  }
166  }
167  }
168 };
