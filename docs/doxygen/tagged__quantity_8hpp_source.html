32 //------------------------------------------------------------------------------
33 /// @brief Value of type T with a specified unit or tag type U.
34 /// @ingroup units
35 template <typename T, typename U>
36 class tagged_quantity {
37 public:
38  /// @brief Alias for the unit type.
39  using unit_type = U;
40 
41  /// @brief Alias for the value type.
42  using value_type = T;
43 
44  /// @brief Default constructor.
45  tagged_quantity() = default;
46 
47  /// @brief Explicit constructor initializing the value.
48  explicit constexpr tagged_quantity(T v) noexcept
49  : _v{std::move(v)} {}
50 
51  /// @brief Converting constructor from another tagged quantity.
52  template <typename X, typename UX>
53  constexpr tagged_quantity(const tagged_quantity<X, UX>& tq) noexcept
54  requires(std::is_convertible_v<X, T>&& units::is_convertible_v<UX, U>)
55  : _v(T(units::value_conv<UX, U>()(tq._v))) {}
56 
57  /// @brief Conversion to a quantity in another unit type.
58  template <typename UX>
59  constexpr auto to() const noexcept
60  requires(units::is_convertible_v<U, UX>) {
61  return make_tagged_quantity<UX>(units::value_conv<U, UX>()(_v));
62  }
63 
64  /// @brief Returns the value.
65  constexpr auto value() const noexcept -> T {
66  return _v;
67  }
68 
69  /// @brief Returns the unit.
70  static constexpr auto unit() noexcept -> U {
71  return {};
72  }
73 
74  /// Brief Explicit conversion of the value to the specified.
75  template <typename X>
76  explicit constexpr operator X() const noexcept
77  requires(std::is_convertible_v<T, X>) {
78  return X(_v);
79  }
80 
81  /// @brief Addition of another tagged quantity.
82  template <typename X, typename UX>
83  auto operator+=(const tagged_quantity<X, UX>& q) noexcept -> auto& requires(
84  std::is_convertible_v<X, T>&& units::is_convertible_v<UX, U>) {
85  _v += T(units::value_conv<UX, U>()(q._v));
86  return *this;
87  }
88 
89  /// @brief Subtraction of another tagged quantity.
90  template <typename X, typename UX>
91  auto operator-=(const tagged_quantity<X, UX>& q) noexcept -> auto& requires(
92  std::is_convertible_v<X, T>&& units::is_convertible_v<UX, U>) {
93  _v -= T(units::value_conv<UX, U>()(q._v));
94  return *this;
95  }
96 
97 private:
98  static_assert(std::is_arithmetic_v<T>);
99  T _v{};
100 
101  template <typename, typename>
102  friend class tagged_quantity;
103 };

155 template <typename T1, typename U1, typename T2, typename U2>
156 struct equal_cmp<tagged_quantity<T1, U1>, tagged_quantity<T2, U2>> {
157 
158  static_assert(units::is_convertible_v<U2, U1>);
159 
160  static constexpr auto check(
161  const tagged_quantity<T1, U1>& l,
162  const tagged_quantity<T2, U2>& r) noexcept -> bool {
163  return are_equal(value(l), units::value_conv<U2, U1>()(value(r)));
164  }
