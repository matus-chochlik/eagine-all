29 /// @ingroup application
30 /// @see video_options

34 /// Instances of this class represent a single (possibly one of several)

36 class video_context {
37 public:

39  execution_context& parent,
40  std::shared_ptr<video_provider> provider) noexcept;
41 
42  /// @brief Returns a reference to the parent application execution context.
43  auto parent() const noexcept -> execution_context& {
44  return _parent;
45  }
46 
47  /// @brief Returns the current video frame number.
48  auto frame_number() const noexcept {
49  return _frame_no;
50  }
51 
52  /// @brief Start working in this video rendering context (make it current).
53  /// @see end
54  /// @see commit
55  void begin();
56 
57  /// @brief Stop working with this video rendering context.
58  /// @see begin
59  /// @see commit
60  void end();
61 
62  /// @brief Apply the rendering commands done since the last commit (swap buffers).
63  /// @see begin
64  /// @see end
65  void commit();
66 
67  /// @brief Tries to intialize the GL rendering API in this video context.
68  /// @see has_gl_api
69  /// @see gl_api
70  auto init_gl_api() noexcept -> bool;
71 
72  /// @brief Indicates if the GL rendering API in this video context is initialized.
73  /// @see init_gl_api
74  /// @see gl_api
75  auto has_gl_api() const noexcept {
76  return bool(_gl_api);
77  }
78 
79  /// @brief Returns a reference to the GL rendering API in this context.
80  /// @see init_gl_api
81  /// @pre has_gl_api()
82  auto gl_api() const noexcept -> oglplus::gl_api& {
83  EAGINE_ASSERT(has_gl_api());
84  return *_gl_api;
85  }
86 
87  /// @brief Returns the rendering surface's dimensions (in pixels).
88  auto surface_size() noexcept -> std::tuple<int, int> {
89  if(_provider) [[likely]] {
90  return extract(_provider).surface_size();
91  }
92  return {1, 1};
93  }
94 
95  /// @brief Returns the rendering surface's aspect ratio.
96  auto surface_aspect() noexcept -> float {
97  if(_provider) [[likely]] {
98  return extract(_provider).surface_aspect();
99  }
100  return 1.F;
101  }
102 
103  /// @brief Adds a clean-up callback.
104  void add_cleanup_op(callable_ref<void(video_context&) noexcept> op);
105 
106  template <typename Obj>
107  auto clean_up_later(Obj& obj) -> auto& {
108  add_cleanup_op(callable_ref<void(video_context&) noexcept>{
109  obj, &_call_clean_up<Obj>});
110  return *this;
111  }
112 
113  /// @brief Cleans up and releases this rendering context and APIs.
114  void clean_up() noexcept;
115 
116 private:
117  template <typename Obj>
118  static void _call_clean_up(Obj* obj, video_context& vc) noexcept {
119  obj->clean_up(vc);
120  }
121 
122  execution_context& _parent;
123  long _frame_no{0};
124  std::shared_ptr<video_provider> _provider{};
125  std::shared_ptr<oglplus::gl_api> _gl_api{};
126  std::shared_ptr<video_context_state> _state{};
127 };
