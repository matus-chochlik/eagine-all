16 /// @brief Installs handlers for interrupt signals and flips switch on receipt.
17 ///
18 /// This class installs handlers for the interrupt and terminate OS IPC signals
19 /// and if one of the signals is received it flips a boolean indicator from
20 /// false to true.
21 ///
22 /// @note Only a single instance per process should be created.
23 class signal_switch {
24 public:
25  /// @brief Default constructor. Stores the original handlers if any.
26  /// @post !bool(*this)
27  signal_switch() noexcept
28  : _intr_handler{std::signal(SIGINT, &_flip)}
29  , _term_handler{std::signal(SIGTERM, &_flip)} {}
30 
31  /// @brief Destructor. Restores the original signal handlers.
32  ~signal_switch() noexcept {
33  [[maybe_unused]] const auto unused_int{
34  std::signal(SIGINT, _intr_handler)};
35  [[maybe_unused]] const auto unused_trm{
36  std::signal(SIGTERM, _term_handler)};
37  }
38 
39  /// @brief Not moveable.
40  signal_switch(signal_switch&&) = delete;
41  /// @brief Not copyable.
42  signal_switch(const signal_switch&) = delete;
43  /// @brief Not move-assignable.
44  auto operator=(signal_switch&&) = delete;
45  /// @brief Not copy-assignable.
46  auto operator=(const signal_switch&) = delete;
47 
48  /// @brief Resets the signal state as if no signal was received.
49  /// @post !bool(*this)
50  auto reset() noexcept -> signal_switch& {
51  _state() = 0;
52  return *this;
53  }
54 
55  /// @brief Indicates if the interrupt signal was received.
56  auto interrupted() const noexcept -> bool {
57  return _state() == SIGINT;
58  }
59 
60  /// @brief Indicates if the terminate signal was received.
61  auto terminated() const noexcept -> bool {
62  return _state() == SIGTERM;
63  }
64 
65  /// @brief Indicates if one of the tracked signals was received.
66  explicit operator bool() const noexcept {
67  return bool(_state());
68  }
69 
70 private:
71  static auto _state() noexcept -> volatile std::sig_atomic_t& {
72  static volatile std::sig_atomic_t state{0};
73  return state;
74  }
75 
76  static void _flip(const int sig_num) noexcept {
77  _state() = sig_num;
78  }
79 
80  using _sighandler_t = void(int);
81  _sighandler_t* _intr_handler{nullptr};
82  _sighandler_t* _term_handler{nullptr};
83 };
