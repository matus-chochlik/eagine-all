71 template <typename T>
72 static constexpr auto adapt_entry_arg(
73  const identifier name,
74  const T value) noexcept requires(has_enumerator_mapping_v<T>) {
75  return [=](auto& backend) {
76  backend.add_string(name, EAGINE_ID(enum), enumerator_name(value));
77  };
78 }
79 //------------------------------------------------------------------------------
80 template <typename T>
81 static constexpr auto adapt_entry_arg(
82  const identifier name,
83  const bitfield<T> bf) noexcept requires(has_enumerator_mapping_v<T>) {
84  return [=](auto& backend) {
85  const auto func = [&backend, name, bf](const auto& info) {
86  if(bf.has(static_cast<T>(info.value))) {
87  backend.add_string(name, EAGINE_ID(bitfield), info.name);
88  }
89  };
90  for_each_enumerator(func, type_identity<T>{});
91  };
92 }
