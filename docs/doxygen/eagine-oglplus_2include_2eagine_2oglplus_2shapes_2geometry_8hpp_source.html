16 /// @brief Class wrapping a vertex attribute array and buffers storing shape geometry.
17 /// @see generator
18 /// @see vertex_attrib_bindings
19 /// @ingroup shapes
20 class geometry {
21 public:
22  /// @brief Default constructor.
23  geometry() noexcept = default;
24 
25  /// @brief Initializes a previously default initialized geometry instance.
26  auto init(
27  const gl_api& glapi,
28  const shape_generator& shape,
29  const vertex_attrib_bindings& bindings,
30  const shapes::drawing_variant var,
31  memory::buffer& temp) -> auto& {
32 
33  const auto& gl = glapi;
34  gl.gen_vertex_arrays() >> _vao;
35  const auto attrib_count{bindings.attrib_count()};
36  auto buffer_count{attrib_count};
37 
38  _ops.resize(integer(shape.operation_count(var)));
39  shape.instructions(glapi, var, cover(_ops));
40 
41  const auto indexed{shape.indexed_drawing(var)};
42 
43  if(indexed) {
44  ++buffer_count;
45  }
46  _buffers.resize(buffer_count);
47  gl.gen_buffers(_buffers.raw_handles());
48 
49  gl.bind_vertex_array(_vao);
50 
51  for(const integer i : integer_range(attrib_count)) {
52  shape.attrib_setup(
53  glapi,
54  _vao,
55  _buffers[i],
56  vertex_attrib_location{i},
57  bindings.attrib_variant(i),
58  temp);
59  }
60  if(indexed) {
61  shape.index_setup(glapi, _buffers[attrib_count], temp);
62  }
63  return *this;
64  }
65 
66  /// @brief Initializes a previously default initialized geometry instance.
67  auto init(
68  const gl_api& glapi,
69  const shape_generator& shape,
70  const vertex_attrib_bindings& bindings,
71  memory::buffer& temp) -> auto& {
72  return init(glapi, shape, bindings, shape.draw_variant(0), temp);
73  }
74 
75  /// @brief Construction using shape generator, attrib bindings and drawing variant.
76  geometry(
77  const gl_api& glapi,
78  const shape_generator& shape,
79  const vertex_attrib_bindings& bindings,
80  const shapes::drawing_variant var,
81  memory::buffer& temp) {
82  init(glapi, shape, bindings, var, temp);
83  }
84 
85  /// @brief Construction using shape generator and attrib bindings.
86  geometry(
87  const gl_api& glapi,
88  const shape_generator& shape,
89  const vertex_attrib_bindings& bindings,
90  memory::buffer& temp)
91  : geometry{glapi, shape, bindings, shape.draw_variant(0), temp} {}
92 
93  /// @brief Releases the used OpenGL resources.
94  void clean_up(const gl_api& gl) {
95  gl.delete_buffers(_buffers.raw_handles());
96  gl.delete_vertex_arrays(std::move(_vao));
97  }
98 
99  /// @brief Prepares and binds the resources to be used (for example by draw).
100  /// @see draw
101  auto use(const gl_api& gl) const {
102  return gl.bind_vertex_array(_vao);
103  }
104 
105  /// @brief Emits geometry draw commands using the specified GL API.
106  /// @see use
107  /// @see draw_instanced
108  auto draw(const gl_api& glapi) const {
109  draw_using_instructions(glapi, view(_ops));
110  }
111 
112  /// @brief Emits geometry draw commands using the specified GL API.
113  /// @see use
114  /// @see draw
115  auto draw_instanced(
116  const gl_api& glapi,
117  const gl_types::sizei_type inst_count) const {
118  draw_instanced_using_instructions(glapi, view(_ops), inst_count);
119  }
120 
121 private:
122  std::vector<shape_draw_operation> _ops;
123  owned_vertex_array_name _vao;
124  gl_object_name_vector<buffer_tag> _buffers;
125 };
