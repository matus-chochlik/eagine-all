18 /// @brief Class loading and managing message bus router address(es).
19 /// @ingroup msgbus
20 class router_address : public main_ctx_object {
21 public:
22  router_address(main_ctx_parent parent, const nothing_t) noexcept
23  : main_ctx_object{EAGINE_ID(RouterAddr), parent} {}
24 
25  /// @brief Construction from parent main context object.
26  router_address(main_ctx_parent parent) noexcept
27  : router_address{parent, nothing} {
28  configure(app_config());
29  }
30 
31  void configure(application_config& config) {
32  if(config.fetch("msgbus.router.address", _addrs)) {
33  log_debug("configured router address(es) ${addr}")
34  .arg_func([&](logger_backend& backend) {
35  for(auto& addr : _addrs) {
36  backend.add_string(
37  EAGINE_ID(address), EAGINE_ID(string), addr);
38  }
39  });
40  }
41  }
42 
43  /// @brief Indicates if this instance contains at least one address.
44  explicit operator bool() const noexcept {
45  return !_addrs.empty();
46  }
47 
48  /// @brief Implicit conversion to string_view, returning the first address.
49  operator string_view() const noexcept {
50  if(_addrs.empty()) {
51  return {};
52  }
53  return {_addrs.front()};
54  }
55 
56  /// @brief Returns the number of addresses stored in this instance.
57  auto count() const noexcept -> span_size_t {
58  return span_size(_addrs.size());
59  }
60 
61  /// @brief Returns a const iterator to the start of the range of addresses.
62  auto begin() const noexcept {
63  return _addrs.cbegin();
64  }
65 
66  /// @brief Returns a const iterator past the end of the range of addresses.
67  auto end() const noexcept {
68  return _addrs.cend();
69  }
70 
71 private:
72  std::vector<std::string> _addrs{};
73 };
