56 static constexpr auto max_code_point(const valid_sequence_length& len) noexcept
57  -> valid_if_not_zero<code_point_t> {
58  return len == 1 ? 0x0000007F : // 7b
59  len == 2 ? 0x000007FF
60  : // 11b
61  len == 3 ? 0x0000FFFF
62  : // 16b
63  len == 4 ? 0x001FFFFF
64  : // 21b
65  len == 5 ? 0x03FFFFFF
66  : // 26b
67  len == 6 ? 0x7FFFFFFF
68  : // 31b
69  0x00000000;
70 }
71 //------------------------------------------------------------------------------
72 static constexpr auto head_data_bitshift(
73  const valid_sequence_length& len) noexcept -> valid_span_size {
74  return {len.is_valid() ? (extract(len) - 1) * 6 : -1};
75 }
76 //------------------------------------------------------------------------------
77 static constexpr auto tail_data_bitshift(
78  const valid_sequence_length& idx,
79  const valid_sequence_length& len) noexcept -> valid_span_size {
80  return {
81  (idx.is_valid() && len.is_valid()) ? (extract(len) - extract(idx) - 1) * 6
82  : -1};
83 }
84 //------------------------------------------------------------------------------
85 static constexpr auto head_code_mask(const valid_sequence_length& len) noexcept
86  -> valid_if_not_zero<byte> {
87  return len == 1 ? 0x80
88  : len == 2 ? 0xE0
89  : len == 3 ? 0xF0
90  : len == 4 ? 0xF8
91  : len == 5 ? 0xFC
92  : len == 6 ? 0xFE
93  : 0x00;
94 }

98  -> optionally_valid<byte> {

115 template <typename P>
116 static constexpr auto head_code_from_mask(const valid_if<byte, P> mask) noexcept
117  -> optionally_valid<byte> {
118  // NOLINTNEXTLINE(hicpp-signed-bitwise)
119  return {byte((mask.value_anyway() << 1) & 0xFF), mask.is_valid()};
120 }
121 //------------------------------------------------------------------------------
122 static constexpr auto head_code(const valid_sequence_length& len) noexcept
123  -> optionally_valid<byte> {
124  return head_code_from_mask(head_code_mask(len));
125 }

130 //------------------------------------------------------------------------------
131 template <typename P1, typename P2>
132 static constexpr auto is_valid_masked_code(
133  const byte b,
134  const valid_if<byte, P1> mask,
135  const valid_if<byte, P2> code) noexcept -> bool {
136  return (mask.is_valid() && code.is_valid())
137  ? (b & extract(mask)) == extract(code)
138  : false;
139 }
140 //------------------------------------------------------------------------------
141 static constexpr auto is_valid_head_byte(
142  const byte b,
143  const valid_sequence_length& l) noexcept -> bool {
144  return is_valid_masked_code(b, head_code_mask(l), head_code(l));
145 }
146 //------------------------------------------------------------------------------
147 static constexpr auto is_valid_head_byte(const byte b) noexcept -> bool {
148  return is_valid_head_byte(b, 1) || is_valid_head_byte(b, 2) ||
149  is_valid_head_byte(b, 3) || is_valid_head_byte(b, 4) ||
150  is_valid_head_byte(b, 5) || is_valid_head_byte(b, 6);
151 }
152 //------------------------------------------------------------------------------
153 static constexpr auto is_valid_tail_byte(
154  const byte b,
155  const valid_sequence_length&,
156  const valid_sequence_length&) noexcept -> bool {
157  return is_valid_masked_code(b, tail_code_mask(), tail_code());
158 }

160 static constexpr auto required_sequence_length(const code_point_t cp) noexcept
161  -> valid_sequence_length {
162  return (max_code_point(1) > cp) ? 1
