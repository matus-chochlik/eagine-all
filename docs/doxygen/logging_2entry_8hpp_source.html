21 static inline auto adapt_entry_arg(
22  const identifier name,
23  logger_backend* value) {
24  return [name, value](logger_backend& backend) noexcept {
25  if(value) {
26  backend.add_identifier(
27  name, EAGINE_ID(LogBkEndId), value->type_id());
28  } else {
29  backend.add_nothing(name, EAGINE_ID(LogBkEndId));
30  }
31  };
32 }

745 class stream_log_entry {
746 public:
747  /// @brief Implicit conversion to an istream.
748  operator std::ostream&() noexcept {
749  return _out;
750  }
751 
752  stream_log_entry(
753  const identifier source_id,
754  const logger_instance_id instance_id,
755  const log_event_severity severity,
756  logger_backend* backend) noexcept
757  : _source_id{source_id}
758  , _instance_id{instance_id}
759  , _backend{backend}
760  , _severity{severity} {}
761 
762  stream_log_entry(stream_log_entry&&) = default;
763  stream_log_entry(const stream_log_entry&) = delete;
764  auto operator=(stream_log_entry&&) = delete;
765  auto operator=(const stream_log_entry&) = delete;
766 
767  ~stream_log_entry() noexcept {
768  try {
769  auto fmt_str(_out.str());
770  if(!fmt_str.empty()) {
771  if(_backend) {
772  if(_backend->begin_message(
773  _source_id,
774  _entry_tag,
775  _instance_id,
776  _severity,
777  fmt_str)) [[likely]] {
778  _backend->finish_message();
779  _backend = nullptr;
780  }
781  }
782  }
783  } catch(...) {
784  }
785  }
786 
787  auto tag(const identifier entry_tag) noexcept -> auto& {
788  _entry_tag = entry_tag;
789  return *this;
790  }
791

794  identifier _source_id{};
795  identifier _entry_tag{};
796  logger_instance_id _instance_id{};
797  logger_backend* _backend{nullptr};
