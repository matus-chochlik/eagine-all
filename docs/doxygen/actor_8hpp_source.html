16 /// @brief Base class for message bus actors with fixed number of message handlers.
17 /// @ingroup msgbus
18 /// @see static_subscriber

21 template <
22  std::size_t N,
23  template <std::size_t> class Subscriber = static_subscriber>
24 class actor
25  : public connection_user
26  , public friend_of_endpoint {
27  using friend_of_endpoint::_accept_message;
28  using friend_of_endpoint::_make_endpoint;
29  using friend_of_endpoint::_move_endpoint;
30 
31 public:
32  /// @brief Not move constructible.
33  actor(actor&&) = delete;
34 
35  /// @brief Not copy constructible.
36  actor(const actor&) = delete;
37 
38  /// @brief Not moved assignable.
39  auto operator=(actor&&) = delete;
40 
41  /// @brief Not copy assignable.
42  auto operator=(const actor&) = delete;
43 
44  /// @brief Returns a reference to the associated endpoint.
45  auto bus_node() noexcept -> endpoint& {
46  return _endpoint;
47  }
48 
49  /// @brief Adds a connection to the associated endpoint.
50  auto add_connection(std::unique_ptr<connection> conn) noexcept
51  -> bool final {
52  return _endpoint.add_connection(std::move(conn));
53  }
54 
55  void allow_subscriptions() noexcept {
56  _subscriber.allow_subscriptions();
57  }
58 
59  /// @brief Processes a single enqueued message for which there is a handler.
60  /// @see process_all
61  auto process_one() noexcept {
62  _endpoint.update();
63  return _subscriber.process_one();
64  }
65 
66  /// @brief Processes all enqueued messages for which there are handlers.
67  /// @see process_one
68  auto process_all() noexcept {
69  _endpoint.update();
70  return _subscriber.process_all();
71  }
72 
73 protected:
74  auto _process_message(
75  const message_id msg_id,
76  const message_age,
77  const message_view& message) noexcept -> bool {
78  // TODO: use message age
79  if(!_accept_message(_endpoint, msg_id, message)) {
80  if(!is_special_message(msg_id)) {
81  _endpoint.block_message_type(msg_id);
82  }
83  }
84  return true;
85  }
86 
87  /// @brief Constructor usable from derived classes
88  template <typename Class, typename... MsgMaps>
89  actor(
90  main_ctx_object obj, // NOLINT(performance-unnecessary-value-param)
91  Class* instance,
92  MsgMaps... msg_maps) noexcept requires(sizeof...(MsgMaps) == N)
93  : _endpoint{_make_endpoint(
94  std::move(obj),
95  EAGINE_THIS_MEM_FUNC_REF(_process_message))}
96  , _subscriber{_endpoint, instance, msg_maps...} {
97  _subscriber.announce_subscriptions();
98  }
99 
100  /// @brief Constructor usable from derived classes
101  template <typename Derived, typename Class, typename... MsgMaps>
102  actor(Derived&& temp, Class* instance, const MsgMaps... msg_maps) noexcept
103  requires((sizeof...(MsgMaps) == N) && std::is_base_of_v<actor, Derived>)
104  : _endpoint{_move_endpoint(
105  std::move(temp._endpoint),
106  EAGINE_THIS_MEM_FUNC_REF(_process_message))}
107  , _subscriber{_endpoint, instance, msg_maps...} {}
108 
109  ~actor() noexcept override {
110  try {
111  _subscriber.retract_subscriptions();
112  _endpoint.finish();
113  } catch(...) {
114  }
115  }
116 
117 private:
118  endpoint _endpoint;
119  Subscriber<N> _subscriber;
120 };
