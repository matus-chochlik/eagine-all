17 template <std::size_t N, typename I>
18 constexpr auto integer_rotate_right(
19  const I x,
20  const type_identity<I> = {},
21  const size_constant<N> = {}) noexcept -> I {
22  static_assert(N < sizeof(I) * 8);
23  return I(x << N) | I(x >> (-N & (sizeof(I) * 8U - 1U)));
24 }
25 //------------------------------------------------------------------------------
26 template <typename H, typename I>
27 constexpr auto integer_hash_init(
28  const I x,
29  const type_identity<H> = {},
30  const type_identity<I> = {}) noexcept -> H {
31  using std::is_same_v;
32 
33  using UI = std::make_unsigned_t<I>;
34  UI ux(x);
35 
36  if constexpr(sizeof(H) > sizeof(UI)) {
37  H result{};
38  for(std::size_t i = 0; i < sizeof(H) / sizeof(UI); ++i) {
39  result <<= 8U * sizeof(UI);
40  result ^= H(ux);
41  }
42  return result;
43  } else if constexpr(sizeof(H) < sizeof(UI)) {
44  H result{};
45  for(std::size_t i = 0; i < sizeof(UI) / sizeof(H); ++i) {
46  result ^= H(ux);
47  ux >>= 8U * sizeof(H);
48  }
49  return result;
50  } else {
51  return ux;
52  }
53 }
