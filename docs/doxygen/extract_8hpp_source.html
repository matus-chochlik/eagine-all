107 //------------------------------------------------------------------------------
108 /// @brief Value typically wrapping function call result and success indicator.
109 /// @tparam Outcome the actual function call result type.
110 /// @see ok_traits
111 /// @see valid_if
112 ///
113 /// Instances of this class are used to store the result of a function call
114 /// and store the actual result together with some ingication whether the call
115 /// was actually successful. This means that the stored result value is
116 /// only optionally valid as indicated by the conversion-to-boolean operator.
117 ///
118 /// The ok class can, depending on specializations of ok_traits, provide
119 /// additional information about why the associated function call failed,
120 /// for example an error code or error message string, etc.
121 ///
122 /// This class is typically used to store and retrieve C-API function call
123 /// results where the success is indicated by @c errno or some such global
124 /// variable.
125 template <typename Outcome>
126 class ok {
127 private:
128  Outcome _outcome{};
129 
130  using _traits = ok_traits<Outcome>;
131 
132 public:
133  /// @brief Construction from a function call @p outcome object.
134  constexpr ok(Outcome&& outcome) noexcept(noexcept(std::declval<Outcome&&>()))
135  : _outcome{std::move(outcome)} {}
136 
137  /// @brief Indicates if the stored outcome contains valid result value.
138  explicit constexpr operator bool() noexcept(
139  noexcept(bool(std::declval<Outcome&>()))) {
140  return bool(_outcome);
141  }
142 
143  /// @brief Indicates if the stored outcome contains valid result value.
144  explicit constexpr operator bool() const
145  noexcept(noexcept(bool(std::declval<const Outcome&>()))) {
146  return bool(_outcome);
147  }
148 
149  /// @brief Extracts the stored outcome value.
150  /// @pre bool(*this)
151  constexpr auto get() noexcept(noexcept(extract(_outcome)))
152  -> decltype(extract(_outcome)) {
153  return extract(_outcome);
154  }
155 
156  /// @brief Extracts the stored outcome value.
157  /// @pre bool(*this)
158  constexpr auto get() const noexcept(noexcept(extract(_outcome)))
159  -> decltype(extract(_outcome)) {
160  return extract(_outcome);
161  }
162 
163  /// @brief Implicit conversion to the stored outcome value.
164  /// @pre bool(*this)
165  constexpr operator decltype(extract(std::declval<Outcome&>()))() noexcept(
166  noexcept(extract(_outcome))) {
167  return extract(_outcome);
168  }
169 
170  /// @brief Implicit conversion to the stored outcome value.
171  /// @pre bool(*this)
172  constexpr operator decltype(extract(std::declval<const Outcome&>()))() const
173  noexcept(noexcept(extract(_outcome))) {
174  return extract(_outcome);
175  }
176 
177  constexpr auto nok() const noexcept
178  -> decltype(_traits::nok_info(_outcome)) {
179  return _traits::nok_info(_outcome);
180  }
181 
182  constexpr auto operator!() const noexcept
183  -> decltype(_traits::nok_info(_outcome)) {
184  return _traits::nok_info(_outcome);
185  }
186 };
