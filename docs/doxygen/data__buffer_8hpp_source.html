137 template <identifier_t Sid, typename T, std::size_t N, typename Selector>
138 struct get_serialize_buffer_size<Sid, std::array<T, N>, Selector>
139  : serialize_size_constant<
140  (4 + (2 + serialize_buffer_size_v<Sid, T, Selector>)*N),
141  (get_serialize_buffer_size<Sid, T, Selector>::is_constant)> {
142 
143  static constexpr auto get(
144  string_view s,
145  const std::array<T, N>& v) noexcept {
146  auto result = std_size(12 + s.size());
147  for(const auto& e : v) {
148  result +=
149  2 + get_serialize_buffer_size<Sid, T, Selector>::get({}, e);
150  }
151  return result;
152  }
153 };

155 template <identifier_t Sid, typename... T, typename Selector>
156 struct get_serialize_buffer_size<Sid, std::tuple<T...>, Selector>
157  : serialize_size_constant<
158  (4 + ... + (2 + serialize_buffer_size_v<Sid, T, Selector>)),
159  (true && ... && (serialize_buffer_is_constant_v<Sid, T, Selector>))> {
160 
161  template <typename Tup>
162  static constexpr auto get(string_view s, const Tup& v) noexcept {
163  return std_size(
164  12 + s.size() + 2 +
165  _do_get(v, std::make_index_sequence<sizeof...(T)>()));
166  }
167 
168 private:
169  template <typename Tup, std::size_t... I>
170  static constexpr auto _do_get(
171  const Tup& t,
172  std::index_sequence<I...>) noexcept {
173  return (
174  0 + ... +
175  (2 + get_serialize_buffer_size<Sid, T, Selector>::get(
176  {}, std::get<I>(t))));
177  }
178 };

180 template <identifier_t Sid, typename T, typename Selector>
181 struct get_struct_serialize_buffer_size : serialize_size_constant<0, false> {
182 
183  static constexpr auto get(string_view s, const T& v) noexcept {
184  return std_size(12 + s.size() + 2 + _get(map_data_members(v)));
185  }
186 
187 private:
188  template <typename Mapped>
189  static constexpr auto _get(const Mapped& mapped) noexcept {
190  return _do_get(
191  mapped, std::make_index_sequence<std::tuple_size_v<Mapped>>());
192  }
193 
194  template <typename... M, std::size_t... I>
195  static constexpr auto _do_get(
196  const std::tuple<std::pair<const string_view, const M&>...>& mapped,
197  std::index_sequence<I...>) noexcept {
198  return span_size(
199  (0 + ... +
200  (2 + get_serialize_buffer_size<Sid, M, Selector>::get(
201  std::get<I>(mapped).first, std::get<I>(mapped).second))));
202  }
203 };

222 inline auto serialize_buffer_size_for(const T& inst, Selector) noexcept
223  -> span_size_t {
224  return get_serialize_buffer_size<SerializerId, T, Selector>::get({}, inst);

226 //------------------------------------------------------------------------------
227 template <identifier_t SerializerId, typename T, typename Selector>
228 inline auto get_serialize_vector_for(const T& inst, Selector sel) noexcept
229  -> std::vector<byte> {
230  std::vector<byte> result;
231  result.resize(serialize_buffer_size_for<SerializerId>(inst, sel));
232  return result;
233 }
234 //------------------------------------------------------------------------------
235 /// @brief Returns a buffer large enough for the serialization of the specified instance.
236 /// @ingroup serialization
237 template <
238  identifier_t SerializerId,
239  typename T,
240  typename Selector = default_selector_t>
241 constexpr auto serialize_buffer_for(const T& inst, Selector sel = {}) noexcept {
242  if constexpr(get_serialize_buffer_size<SerializerId, T, Selector>::
243  is_constant) {
244  return get_serialize_array_for<SerializerId>(inst, sel);
245  } else {
246  return get_serialize_vector_for<SerializerId>(inst, sel);
247  }
