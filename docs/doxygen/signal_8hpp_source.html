20 template <typename Signature>
21 class signal;

23 //------------------------------------------------------------------------------
24 /// @brief Class that allows connecting and invoking callables with the specified signature.
25 /// @ingroup msgbus
26 /// @see signal_binding
27 template <typename... Params>
28 class signal<void(Params...) noexcept> {
29 public:
30  /// @brief Type of key that can be used to remove connected slot.
31  using binding_key = signal_binding_key;
32 
33  /// @brief Connects the specified callable, returns a key for removal.
34  /// @see bind
35  /// @see disconnect
36  auto connect(const callable_ref<void(Params...) noexcept> slot)
37  -> binding_key {
38  const auto get_key = [](const auto& entry) {
39  return std::get<0>(entry);
40  };
41  auto key = 1U;
42  auto pos = _slots.begin();
43 
44  if(!_slots.empty()) {
45  key = get_key(_slots.back());
46  key = key != std::numeric_limits<signal_binding_key>::max()
47  ? key + 1
48  : 1U;
49  pos = [&]() {
50  while(true) {
51  auto [p, unused] = _slots.find_insert_position(key);
52  if(unused) {
53  return p;
54  }
55  ++key;
56  }
57  }();
58  }
59  EAGINE_ASSERT(key);
60  _slots.insert(pos, {key, slot});
61  return key;
62  }
63 
64  /// @brief Connects the specified callable, returns a binding object.
65  /// @see connect
66  /// @see signal_binding
67  [[nodiscard]] auto bind(const callable_ref<void(Params...) noexcept> slot)
68  -> signal_binding;
69 
70  /// @brief Disconnects the callable by specifying the connection key.
71  /// @see connect
72  auto disconnect(const binding_key key) noexcept -> auto& {
73  _slots.erase(key);
74  return *this;
75  }
76 
77  void unbind(const binding_key key) noexcept {
78  _slots.erase(key);
79  }
80 
81  /// @brief The call operator.
82  void operator()(Params... args) const noexcept {
83  _call(args...);
84  }
85 
86  /// @brief Implicit conversion to a compatible callable_ref.
87  template <typename... P>
88  operator callable_ref<void(P...) noexcept>() const noexcept {
89  return {
90  this,
91  member_function_constant<
92  void (signal::*)(P...) const noexcept,
93  &signal::_call>{}};
94  }
95 
96 private:
97  template <typename... P>
98  void _call(P... args) const noexcept {
99  for(const auto& entry : _slots) {
100  std::get<1>(entry)(args...);
101  }
102  }
103 
104  flat_map<unsigned, callable_ref<void(Params...) noexcept>> _slots;
105 };
106 //------------------------------------------------------------------------------
107 /// @brief Class that keeps a signam binding key and disconnects it when destroyed.
108 /// @ingroup msgbus
109 /// @see signal
110 class signal_binding {
111 public:
112  /// @brief Default constructor.
113  constexpr signal_binding() noexcept = default;
114 
115  template <typename Signature>
116  signal_binding(const signal_binding_key key, signal<Signature>* sig) noexcept
117  : _key{key}
118  , _disconnect{sig, EAGINE_MEM_FUNC_C(signal<Signature>, unbind)} {}
119 
120  /// @brief Move constructible.
121  signal_binding(signal_binding&& temp) noexcept
122  : _key{std::exchange(temp._key, 0U)}
123  , _disconnect{std::move(temp._disconnect)} {}
124 
125  /// @brief Not copy constructible.
126  signal_binding(const signal_binding&) = delete;
127 
128  /// 2Brief Disconnects the associated binding.
129  ~signal_binding() noexcept {
130  if(_key && _disconnect) {
131  _disconnect(_key);
132  }
133  }
134 
135  /// @brief Move assignable.
136  auto operator=(signal_binding&& temp) noexcept -> signal_binding& {
137  using std::swap;
138  swap(this->_key, temp._key);
139  swap(this->_disconnect, temp._disconnect);
140  return *this;
141  }
142 
143  /// @brief Not copy assignable.
144  auto operator=(const signal_binding&) = delete;
145 
146  /// @brief Indicates if the binding is valid.
147  constexpr operator bool() const noexcept {
148  return _key != 0;
149  }
150 
151 private:
152  signal_binding_key _key{0U};
153  callable_ref<void(signal_binding_key) noexcept> _disconnect;
154 };
155 //------------------------------------------------------------------------------
156 template <typename... Params>
157 [[nodiscard]] auto signal<void(Params...) noexcept>::bind(
158  callable_ref<void(Params...) noexcept> slot) -> signal_binding {
159  return {connect(slot), this};
160 }
