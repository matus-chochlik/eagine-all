26 /// @brief Implementation of class template storing a reference to a callable object.
27 /// @ingroup functional
28 /// @see callable_ref
29 ///
30 /// Unlike std::function basic_callable_ref does not do any conversion on
31 /// the parameters or return values, the referenced callable must match,
32 /// and it does not copy the callable, it must remain in scope while the
33 /// reference is used.
34 template <typename RV, typename... P, bool NE>
35 class basic_callable_ref<RV(P...) noexcept(NE), NE> {
36 public:
37  /// @brief Alias for the callable's argument type tuple.
38  using argument_tuple_type =
39  std::tuple<std::remove_cv_t<std::remove_reference_t<P>>...>;
40 
41  /// @brief Creates a default-constructed instance of argument_type_tuple.
42  static auto argument_tuple() noexcept -> argument_tuple_type {
43  return {};
44  }
45 
46  /// @brief Default constructor.
47  /// @post !is_valid
48  constexpr basic_callable_ref() noexcept = default;
49 
50  /// @brief Move constructor.
51  constexpr basic_callable_ref(basic_callable_ref&& temp) noexcept
52  : _data{std::exchange(temp._data, nullptr)}
53  , _func{std::exchange(temp._func, nullptr)} {}
54 
55  /// @brief Copy constructor.
56  constexpr basic_callable_ref(const basic_callable_ref&) noexcept = default;
57 
58  /// @brief Move assignment operator.
59  constexpr auto operator=(basic_callable_ref&& temp) noexcept
60  -> basic_callable_ref& {
61  std::swap(temp._data, _data);
62  std::swap(temp._func, _func);
63  return *this;
64  }
65 
66  /// @brief Copy assignment operator.
67  constexpr auto operator=(const basic_callable_ref&) noexcept
68  -> basic_callable_ref& = default;
69 
70  ~basic_callable_ref() noexcept = default;
71 
72  /// @brief Construction from a pointer to function.
73  basic_callable_ref(RV (*func)(P...) noexcept(NE)) noexcept
74  : _func{reinterpret_cast<_func_t>(func)} {}
75 
76  /// @brief Construction from pointers to an object and a function.
77  template <typename T>
78  basic_callable_ref(T* data, RV (*func)(T*, P...) noexcept(NE)) noexcept
79  : _data{static_cast<void*>(data)}
80  , _func{reinterpret_cast<_func_t>(func)} {
81  EAGINE_ASSERT(_data != nullptr);
82  }
83 
84  /// @brief Construction from a reference to an object and a pointer to function.
85  template <typename T>
86  basic_callable_ref(T& data, RV (*func)(T*, P...) noexcept(NE)) noexcept
87  : _data{static_cast<void*>(&data)}
88  , _func{reinterpret_cast<_func_t>(func)} {}
89 
90  /// @brief Construction a reference to object with a call operator.
91  template <typename C>
92  basic_callable_ref(construct_from_t, C& obj) noexcept
93  requires(!std::is_same_v<C, basic_callable_ref>)
94  : _data{static_cast<void*>(&obj)}
95  , _func{reinterpret_cast<_func_t>(&_cls_fn_call_op<C>)} {}
96 
97  /// @brief Construction a const reference to object with a call operator.
98  template <typename C>
99  basic_callable_ref(construct_from_t, const C& obj) noexcept
100  requires(!std::is_same_v<C, basic_callable_ref>)
101  : _data{static_cast<void*>(const_cast<C*>(&obj))}
102  , _func{reinterpret_cast<_func_t>(&_cls_fn_call_op_c<C>)} {}
103 
104  /// @brief Construction a pointer to object and member function constant.
105  template <typename C, RV (C::*Ptr)(P...) noexcept(NE)>
106  basic_callable_ref(
107  C* obj,
108  member_function_constant<RV (C::*)(P...) noexcept(NE), Ptr> mfc) noexcept
109  : _data{static_cast<void*>(obj)}
110  , _func{reinterpret_cast<_func_t>(mfc.make_free())} {
111  EAGINE_ASSERT(_data != nullptr);
112  EAGINE_ASSERT(_func != nullptr);
113  }
114 
115  /// @brief Construction a pointer to const object and member function constant.
116  template <typename C, RV (C::*Ptr)(P...) const noexcept(NE)>
117  basic_callable_ref(
118  const C* obj,
119  member_function_constant<RV (C::*)(P...) const noexcept(NE), Ptr>
120  mfc) noexcept
121  : _data{static_cast<void*>(const_cast<C*>(obj))}
122  , _func{reinterpret_cast<_func_t>(mfc.make_free())} {
123  EAGINE_ASSERT(_data != nullptr);
124  EAGINE_ASSERT(_func != nullptr);
125  }

127  /// @brief Indicates if this object stores a valid callable reference.
128  constexpr auto is_valid() const noexcept {
129  return _func != nullptr;
130  }
131 
132  /// @brief Indicates if this object stores a valid callable reference.
133  /// @see is_valid
134  explicit constexpr operator bool() const noexcept {
135  return is_valid();
136  }
137 
138  /// @brief Call operator.
139  /// @pre is_valid()
140  template <typename... A>
141  auto operator()(A&&... a) const -> RV {
142  EAGINE_ASSERT(is_valid());
143  if(_data == nullptr) {
144  return (reinterpret_cast<_func_pt>(_func))(std::forward<A>(a)...);
145  } else {
146  return (reinterpret_cast<_func_vpt>(_func))(
147  _data, std::forward<A>(a)...);
148  }
149  }
150 
151 private:
152  void* _data{nullptr};
153  void (*_func)() noexcept(NE){nullptr};
154 
155  using _func_t = void (*)() noexcept(NE);
156  using _func_pt = RV (*)(P...) noexcept(NE);
157  using _func_vpt = RV (*)(void*, P...) noexcept(NE);
158 
159  template <typename C>
160  static auto _cls_fn_call_op(void* that, P... p) noexcept(NE) -> RV {
161  EAGINE_ASSERT(that);
162  C& obj = *(static_cast<C*>(that));
163 
164  // NOLINTNEXTLINE(hicpp-braces-around-statements,readability-braces-around-statements)
165  if constexpr(std::is_void_v<RV>) {
166  obj(std::forward<P>(p)...);
167  } else {
168  return obj(std::forward<P>(p)...);
169  }
170  }
171 
172  template <typename C>
173  static auto _cls_fn_call_op_c(void* that, P... p) noexcept(NE) -> RV {
174  EAGINE_ASSERT(that);
175  const C& obj = *(static_cast<const C*>(that));
176 
177  // NOLINTNEXTLINE(hicpp-braces-around-statements,readability-braces-around-statements)
178  if constexpr(std::is_void_v<RV>) {
179  obj(std::forward<P>(p)...);
180  } else {
181  return obj(std::forward<P>(p)...);
182  }
183  }
184 };
