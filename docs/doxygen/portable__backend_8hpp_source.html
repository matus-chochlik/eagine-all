115  template <std::unsigned_integral I>
116  auto _write_one(I value, const type_identity<I>) noexcept -> result {
117  result errors{};
118  const auto encode = [&]() {
119  // clang-format off
120  const char c[16] = {
121  '0','1','2','3','4','5','6','7',
122  '8','9','A','B','C','D','E','F'};
123  // clang-format on
124  errors |= do_sink(c[(value & 0xFU)]);

287  template <std::unsigned_integral I>
288  auto _read_one(I& value, const char delimiter) noexcept -> result {
289  value = I(0);
290  result errors{};
291  auto src{this->string_before(delimiter, 48)};
292  if(src) [[likely]] {
293  const auto skip_len = src.size() + 1;
294  unsigned shift = 0U;
295  while(src) {
296  const char c = extract(src);
297  I frag{};
298  if((c >= '0') && (c <= '9')) {
299  frag = I(c - '0');
300  } else if((c >= 'A') && (c <= 'F')) {
301  frag = I(10 + c - 'A');
302  } else {
303  errors |= deserialization_error_code::invalid_format;
304  return errors;
305  }
306  EAGINE_ASSERT(frag < 16U);
307  value |= I(frag << shift);
308  shift += 4U;
309  src = skip(src, 1);
310  }
311  pop(skip_len);
312  } else {
313  errors |= error_code::not_enough_data;
314  }
315  return errors;
316  }

318  template <std::signed_integral I>
319  auto _read_one(I& value, const char delimiter) noexcept -> result {
320  using U = std::make_unsigned_t<I>;
321  const char sign = extract_or(top_char(), '\0');
322  result errors{};
323  if((sign == '+') || (sign == '-')) [[likely]] {
324  pop(1);
325  U temp{};
326  errors |= _read_one(temp, delimiter);
327  const auto conv{convert_if_fits<I>(temp)};
328  if(conv) [[likely]] {
329  value = (sign == '-') ? -extract(conv) : extract(conv);
330  } else {
331  errors |= deserialization_error_code::invalid_format;
332  }
333  } else {

335  }
