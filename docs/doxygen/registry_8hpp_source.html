32 /// @ingroup msgbus
33 class registry : public main_ctx_object {
34 public:
35  /// @brief Construction from parent main context object.

37 
38  /// @brief Establishes a new endpoint with the specified logger identifier.
39  /// @see emplace
40  [[nodiscard]] auto establish(const identifier log_id) noexcept
41  -> endpoint& {
42  return extract(_add_entry(log_id)._endpoint);
43  }
44 
45  /// @brief Establishes an endpoint and instantiates a service object tied to it.
46  /// @see establish
47  template <typename Service, typename... Args>
48  auto emplace(const identifier log_id, Args&&... args) noexcept
49  -> Service& requires(std::is_base_of_v<service_interface, Service>) {
50  auto& entry = _add_entry(log_id);
51  auto temp{std::make_unique<Service>(
52  extract(entry._endpoint), std::forward<Args>(args)...)};
53  auto& result = extract(temp);
54  entry._service = std::move(temp);
55  return result;
56  }
57 
58  /// @brief Removes a previously emplaced service.
59  void remove(service_interface&) noexcept;
60 
61  auto update() noexcept -> work_done;
62  auto update_all() noexcept -> work_done;
63 
64  auto is_done() noexcept -> bool {
65  return _router.is_done();
66  }
67 
68  void finish() noexcept {
69  _router.finish();
70  }
