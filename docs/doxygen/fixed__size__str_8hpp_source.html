20 /// @brief String with maximum fixed-size internal storage.
21 /// @ingroup string_utils
22 template <span_size_t N>
23 class fixed_size_string {
24 public:
25  /// @brief Default construction.
26  /// @post empty
27  constexpr fixed_size_string() noexcept = default;
28 
29  /// @brief Construction from a pack of characters.
30  template <typename... C>
31  constexpr fixed_size_string(const C... c) noexcept
32  requires(sizeof...(C) == N)
33  : _str{c...} {}
34 
35  /// @brief Construction from a C-string literal.
36  constexpr fixed_size_string(const char (&s)[N]) noexcept {
37  std::copy(std::begin(s), std::end(s), std::begin(_str));
38  }
39 
40  template <span_size_t N1, span_size_t N2>
41  constexpr fixed_size_string(
42  const fixed_size_string<N1>& s1,
43  const fixed_size_string<N2>& s2) noexcept requires(N1 + N2 == N + 1) {
44  std::copy(std::begin(s1._str), std::end(s1._str), _str);
45  std::copy(std::begin(s2._str), std::end(s1._str), _str + N1 - 1);
46  }
47 
48  /// @brief Alias for length type.
49  using size_type = span_size_t;
50 
51  /// @brief Alias for iterator type.
52  using iterator = char*;
53 
54  /// @brief Alias for const iterator type.
55  using const_iterator = const char*;
56 
57  /// @brief Indicates if the string is empty.
58  constexpr auto empty() const noexcept {
59  return _str[0] == '\0';
60  }
61 
62  /// @brief Returns the size of the string.
63  constexpr auto size() const noexcept -> span_size_t {
64  return N - 1;
65  }
66 
67  /// @brief Returns pointer to the internally-stored character data.
68  constexpr auto data() const noexcept {
69  return static_cast<const char*>(_str);
70  }
71 
72  /// @brief Returns pointer to the internally-stored character data.
73  ///
74  /// This function ensures that the internal storage is zero terminated.
75  constexpr auto c_str() const noexcept {
76  return EAGINE_CONSTEXPR_ASSERT(_str[N - 1] == '\0', data());
77  }
78 
79  /// @brief Returns iterator to the start of the string.
80  constexpr auto begin() noexcept -> iterator {
81  return _str;
82  }
83 
84  /// @brief Returns const iterator to the start of the string.
85  constexpr auto begin() const noexcept -> const_iterator {
86  return _str;
87  }
88 
89  /// @brief Returns iterator past the end of the string.
90  constexpr auto end() noexcept -> iterator {
91  return _str + N - 1;
92  }
93 
94  /// @brief Returns const iterator past the end of the string.
95  constexpr auto end() const noexcept -> const_iterator {
96  return _str + N - 1;
97  }
98 
99  /// @brief Returns a view over the internal character data storage.
100  constexpr auto view() const noexcept -> string_view {
101  return {_str, N - 1};
102  }
103 
104  /// @brief Returns a view over the internal character data storage.
105  /// @see view
106  constexpr operator string_view() const noexcept {
107  return view();
108  }
109 
110 private:
111  static_assert(N > 0, "Zero-length fixed size strings are not supported");
112 
113  char _str[N]{};
114 
115  template <span_size_t>
116  friend class fixed_size_string;
117 };

119 /// @brief Creates a fixed_size_string from a C-string literal.
120 /// @ingroup string_utils
121 template <span_size_t N>
122 static constexpr auto make_fixed_size_string(const char (&str)[N]) noexcept {
123  return fixed_size_string<N>(str);
124 }

126 /// @brief Concatenation operator for fixed_size_string.
127 /// @ingroup string_utils
128 template <span_size_t N1, span_size_t N2>
129 static constexpr auto operator+(
130  const fixed_size_string<N1>& s1,
131  const fixed_size_string<N2>& s2) noexcept {
132  return fixed_size_string<N1 + N2 - 1>(s1, s2);
133 }

135 /// @brief Converts a single-digit decimal number into fixed_size_string.
136 /// @ingroup string_utils
137 template <int I>
138 static constexpr auto to_fixed_size_string(const int_constant<I>) noexcept
139  requires((I >= 0) && (I < 10)) {
140  return fixed_size_string<2>(char('0' + I), '\0');
141 }
142 
143 /// @brief Converts a multi-digit decimal number into fixed_size_string.
144 /// @ingroup string_utils
145 template <int I>
146 static constexpr auto to_fixed_size_string(const int_constant<I>) noexcept
147  requires(I > 9) {
148  return to_fixed_size_string(int_constant<I / 10>()) +
149  fixed_size_string<2>(char('0' + I % 10), '\0');
150 }
151 
152 /// @brief Converts a negative decimal number into fixed_size_string.
153 /// @ingroup string_utils
154 template <int I>
155 static constexpr auto to_fixed_size_string(const int_constant<I>) noexcept
156  requires(I < 0) {
157  return fixed_size_string<2>("-") + to_fixed_size_string(int_constant<-I>());
158 }
