282  static void _translate(string_view format, _message_state& msg) {
283  while(auto pos{find_position(format, string_view{"${"})}) {
284  const string_view prev{head(format, extract(pos))};
285  msg.format.append(prev);
286  format = skip(format, extract(pos) + 2);
287  if(const auto end{find_position(format, string_view{"}"})}) {
288  msg.arg_idx[msg.arg_count++] =
289  identifier(head(format, extract(end))).value();
290  msg.format.append("%s");
291  format = skip(format, extract(end) + 1);
292  }
293  }
294  msg.format.append(format);
295  }
