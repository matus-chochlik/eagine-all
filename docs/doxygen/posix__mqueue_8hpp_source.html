290  /// @brief Sends a block of data with the specified priority.
291  auto send(const unsigned priority, const span<const char> blk) noexcept
292  -> auto& {
293  if(is_open()) {
294  errno = 0;
295  ::mq_send(_ohandle, blk.data(), integer(blk.size()), priority);
296  _last_errno = errno;
297  if((_last_errno != 0) && (_last_errno != EAGAIN)) [[unlikely]] {
298  log_error("failed to send message")
299  .arg(EAGINE_ID(name), get_name())
300  .arg(EAGINE_ID(errno), _last_errno)
301  .arg(EAGINE_ID(data), as_bytes(blk));
302  }
303  }
304  return *this;
305  }

307  /// @brief Alias for received message handler.
308  /// @see receive
309  using receive_handler =
310  callable_ref<void(unsigned, span<const char>) noexcept>;
311 
312  /// @brief Receives messages and calls the specified handler on them.
313  auto receive(memory::span<char> blk, const receive_handler handler) noexcept
314  -> bool {
315  if(is_open()) {
316  unsigned priority{0U};
317  errno = 0;
318  const auto received =
319  ::mq_receive(_ihandle, blk.data(), blk.size(), &priority);
320  _last_errno = errno;
321  if(received > 0) {
322  handler(priority, head(blk, received));
323  return true;
324  } else if(
325  (_last_errno != 0) && (_last_errno != EAGAIN) &&
326  (_last_errno != ETIMEDOUT)) [[unlikely]] {
327  log_error("failed to receive message")
328  .arg(EAGINE_ID(name), get_name())
329  .arg(EAGINE_ID(errno), _last_errno);
330  }
331  }
332  return false;
333  }

418  auto send(const message_id msg_id, const message_view& message) noexcept
419  -> bool final {
420  if(is_usable()) [[likely]] {
421  block_data_sink sink(cover(_buffer));
422  default_serializer_backend backend(sink);
423  auto errors = serialize_message(msg_id, message, backend);
424  if(!errors) {
425  std::unique_lock lock{_mutex};
426  _outgoing.push(sink.done());
427  return true;
428  } else {
429  log_error("failed to serialize message");
430  }
431  }
432  return false;
433  }

435  auto fetch_messages(const fetch_handler handler) noexcept
436  -> work_done final {
437  std::unique_lock lock{_mutex};

439  }

441  auto query_statistics(connection_statistics&) noexcept -> bool final {
442  return false;
443  }

445 protected:
446  auto _checkup(posix_mqueue& connect_queue) noexcept -> work_done {
447  some_true something_done{};
448  if(connect_queue.is_usable()) {
449  if(!_data_queue.is_usable()) {
450  if(_reconnect_timeout) {
451  _data_queue.close();
452  _data_queue.unlink();
453  EAGINE_ASSERT(_shared_state);
454 
455  log_debug("connecting to ${name}")
456  .arg(EAGINE_ID(name), connect_queue.get_name());
457 
458  if(!_data_queue.set_name(_shared_state->make_id())
459  .create()
460  .had_error()) {
461 
462  _buffer.resize(connect_queue.data_size());
463 
464  block_data_sink sink(cover(_buffer));
465  default_serializer_backend backend(sink);
466  const auto errors = serialize_message(
467  EAGINE_MSGBUS_ID(pmqConnect),
468  message_view(_data_queue.get_name()),
469  backend);
470  if(!errors) [[likely]] {
471  connect_queue.send(1, as_chars(sink.done()));
472  _buffer.resize(_data_queue.data_size());
473  something_done();
474  } else {
475  log_error("failed to serialize connection name")
476  .arg(EAGINE_ID(client), _data_queue.get_name())
477  .arg(EAGINE_ID(server), connect_queue.get_name());
478  }
479  } else {
480  log_warning("failed to connect to ${server}")
481  .arg(EAGINE_ID(server), connect_queue.get_name());
482  }
483  _reconnect_timeout.reset();
484  }
485  }
486  }
487  return something_done;
488  }
489 
490  auto _receive() noexcept -> work_done {
491  some_true something_done{};
492  if(_data_queue.is_usable()) {
493  while(_data_queue.receive(
494  as_chars(cover(_buffer)),
495  posix_mqueue::receive_handler(
496  EAGINE_THIS_MEM_FUNC_REF(_handle_receive)))) {
497  something_done();
498  }
499  }
500  return something_done;
501  }
502 
503  auto _send() noexcept -> bool {
504  if(_data_queue.is_usable()) {
505  return _outgoing.fetch_all(EAGINE_THIS_MEM_FUNC_REF(_handle_send));
506  }
507  return false;
508  }
509 
510 protected:
511  auto _handle_send(
512  const message_timestamp,
513  const memory::const_block data) noexcept -> bool {
514  return !_data_queue.send(1, as_chars(data)).had_error();
515  }
516 
517  void _handle_receive(
518  const unsigned,
519  const memory::span<const char> data) noexcept {
520  _incoming.push_if(
521  [data](
522  message_id& msg_id, message_timestamp&, stored_message& message) {
523  block_data_source source(as_bytes(data));
524  default_deserializer_backend backend(source);
525  const auto errors = deserialize_message(msg_id, message, backend);
526  return !errors;
527  });
528  }

534  posix_mqueue _data_queue{*this};

675  auto _checkup() noexcept -> work_done {
676  some_true something_done{};
677  if(!_accept_queue.is_usable()) {
678  if(_reconnect_timeout) {
679  _accept_queue.close();
680  _accept_queue.unlink();
681  if(!_accept_queue.create().had_error()) {
682  _buffer.resize(_accept_queue.data_size());
683  something_done();
684  }
685  _reconnect_timeout.reset();
686  }
687  }
688  return something_done;
689  }
690 
691  auto _receive() noexcept -> work_done {
692  some_true something_done{};
693  if(_accept_queue.is_usable()) {
694  while(_accept_queue.receive(
695  as_chars(cover(_buffer)),
696  EAGINE_THIS_MEM_FUNC_REF(_handle_receive))) {
697  something_done();
698  }
699  }
700  return something_done;
701  }
702 
703  void _handle_receive(
704  const unsigned,
705  const memory::span<const char> data) noexcept {
706  _requests.push_if(
707  [data](
708  message_id& msg_id, message_timestamp&, stored_message& message) {
709  block_data_source source(as_bytes(data));
710  default_deserializer_backend backend(source);
711  const auto errors = deserialize_message(msg_id, message, backend);
712  if(is_special_message(msg_id)) [[likely]] {
713  if(msg_id.has_method(EAGINE_ID(pmqConnect))) [[likely]] {
714  return !errors;
715  }
716  }
717  return false;
718  });
719  }
720 
721  memory::buffer _buffer;
722  message_storage _requests;
723  posix_mqueue _accept_queue;
724  timeout _reconnect_timeout{std::chrono::seconds{2}, nothing};
725  std::shared_ptr<posix_mqueue_shared_state> _shared_state;
726 };
