315  auto consume(string_view what, result& errors) noexcept -> bool {
316  errors = {};
317  auto top_str = top_string(what.size());
318  if(top_str.size() < what.size()) {
319  errors |= error_code::not_enough_data;
320  } else {
321  if(are_equal(top_str, what)) {
322  pop(what.size());
323  return true;
324  } else {
325  errors |= error_code::unexpected_data;
326  }
327  }
328  return false;
329  }
330 
331  auto consume(char what, result& errors) noexcept -> bool {
332  return consume(view_one(what), errors);
333  }

335  template <typename Function>
336  void consume_until(
337  Function predicate,
338  const span_size_t step = 256) noexcept {
339  while(const auto pos{_source->scan_until(predicate, step, step)}) {
340  if(extract(pos) > 0) {
341  pop(extract(pos));
342  } else {
343  break;
344  }
345  }
346  }

348  auto require(string_view what) noexcept -> result {
349  result errors{};
350  auto top_str = top_string(what.size());
351  if(top_str.size() < what.size()) {
352  errors |= error_code::not_enough_data;
353  } else {
354  if(are_equal(top_str, what)) {
355  pop(what.size());
356  } else {
357  errors |= error_code::invalid_format;
358  }
359  }
360 
361  return errors;
362  }
