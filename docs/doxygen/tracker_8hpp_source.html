220  auto update() noexcept -> work_done {
221  some_true something_done{};
222  something_done(base::update());
223 
224  if(_should_query_topology) {
225  this->discover_topology();
226  something_done();
227  }
228 
229  if(_should_query_stats) {
230  this->discover_statistics();
231  something_done();
232  }
233 
234  const bool should_query_info{_should_query_info};
235 
236  _tracker.for_each_host_state([&](const auto host_id, auto& host) {
237  _handle_host_change(host_id, host);
238  });
239 
240  _tracker.for_each_instance_state([&](const auto inst_id, auto& inst) {
241  _handle_inst_change(inst_id, inst);
242  });
243 
244  _tracker.for_each_node_state([&](const auto node_id, auto& node) {
245  if(should_query_info) {
246  if(!node.has_known_kind()) {
247  this->query_topology(node_id);
248  }
249  if(!node.host_id()) {
250  this->query_host_id(node_id);
251  }
252  if(!node.has_endpoint_info()) {
253  this->query_endpoint_info(node_id);
254  }
255  if(!node.instance().compiler()) {
256  this->query_compiler_info(node_id);
257  }
258  if(!node.instance().build()) {
259  this->query_build_info(node_id);
260  }
261  if(node.is_responsive()) {
262  if(const auto inst{node.instance_state()}) {
263  if(!inst.application_name()) {
264  this->query_application_name(node_id);
265  }
266  }
267  if(auto host{node.host_state()}) {
268  if(!host.name()) {
269  this->query_hostname(node_id);
270  }
271  if(!host.cpu_concurrent_threads()) {
272  this->query_cpu_concurrent_threads(node_id);
273  }
274  if(!host.total_ram_size()) {
275  this->query_total_ram_size(node_id);
276  }
277  if(!host.total_swap_size()) {
278  this->query_total_swap_size(node_id);
279  }
280 
281  if(node.can_query_system_info()) {
282  if(host.should_query_sensors()) {
283  this->query_sensors(node_id);
284  host.sensors_queried();
285  }
286  }
287  }
288  }
289  }
290 
291  if(node.is_pingable()) {
292  const auto [should_ping, max_time] = node.should_ping();
293  if(should_ping) {
294  this->ping(node_id, max_time);
295  node.pinged();
296  something_done();
297  }
298  }
299  _handle_node_change(node_id, node);
300  });
301 
302  return something_done;
303  }
