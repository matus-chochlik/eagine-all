36  const string_view title,
37  span_size_t total_steps) noexcept = 0;

52  }

55  /// @see unregister_observer

57 
58  /// @brief Un-registers an observer object to be notified.
59  /// @see register_observer
60  virtual void unregister_observer(progress_observer&) noexcept = 0;
61 
62  /// @brief Requests the id for a new activity progress.
63  /// @see finish_activity
64  virtual auto begin_activity(
65  const activity_progress_id_t parent_id,
66  const string_view title,
67  span_size_t total_steps) -> activity_progress_id_t = 0;
68 
69  /// @brief Specifies the current number of steps done in the activity.
70  /// @see advance_progress
71  ///
72  /// The return value indicates if the activity was canceled, true values
73  /// means that the activity should continue, false means that the activity
74  /// was canceled and should discontinue.
75  virtual auto update_progress(
76  const activity_progress_id_t activity_id,
77  span_size_t current) noexcept -> bool = 0;
78 
79  /// @brief Advances the current number of steps done in the activity.
80  /// @see update_progress
81  ///
82  /// The return value indicates if the activity was canceled, true values
83  /// means that the activity should continue, false means that the activity
84  /// was canceled and should discontinue.
85  virtual auto advance_progress(
86  const activity_progress_id_t activity_id,
87  span_size_t increment) noexcept -> bool = 0;
88 
89  /// @brief Indicates that an activity has finished.
90  /// @see begin_activity
91  virtual void finish_activity(
92  const activity_progress_id_t activity_id) noexcept = 0;
93 
94  /// @brief Assigns a function to be called on progress update.

96  virtual void set_update_callback(
97  const callable_ref<bool() noexcept>,

99 
100  /// @brief Resets the function called on progress update.
101  /// @see set_update_callback
102  virtual void reset_update_callback() noexcept = 0;
103 };
