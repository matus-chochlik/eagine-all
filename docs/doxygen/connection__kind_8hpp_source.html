16 namespace eagine::msgbus {
17 //------------------------------------------------------------------------------
18 /// @brief Message bus connection kind bits enumeration.
19 /// @ingroup msgbus
20 /// @see connection_kinds
21 enum class connection_kind : std::uint8_t {
22  /// @brief Unknown connection kind.
23  unknown = 0U,
24  /// @brief In-process connection (cannot be used for inter-process communication).
25  in_process = 1U << 0U,
26  /// @brief Inter-process connection for local communication.
27  local_interprocess = 1U << 1U,
28  /// @brief Inter-process connection for remote communucation
29  remote_interprocess = 1U << 2U
30 };
31 //------------------------------------------------------------------------------
32 #if !EAGINE_CXX_REFLECTION

36  const Selector) noexcept {

38  {{"unknown", connection_kind::unknown},
39  {"in_process", connection_kind::in_process},
40  {"local_interprocess", connection_kind::local_interprocess},
41  {"remote_interprocess", connection_kind::remote_interprocess}}};
42 }
43 #endif
