45 class basic_logger : protected BackendGetter {
46 public:
47  /// @brief The entry type for the specified log_event_severity.
48  /// @see log_entry

50  template <log_event_severity severity>

52  conditional_t<is_log_level_enabled_v<severity>, log_entry, no_log_entry>;
53 
54  /// @brief Returns a pointer to the backend of this logger object.
55  auto backend() const noexcept {
56  return _backend_getter()();

58 
59  /// @brief Returns the unique id of this logger instance.
60  auto instance_id() const noexcept -> logger_instance_id {
61  return reinterpret_cast<logger_instance_id>(this);
62  }
63 
64  auto configure(application_config& config) const -> bool {
65  if(auto lbe{backend()}) {
66  extract(lbe).configure(config);

75  std::is_nothrow_move_constructible_v<BackendGetter>)

208  void log_chart_sample(
209  [[maybe_unused]] const identifier source,
210  [[maybe_unused]] const identifier series,
211  [[maybe_unused]] const float value) const noexcept {
212  if constexpr(is_log_level_enabled_v<log_event_severity::stat>) {
213  if(auto lbe{_entry_backend(source, log_event_severity::stat)}) {
214  extract(lbe).log_chart_sample(
215  source, instance_id(), series, value);
216  }
217  }
218  }
219 
220  auto _entry_backend(
221  const identifier source,
222  const log_event_severity severity) const noexcept -> logger_backend* {
223  if(is_log_level_enabled(severity)) {
224  if(auto lbe{backend()}) {
225  return lbe->entry_backend(source, severity);
226  }
227  }
228  return nullptr;
229  }
230 
231 private:
232  auto _backend_getter() const noexcept -> const BackendGetter& {
233  return *this;
234  }

241 /// This class is used as the base class for other classes that need to do

243 template <typename BackendGetter>
244 class named_logging_object : public basic_logger<BackendGetter> {
245  using base = basic_logger<BackendGetter>;
246 
247 public:
248  /// @brief The entry type for the specified log_event_severity.
249  /// @see log_entry
250  /// @see no_log_entry
251  template <log_event_severity severity>
252  using entry_type = std::
253  conditional_t<is_log_level_enabled_v<severity>, log_entry, no_log_entry>;
254 
255  using base::log_lifetime;
256 
257  /// @brief Constructor from identifier and backend_getter object.
258  /// @note For internal use-only.
259  named_logging_object(
260  const identifier id,
261  BackendGetter backend_getter) noexcept
262  : base{BackendGetter(std::move(backend_getter))}
263  , _object_id{id} {
264  log_lifetime(_object_id, "${self} created with ${backend} backend")
265  .arg(EAGINE_ID(backend), this->backend())
266  .arg(EAGINE_ID(self), EAGINE_ID(LogId), _object_id);
267  }
268 
269  /// @brief Constructor from logger id and parent logging object.
270  named_logging_object(
271  const identifier id,
272  const named_logging_object& parent) noexcept
273  : base{static_cast<const base&>(parent)}
274  , _object_id{id} {
275  log_lifetime(_object_id, "created as a child of ${parent}")
276  .arg(EAGINE_ID(parent), EAGINE_ID(LogId), parent._object_id);
277  }
278 
279  /// @brief Construct logging object without backend.

287  }
288 
289  /// @brief Copy constructor.

293  log_lifetime(_object_id, "being copied");
294  }

421  auto make_log_entry(

427  auto make_log_stream(const log_event_severity severity) const noexcept {
428  return base::make_log_stream(_object_id, severity);
429  }
