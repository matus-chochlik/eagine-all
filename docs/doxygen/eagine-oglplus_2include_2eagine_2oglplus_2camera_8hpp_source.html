21 /// @brief Class representing a camera orbiting around its target.
22 /// @ingroup gl_utils
23 class orbiting_camera {
24 public:
25  /// @brief Sets the target position.
26  auto set_target(const vec3 target) noexcept -> auto& {
27  _target = target;
28  return *this;
29  }
30 
31  /// @brief Sets the y-axis FOV angle.
32  auto set_fov(const radians_t<float> angle) noexcept -> auto& {
33  _fov = angle;
34  return *this;
35  }
36 
37  /// @brief Sets the distance of the near plane.
38  auto set_near(valid_if_positive<float> dist) noexcept -> auto& {
39  _near = dist.value();
40  return *this;
41  }
42 
43  /// @brief Sets the distance of the far plane.
44  auto set_far(valid_if_positive<float> dist) noexcept -> auto& {
45  _far = dist.value();
46  return *this;
47  }
48 
49  /// @brief Set minimal orbit value.
50  auto set_orbit_min(valid_if_positive<float> orbit) noexcept -> auto& {
51  _orbit_min = orbit.value();
52  return *this;
53  }
54 
55  /// @brief Set maximal orbit value.
56  auto set_orbit_max(valid_if_positive<float> orbit) noexcept -> auto& {
57  _orbit_max = orbit.value();
58  return *this;
59  }
60 
61  /// @brief Set the factor for calculating the orbit between min and max.
62  auto set_orbit_factor(const float factor) noexcept -> auto& {
63  _orbit_factor = factor;
64  return *this;
65  }
66 
67  /// @brief Sets the azimuth angle (longitude)
68  auto set_azimuth(const radians_t<float> turns) noexcept -> auto& {
69  _turns = turns;
70  return *this;
71  }
72 
73  /// @brief Sets the elevation angle (latitude)
74  auto set_elevation(const radians_t<float> pitch) noexcept -> auto& {
75  _pitch = pitch;
76  return *this;
77  }
78 
79  /// @brief Returns the near distance.
80  /// @see far
81  /// @see midpoint
82  auto near() const noexcept -> float {
83  return _near;
84  }
85 
86  /// @brief Returns the far distance.
87  /// @see near
88  /// @see midpoint
89  /// @see skybox_distance
90  auto far() const noexcept -> float {
91  return _far;
92  }
93 
94  /// @brief Returns the midpoint distance between near and far range.
95  /// @see near
96  /// @see far
97  /// @see skybox_distance
98  auto midpoint() const noexcept -> float {
99  return (near() + far()) * 0.5F;
100  }
101 
102  /// @brief Returns the y-axis FOV angle
103  auto fov() const noexcept -> radians_t<float> {
104  return _fov;
105  }
106 
107  /// @brief Returns the orbit altitude value.
108  /// @see azimuth
109  /// @see elevation
110  /// @see near
111  /// @see far
112  auto orbit() const noexcept -> float {
113  return smooth_lerp(_orbit_min, _orbit_max, _orbit_factor);
114  }
115 
116  /// @brief Returns the prefered distance to the farthest part of a skybox.
117  /// @see near
118  /// @see midpoint
119  /// @see far
120  /// @see orbit
121  auto skybox_distance() const noexcept -> float {
122  return far() - orbit();
123  }
124 
125  /// @brief Returns the azimuth angle (longitude).
126  /// @see orbit
127  /// @see elevation
128  auto azimuth() const noexcept -> radians_t<float> {
129  return _turns;
130  }
131 
132  /// @brief Returns the elevation angle (latitude).
133  /// @see orbit
134  /// @see azimuth
135  auto elevation() const noexcept -> radians_t<float> {
136  return _pitch;
137  }
138 
139  /// @brief Returns the target position.
140  /// @see position
141  auto target() const noexcept -> vec3 {
142  return _target;
143  }
144 
145  /// @brief Returns the target-to-camera direction vector.
146  /// @see camera_to_target_direction
147  auto target_to_camera_direction() const noexcept -> vec3;
148 
149  /// @brief Returns the camera-to-target direction vector.
150  /// @see target_to_camera_direction
151  auto camera_to_target_direction() const noexcept -> vec3 {
152  return -target_to_camera_direction();
153  }
154 
155  /// @brief Returns the position of the camera.
156  /// @see target
157  auto position() const noexcept -> vec3 {
158  return target() + target_to_camera_direction() * orbit();
159  }
160 
161  auto perspective_matrix_ctr(const float aspect) const noexcept {
162  return matrix_perspective::y(_fov, aspect, _near, _far);
163  }
164 
165  /// @brief Returns the perspective matrix for the given aspect ratio.
166  auto perspective_matrix(const float aspect) const noexcept {
167  return perspective_matrix_ctr(aspect)();
168  }
169 
170  auto transform_matrix_ctr() const noexcept {
171  return matrix_orbiting_y_up(_target, orbit(), azimuth(), elevation());
172  }
173 
174  /// @brief Returns the camera transformation matrix.
175  auto transform_matrix() const noexcept {
176  return transform_matrix_ctr()();
177  }
178 
179  /// @brief Returns the camera matrix (perspective * projection).
180  auto matrix(const float aspect) const noexcept {
181  return perspective_matrix_ctr(aspect) * transform_matrix_ctr();
182  }
183 
184  /// @brief Returns the 3D position of a point (in NDC) on the target plane.
185  auto target_plane_point(
186  const float ndcx,
187  const float ndcy,
188  const float aspect) const noexcept -> optionally_valid<vec3>;
189 
190  /// @brief Returns a ray from the camera through a point on the target plane.
191  auto pointer_ray(const float ndcx, const float ndcy, const float aspect)
192  const noexcept -> optionally_valid<line>;
193 
194  auto grab_sphere_radius() const noexcept -> float;
195  auto grab_sphere() const noexcept -> sphere;
196 
197 private:
198  vec3 _target{};
199  radians_t<float> _fov{right_angle_()};
200 
201  float _near{0.5F};
202  float _far{10.F};
203 
204 protected:
205  radians_t<float> _turns{};
206  radians_t<float> _pitch{};
207 
208  float _orbit_min{1.5F};
209  float _orbit_max{5.5F};
210  float _orbit_factor = 0.50F;
211 };
