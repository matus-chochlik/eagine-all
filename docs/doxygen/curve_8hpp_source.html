23 /// @brief A sequence of Bezier curves, possibly connected at end points.
24 /// @ingroup math
25 /// @see CubicBezierLoop
26 ///
27 /// This class stores the data for a sequence of connected Bezier curves
28 /// of a given @c Order. The begin of the i-th curve (segment) is equal
29 /// to the end of the (i-1)-th curve and the end of the i-th curve is
30 /// equal to the begin of the (i+1)-th curve. Between the begin and
31 /// end (control) points there is a fixed number (Order - 1) of curve control
32 /// points. The curves pass through the begin and end and are influenced
33 /// by the other control points.
34 template <typename Type, typename Parameter, span_size_t Order>
35 class bezier_curves {
36 public:
37  /// @brief Returns whether the curves made from points are connected.
38  static auto are_connected(const std::vector<Type>& points) noexcept
39  -> bool {
40  return ((points.size() - 1) != Order) &&
41  ((points.size() - 1) % Order) == 0;
42  }
43 
44  /// @brief Returns true if the individual curves are connected.
45  auto is_connected() const noexcept -> bool {
46  return _connected;
47  }
48 
49  static auto are_separated(const std::vector<Type>& points) noexcept
50  -> bool {
51  return (points.size() % (Order + 1)) == 0;
52  }
53 
54  /// @brief Returns true if the individual curves are disconnected.
55  auto is_separated() const noexcept -> bool {
56  return !_connected;
57  }
58 
59  /// @brief checks if the sequence of control points is OK for this curve type.
60  static auto points_are_ok(const std::vector<Type>& points) noexcept
61  -> bool {
62  return !points.empty() &&
63  (are_connected(points) || are_separated(points));
64  }
65 
66  /// @brief Creates the bezier curves from the control @c points.
67  /// @pre points_are_ok(points)
68  ///
69  /// The number of points must be ((C * Order) + 1) or (C * (Order + 1))
70  /// where @em C is the number of curves (segments) in the sequence.
71  /// If both of the above are true then the curves are considered
72  /// to be connected.
73  bezier_curves(std::vector<Type> points)
74  : _points{std::move(points)}
75  , _connected{are_connected(_points)} {
76  EAGINE_ASSERT(points_are_ok(_points));
77  }
78 
79  /// @brief Creates the bezier curves from the control @c points.
80  /// @pre points_are_ok(points)
81  /// @pre are_connected(points) == connected
82  ///
83  /// The number of points must be ((C * Order) + 1) and connected
84  /// or (C * (Order + 1)) and not(connected),
85  /// where @em C is the number of curves (segments) in the sequence.
86  bezier_curves(std::vector<Type> points, const bool connected)
87  : _points{std::move(points)}
88  , _connected{connected} {
89  EAGINE_ASSERT(points_are_ok(_points));
90  EAGINE_ASSERT(are_connected(_points) == _connected);
91  }
92 
93  /// @brief Creates the bezier curves from the control @c points.
94  /// @pre points_are_ok(points)
95  ///
96  /// The number of points must be ((C * Order) + 1) or (C * (Order + 1))
97  /// where @em C is the number of curves (segments) in the sequence.
98  /// If both of the above are true then the curves are considered
99  /// to be connected.
100  template <typename P, typename S>
101  bezier_curves(memory::basic_span<const Type, P, S> points)
102  : _points{points.begin(), points.end()}
103  , _connected{are_connected(_points)} {
104  EAGINE_ASSERT(points_are_ok(_points));
105  }
106 
107  /// @brief Creates the bezier curves from the control @c points.
108  /// @pre points_are_ok(points)
109  /// @pre are_connected(points) == connected
110  ///
111  /// The number of points must be ((C * Order) + 1) or (C * (Order + 1))
112  /// where @em C is the number of curves (segments) in the sequence.
113  /// If both of the above are true then the curves are considered
114  /// to be connected.
115  template <typename P, typename S>
116  bezier_curves(
117  memory::basic_span<const Type, P, S> points,
118  const bool connected)
119  : _points{points.begin(), points.end()}
120  , _connected{connected} {
121  EAGINE_ASSERT(points_are_ok(_points));
122  EAGINE_ASSERT(are_connected(_points) == _connected);
123  }
124 
125  auto segment_step() const noexcept -> span_size_t {
126  EAGINE_ASSERT(points_are_ok(_points));
127  return _connected ? Order : Order + 1;
128  }
129 
130  /// @brief Returns the count of individual curves in the sequence.
131  auto segment_count() const noexcept -> span_size_t {
132  EAGINE_ASSERT(points_are_ok(_points));
133 
134  return _connected ? span_size(_points.size() - 1) / Order
135  : span_size(_points.size()) / (Order + 1);
136  }
137 
138  /// @brief Returns the contol points of the curve.
139  auto control_points() const noexcept -> const std::vector<Type>& {
140  return _points;
141  }
142 
143  /// @brief Wraps the parameter value to [0.0, 1.0].
144  static auto wrap(Parameter t) noexcept -> Parameter {
145  const Parameter zero{0};
146  const Parameter one{1};
147  if(t < zero) {
148  t += std::floor(std::fabs(t)) + one;
149  } else if(t > one) {
150  t -= std::floor(t);
151  }
152  EAGINE_ASSERT(t >= zero && t <= one);
153  return t;
154  }
155 
156  /// @brief Gets the point on the curve at position t (must be in (0.0, 1.0)).
157  auto position01(Parameter t) const noexcept -> Type {
158  const Parameter zero{0};
159  const Parameter one{1};
160 
161  if(t >= one) {
162  t = zero;
163  }
164  EAGINE_ASSERT(t >= zero && t < one);
165 
166  const auto toffs = t * Parameter(segment_count());
167  const auto t_sub = toffs - std::floor(toffs);
168  const auto poffs = span_size_t(toffs) * segment_step();
169  EAGINE_ASSERT(poffs < span_size(_points.size()) - Order);
170 
171  return _bezier(t_sub, skip(view(_points), poffs));
172  }
173 
174  /// @brief Gets the point on the curve at position t wrapped to [0.0, 1.0].
175  auto position(const Parameter t) const noexcept -> Type {
176  return position01(wrap(t));
177  }
178 
179  /// @brief Makes a sequence of points on the curve (n points per segment).
180  void approximate(
181  std::vector<Type>& dest,
182  const valid_if_positive<span_size_t>& n) const noexcept {
183  const auto sstep = segment_step();
184  const auto s = segment_count();
185 
186  dest.resize(integer(s * extract(n) + 1));
187 
188  auto p = dest.begin();
189  const Parameter t_step = Parameter(1) / Parameter(extract(n));
190 
191  for(const auto i : integer_range(s)) {
192  const auto poffs = i * sstep;
193  auto t_sub = Parameter(0);
194  for([[maybe_unused]] const auto j : integer_range(extract(n))) {
195  EAGINE_ASSERT(p != dest.end());
196  *p = Type(_bezier(t_sub, skip(view(_points), poffs)));
197  ++p;
198 
199  t_sub += t_step;
200  }
201  }
202  EAGINE_ASSERT(p != dest.end());
203  *p = _points.back();
204  ++p;
205  EAGINE_ASSERT(p == dest.end());
206  }
207 
208  /// @brief Returns a sequence of points on the curve (n points per segment).
209  auto approximate(const valid_if_positive<span_size_t> n) const
210  -> std::vector<Type> {
211  std::vector<Type> result;
212  approximate(result, n);
213  return result;
214  }
215 
216  /// @brief Returns a derivative of this curve
217  auto derivative() const noexcept
218  -> bezier_curves<Type, Parameter, Order - 1> {
219  const auto sstep = segment_step();
220  const auto s = segment_count();
221 
222  std::vector<Type> new_points(integer(s * Order));
223  auto p = new_points.begin();
224 
225  for(const auto i : integer_range(s)) {
226  for(const auto j : integer_range(Order)) {
227  const auto k = i * sstep + j;
228  EAGINE_ASSERT(p != new_points.end());
229  *p = (_points[integer(k + 1)] - _points[integer(k)]) * Order;
230  ++p;
231  }
232  }
233  EAGINE_ASSERT(p == new_points.end());
234 
235  return {std::move(new_points), false};
236  }
237 
238 private:
239  static_assert(Order > 0);
240 
241  std::vector<Type> _points;
242  bezier_t<Type, Parameter, Order + 1> _bezier{};
243  bool _connected;
244 };

246 /// @brief A closed smooth cubic Bezier spline passing through all input points.
247 /// @ingroup math
248 ///
249 /// This class constructs a closed sequence of Bezier curves that are smooth
250 /// at the curve connection points. The control points between the begin
251 /// and end points of each segment are calculated automatically to make
252 /// the transition between the individual segments smooth.
253 template <typename Type, typename Parameter>
254 class cubic_bezier_loop : public bezier_curves<Type, Parameter, 3> {
255 public:
256  /// @brief Creates a loop passing through the sequence of the input points.
257  template <typename P, typename S>
258  cubic_bezier_loop(
259  const memory::basic_span<const Type, P, S> points,
260  const Parameter r = Parameter(1) / Parameter(3))
261  : bezier_curves<Type, Parameter, 3>(_make_cpoints(points, r)) {}
262 
263 private:
264  template <typename P, typename S>
265  static auto _make_cpoints(
266  const memory::basic_span<const Type, P, S> points,
267  const Parameter r) -> std::vector<Type> {
268  span_size_t i = 0, n = points.size();
269  EAGINE_ASSERT(n != 0);
270 
271  std::vector<Type> result(integer(n * 3 + 1));
272  auto ir = result.begin();
273 
274  while(i != n) {
275  const auto a = (n + i - 1) % n;
276  const auto b = i;
277  const auto c = (i + 1) % n;
278  const auto d = (i + 2) % n;
279 
280  EAGINE_ASSERT(ir != result.end());
281  *ir = points[b];
282  ++ir;
283  EAGINE_ASSERT(ir != result.end());
284  *ir = Type(points[b] + (points[c] - points[a]) * r);
285  ++ir;
286  EAGINE_ASSERT(ir != result.end());
287  *ir = Type(points[c] + (points[b] - points[d]) * r);
288  ++ir;
289  ++i;
290  }
291  EAGINE_ASSERT(ir != result.end());
292  *ir = points[0];
293  ++ir;
294  EAGINE_ASSERT(ir == result.end());
295 
296  return result;
297  }
298 };
