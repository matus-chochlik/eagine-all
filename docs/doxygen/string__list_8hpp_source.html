28 static inline auto element_header_size(const string_view elem) noexcept
29  -> span_size_t {
30  return extract_or(
31  mbs::decode_sequence_length(mbs::make_cbyte_span(elem)), 0);
32 }
33 //------------------------------------------------------------------------------
34 static inline auto element_value_size(
35  const string_view elem,
36  const span_size_t l) noexcept -> span_size_t {
37  return extract_or(
38  mbs::do_decode_code_point(mbs::make_cbyte_span(elem), l), 0U);
39 }
40 //------------------------------------------------------------------------------
41 static inline auto element_value_size(const string_view elem) noexcept
42  -> span_size_t {
43  return element_value_size(elem, elem.size());
44 }
45 //------------------------------------------------------------------------------
46 static inline auto rev_seek_header_start(const string_view elem) noexcept
47  -> span_size_t {
48  for(auto i = elem.rbegin(); i != elem.rend(); ++i) {
49  if(mbs::is_valid_head_byte(byte(*i))) {
50  return elem.rend() - i - 1;
51  }
52  }
53  return 0;
54 }

56 static inline auto front_value(const string_view list) noexcept -> string_view {
57  const span_size_t k = element_header_size(list);
58  const span_size_t l = element_value_size(list, k);
59  return slice(list, k, l);
60 }
61 //------------------------------------------------------------------------------
62 static inline auto back_value(const string_view list) noexcept -> string_view {
63  const span_size_t i = rev_seek_header_start(list);
64  const string_view header = skip(list, i);
65  const span_size_t k = element_header_size(header);
66  const span_size_t l = element_value_size(header, k);
67  return slice(list, i - l, l);
68 }
69 //------------------------------------------------------------------------------
70 static inline auto pop_back(const string_view list) noexcept -> string_view {
71  const span_size_t i = rev_seek_header_start(list);
72  string_view header = skip(list, i);
73  const span_size_t k = element_header_size(header);
74  const span_size_t l = element_value_size(header, k);
75  EAGINE_ASSERT(i >= k + l);
76  return head(list, i - k - l);
77 }
78 //------------------------------------------------------------------------------
79 static inline void push_back(
80  std::string& list,
81  const string_view value) noexcept {
82  const span_size_t vl = value.size();
83  const std::string elen = encode_length(vl);
84  const std::size_t nl = list.size() + elen.size() * 2 + std_size(vl);
85  if(list.capacity() < nl) {
86  list.reserve(nl);
87  }
88  list.append(elen);
89  list.append(value.data(), integer(vl));
90  list.append(elen);
91 }

102  static inline auto _fit(const string_view s) noexcept -> string_view {
103  const span_size_t hs = element_header_size(s);
104  const span_size_t vs = element_value_size(s, hs);
105  EAGINE_ASSERT(s.size() >= hs + vs + hs);
106  return {s.data(), hs + vs + hs};
107  }
108 
109  static inline auto _fit(const char* ptr, const span_size_t max_size) noexcept
110  -> string_view {
111  return _fit(string_view(ptr, max_size));
112  }
113 
114  static inline auto _rev_fit(
115  const string_view s,
116  [[maybe_unused]] const span_size_t rev_sz) noexcept -> string_view {
117  const span_size_t hs = element_header_size(s);
118  const span_size_t vs = element_value_size(s, hs);
119  EAGINE_ASSERT(rev_sz >= hs + vs);
120  return {s.data() - hs - vs, hs + vs + hs};
121  }
122 
123  static inline auto _rev_fit(
124  const char* ptr,
125  const span_size_t rev_sz,
126  const span_size_t foot_sz) noexcept -> string_view {
127  return _rev_fit(string_view(ptr, foot_sz), rev_sz);
128  }
129 
130 public:
131  element(const char* ptr, const span_size_t max_size) noexcept
132  : string_view(_fit(ptr, max_size)) {}
133 
134  element(
135  const char* ptr,
136  const span_size_t rev_sz,
137  const span_size_t foot_sz) noexcept
138  : string_view(_rev_fit(ptr, rev_sz, foot_sz)) {}
139 
140  auto header_size() const noexcept -> span_size_t {
141  return element_header_size(_base());
142  }
143 
144  auto header() const noexcept -> string_view {
145  return {data(), header_size()};
146  }
147 
148  auto value_size() const noexcept -> span_size_t {
149  return element_value_size(header());
150  }
151 
152  auto value_data() const noexcept -> const char* {
153  return data() + header_size();
154  }
155 
156  auto value() const noexcept -> string_view {
157  return {value_data(), value_size()};
158  }
159 
160  auto footer_size() const noexcept -> span_size_t {
161  return element_header_size(_base());
162  }
163 
164  auto footer() const noexcept -> string_view {
165  return {data() + header_size() + value_size(), header_size()};
166  }

169 template <typename Func>
170 static inline void for_each_elem(const string_view list, Func func) noexcept {
171  span_size_t i = 0;
172  bool first = true;
173  while(i < list.size()) {
174  element elem(list.data() + i, list.size() - i);
175  func(elem, first);
176  i += elem.size();
177  first = false;
178  }
179 }
180 //------------------------------------------------------------------------------
181 template <typename Func>
182 static inline void for_each(const string_view list, Func func) noexcept {
183  auto adapted_func = [&func](const element& elem, bool) {
184  func(elem.value());
185  };
186  for_each_elem(list, adapted_func);
187 }
188 //------------------------------------------------------------------------------
189 template <typename Func>
190 static inline void rev_for_each_elem(
191  const string_view list,
192  Func func) noexcept {
193  span_size_t i = list.size() - 1;
194  bool first = true;
195  while(i > 0) {
196  while(!mbs::is_valid_head_byte(byte(list[i]))) {
197  EAGINE_ASSERT(i > 0);
198  --i;
199  }
200  element elem(list.data() + i, i, list.size() - i);
201  func(elem, first);
202  i -= elem.header_size() + elem.value_size() + 1;
203  first = false;
204  }
205 }
206 //------------------------------------------------------------------------------
207 template <typename Func>
208 static inline void rev_for_each(const string_view list, Func func) noexcept {
209  auto adapted_func = [&func](const element& elem, bool) {
210  func(elem.value());
211  };
212  rev_for_each_elem(list, adapted_func);
213 }

215 static inline auto split_into(
216  std::string& dst,
217  const string_view str,
218  const string_view sep) noexcept -> span_size_t {
219  span_size_t cnt = 0;
220  for_each_delimited(str, sep, [&dst, &cnt](const auto& x) {
221  push_back(dst, x);
222  ++cnt;
223  });
224  return cnt;
225 }
226 //------------------------------------------------------------------------------
227 static inline auto split(string_view str, string_view sep) noexcept
228  -> std::tuple<std::string, span_size_t> {
229  std::string res;
230  const auto cnt = split_into(res, str, sep);
231  return std::make_tuple(std::move(res), cnt);
232 }
233 //------------------------------------------------------------------------------
234 template <typename Func>
235 static inline auto for_each_separated_c_str(
236  const char* str,
237  const char sep,
238  Func func) noexcept -> span_size_t {
239  span_size_t cnt = 0;
240  const char* bgn = str;
241  const char* pos = bgn;
242  if(sep != '\0') {
243  while(bool(str) && (*pos != '\0')) {
244  if(*pos == sep) {
245  if(pos - bgn > 0) {
246  func(string_view(bgn, pos - bgn));
247  ++cnt;
248  bgn = ++pos;
249  } else {
250  break;
251  }
252  } else {
253  ++pos;
254  }
255  }
256  } else {
257  while(bool(str)) {
258  if((*pos == sep) || (*pos == char(0))) {
259  if(pos - bgn > 1) {
260  func(string_view(bgn, pos - bgn));
261  ++cnt;
262  bgn = ++pos;
263  } else {
264  break;
265  }
266  } else {
267  ++pos;
268  }
269  }
270  }
271  return cnt;
272 }
273 //------------------------------------------------------------------------------
274 static inline auto split_c_str_into(
275  std::string& dst,
276  const char* str,
277  const char sep) noexcept -> span_size_t {
278  return for_each_separated_c_str(
279  str, sep, [&dst](auto elem) { push_back(dst, elem); });
280 }
281 //------------------------------------------------------------------------------
282 static inline auto split_c_str(const char* str, const char sep) noexcept
283  -> std::tuple<std::string, span_size_t> {
284  std::string res;
285  const auto cnt = split_c_str_into(res, str, sep);
286  return std::make_tuple(std::move(res), cnt);
287 }

289 static inline auto join(
290  const string_view list,
291  const string_view sep,
292  const bool trail_sep) noexcept -> std::string {
293  const span_size_t slen = sep.size();
294  span_size_t len = trail_sep ? slen : 0;
295  const auto get_len = [&len, slen](const element& elem, bool first) {
296  if(!first) {
297  len += slen;
298  }
299  len += elem.value_size();
300  };
301  for_each_elem(list, get_len);
302 
303  std::string res;
304  res.reserve(integer(len));
305 
306  auto fill = [&res, sep](const element& elem, bool first) {
307  if(!first) {
308  append_to(res, sep);
309  }
310  res.append(elem.value_data(), integer(elem.value_size()));
311  };
312  for_each_elem(list, fill);
313 
314  if(trail_sep) {
315  append_to(res, sep);
316  }
317  EAGINE_ASSERT(res.size() == std_size(len));
318 
319  return res;
320 }
321 //------------------------------------------------------------------------------
322 static inline auto join(const string_view list, const string_view sep) noexcept
323  -> std::string {
324  return join(list, sep, false);
325 }

327 template <typename Iter>
328 class iterator {
329 public:
330  using difference_type = std::ptrdiff_t;
331  using value_type = string_view;
332  using reference = const value_type&;
333  using pointer = const value_type*;
334  using iterator_category = std::forward_iterator_tag;
335 
336  iterator(Iter pos) noexcept
337  : _pos{pos} {}
338 
339  friend auto operator==(iterator a, iterator b) noexcept {
340  return a._pos == b._pos;
341  }
342 
343  friend auto operator!=(iterator a, iterator b) noexcept {
344  return a._pos != b._pos;
345  }
346 
347  friend auto operator<(iterator a, iterator b) noexcept {
348  return a._pos < b._pos;
349  }
350 
351  auto operator*() const noexcept -> reference {
352  _update();
353  return _tmp;
354  }
355 
356  auto operator->() const noexcept -> pointer {
357  _update();
358  return &_tmp;
359  }
360 
361  auto operator++() noexcept -> auto& {
362  span_size_t ll = _len_len();
363  span_size_t vl = _val_len(ll);
364  _pos += ll + vl + ll;
365  _tmp = string_view();
366  return *this;
367  }
368 
369  auto operator++(int) noexcept -> iterator {
370  iterator result = *this;
371  ++(*this);
372  return result;
373  }
374 
375 private:
376  Iter _pos;
377  mutable string_view _tmp;
378 
379  auto _b() const noexcept -> byte {
380  return byte(*_pos);
381  }
382 
383  auto _len_len() const noexcept -> span_size_t {
384  byte b = _b();
385  EAGINE_ASSERT(mbs::is_valid_head_byte(b));
386  return extract(mbs::do_decode_sequence_length(b));
387  }
388 
389  auto _val_len(const span_size_t ll) const noexcept -> span_size_t {
390  string_view el{&*_pos, ll};
391  return extract_or(
392  mbs::do_decode_code_point(mbs::make_cbyte_span(el), ll), 0U);
393  }
394 
395  void _update() const noexcept {
396  if(_tmp.size() == 0) {
397  span_size_t ll = _len_len();
398  span_size_t vl = _val_len(ll);
399  _tmp = string_view{&*(_pos + ll), vl};
400  }
401  }
402 };

405 class rev_iterator {
406 public:
407  using difference_type = std::ptrdiff_t;
408  using value_type = string_view;
409  using reference = const value_type&;
410  using pointer = const value_type*;
411  using iterator_category = std::forward_iterator_tag;
412 
413  rev_iterator(Iter pos) noexcept
414  : _pos{pos} {}
415 
416  friend auto operator==(rev_iterator a, rev_iterator b) noexcept {
417  return a._pos == b._pos;
418  }
419 
420  friend auto operator!=(rev_iterator a, rev_iterator b) noexcept {
421  return a._pos != b._pos;
422  }
423 
424  friend auto operator<(rev_iterator a, rev_iterator b) noexcept {
425  return a._pos > b._pos;
426  }
427 
428  auto operator*() const noexcept -> reference {
429  _update();
430  return _tmp;
431  }
432 
433  auto operator->() const noexcept -> pointer {
434  _update();
435  return &_tmp;
436  }
437 
438  auto operator++() noexcept -> auto& {
439  _rseek_head();
440  span_size_t ll = _len_len();
441  span_size_t vl = _val_len(ll);
442  _pos -= ll + vl + 1;
443  _tmp = string_view();
444  return *this;
445  }
446 
447  auto operator++(int) noexcept -> rev_iterator {
448  rev_iterator result = *this;
449  ++(*this);
450  return result;
451  }
452 
453 private:
454  mutable Iter _pos;
455  mutable string_view _tmp;
456 
457  auto _b() const noexcept -> byte {
458  return byte(*_pos);
459  }
460 
461  void _rseek_head() const noexcept {
462  while(!mbs::is_valid_head_byte(_b())) {
463  --_pos;
464  }
465  }
466 
467  auto _len_len() const noexcept -> span_size_t {
468  byte b = _b();
469  EAGINE_ASSERT(mbs::is_valid_head_byte(b));
470  return extract(mbs::do_decode_sequence_length(b));
471  }
472 
473  auto _val_len(span_size_t ll) const noexcept -> span_size_t {
474  string_view el{&*_pos, ll};
475  return extract_or(
476  mbs::do_decode_code_point(mbs::make_cbyte_span(el), ll), 0U);
477  }
478 
479  void _update() const noexcept {
480  if(_tmp.size() == 0) {
481  _rseek_head();
482  span_size_t ll = _len_len();
483  span_size_t vl = _val_len(ll);
484  _tmp = string_view{&*(_pos - vl), vl};
485  }
486  }
487 };
