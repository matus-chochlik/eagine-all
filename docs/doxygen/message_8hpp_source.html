200  /// @brief Indicates that the message made too many hops.
201  /// @see hop_count
202  /// @see add_hop
203  auto too_many_hops() const noexcept -> bool {
204  return hop_count >= hop_count_t(64);
205  }
206 
207  /// @brief Increments the hop counter.
208  /// @see hop_count
209  /// @see too_many_hops
210  auto add_hop() noexcept -> auto& {
211  EAGINE_ASSERT(hop_count < std::numeric_limits<hop_count_t>::max());
212  ++hop_count;
213  return *this;
214  }
215 
216  /// @brief Indicates that the message is too old.
217  /// @see age
218  /// @see add_age
219  auto too_old() const noexcept -> bool {
220  switch(priority) {
221  case message_priority::idle:
222  return age_quarter_seconds > 10 * 4;
223  case message_priority::low:
224  return age_quarter_seconds > 20 * 4;
225  case message_priority::normal:
226  return age_quarter_seconds > 30 * 4;
227  case message_priority::high:
228  return age_quarter_seconds == std::numeric_limits<age_t>::max();
229  case message_priority::critical:
230  break;
231  }
232  return false;
233  }
234 
235  /// @brief Adds to the age seconds counter.
236  /// @see age
237  /// @see too_old
238  auto add_age(const message_age age) noexcept -> auto& {
239  const float added_quarter_seconds = (age.count() + 0.20F) * 4.F;
240  if(const auto new_age{convert_if_fits<age_t>(
241  int(age_quarter_seconds) + int(added_quarter_seconds))}) {
242  age_quarter_seconds = extract(new_age);
243  } else {
244  age_quarter_seconds = std::numeric_limits<age_t>::max();
245  }
246  return *this;
247  }
248 
249  /// @brief Returns the message age
250  /// @see too_old
251  /// @see add_age
252  auto age() const noexcept -> message_age {
253  return message_age{float(age_quarter_seconds) * 0.25F};
254  }
255 
256  /// @brief Sets the priority of this message.

259  return *this;
260  }
261

266  }
267

272  }

526 /// @brief Class storing message bus messages.
527 /// @ingroup msgbus
528 /// @see serialized_message_storage
529 class message_storage {
530 public:
531  /// @brief Default constructor.
532  message_storage() {
533  _messages.reserve(64);
534  }
535 
536  /// @brief Indicates if the storage is empty.
537  auto empty() const noexcept -> bool {
538  return _messages.empty();
539  }
540 
541  /// @brief Returns the coung of messages in the storage.
542  auto count() const noexcept -> span_size_t {
543  return span_size(_messages.size());
544  }
545 
546  /// @brief Pushes a message into this storage.
547  void push(const message_id msg_id, const message_view& message) noexcept {
548  _messages.emplace_back(
549  msg_id,
550  stored_message{message, _buffers.get(message.data().size())},
551  _clock_t::now());
552  }
553 
554  /// @brief Pushes a new message and lets a function to fill it.
555  ///
556  /// The function's Boolean return value indicates if the message should be kept.
557  template <typename Function>
558  auto push_if(Function function, const span_size_t req_size = 0) noexcept
559  -> bool {
560  _messages.emplace_back(
561  message_id{},
562  stored_message{{}, _buffers.get(req_size)},
563  _clock_t::now());
564  auto& [msg_id, message, insert_time] = _messages.back();
565  (void)(insert_time);
566  bool rollback = false;
567  try {
568  if(!function(msg_id, insert_time, message)) {
569  rollback = true;
570  }
571  } catch(...) {
572  rollback = true;
573  }
574  if(rollback) {
575  _buffers.eat(message.release_buffer());
576  _messages.pop_back();
577  return false;
578  }
579  return true;
580  }
581 
582  /// @brief Alias for the message fetch handler.
583  /// @see fetch_all
584  ///
585  /// The return value indicates if the message is considered handled
586  /// and should be removed.
587  using fetch_handler = callable_ref<
588  bool(const message_id, const message_age, const message_view&) noexcept>;
589 
590  /// @brief Fetches all currently stored messages and calls handler on them.
591  auto fetch_all(const fetch_handler handler) noexcept -> bool;
592 
593  /// @brief Alias for message cleanup callable predicate.
594  /// @see cleanup
595  ///
596  /// The return value indicates if a message should be removed.
597  using cleanup_predicate = callable_ref<bool(const message_age) noexcept>;
598 
599  /// @brief Removes messages based on the result of the specified predicate.
600  void cleanup(const cleanup_predicate predicate) noexcept;
601 
602  void log_stats(main_ctx_object&);
603 
604 private:
605  using _clock_t = std::chrono::steady_clock;
606  memory::buffer_pool _buffers{};
607  std::vector<std::tuple<message_id, stored_message, message_timestamp>>
608  _messages;
609 };

611 class message_pack_info {
612 public:
613  using bit_set = std::uint64_t;
614 
615  message_pack_info(const span_size_t total_size) noexcept
616  : _total_size{limit_cast<std::uint16_t>(total_size)} {}
617 
618  operator bool() const noexcept {
619  return !is_empty();
620  }
621 
622  auto is_empty() const noexcept -> bool {
623  return _packed_bits == 0U;
624  }
625 
626  auto bits() const noexcept -> bit_set {
627  return _packed_bits;
628  }
629 
630  auto count() const noexcept -> span_size_t {
631  span_size_t result = 0;
632  auto bits = _packed_bits;
633  while(bits) {
634  ++result;
635  bits &= (bits - 1U);
636  }
637  return result;
638  }
639 
640  auto used() const noexcept -> span_size_t {
641  return span_size(_packed_size);
642  }
643 
644  auto total() const noexcept -> span_size_t {
645  return span_size(_total_size);
646  }
647 
648  auto usage() const noexcept {
649  return float(used()) / float(total());
650  }
651 
652  void add(const span_size_t msg_size, const bit_set current_bit) noexcept {
653  _packed_size += limit_cast<std::uint16_t>(msg_size);
654  _packed_bits |= current_bit;
655  }
656 
657 private:
658  bit_set _packed_bits{0U};
659  std::uint16_t _packed_size{0};
660  const std::uint16_t _total_size{0};
661 };

744 //------------------------------------------------------------------------------
745 class message_priority_queue {
746 public:
747  using handler_type =
748  callable_ref<bool(const message_context&, const stored_message&) noexcept>;
749 
750  message_priority_queue() noexcept {
751  _messages.reserve(128);
752  }
753 
754  auto size() const noexcept {
755  return _messages.size();
756  }
757 
758  auto push(const message_view& message) noexcept -> stored_message& {
759  const auto pos = std::lower_bound(
760  _messages.begin(),
761  _messages.end(),
762  message.priority,
763  [](auto& msg, auto pri) { return msg.priority < pri; });
764 
765  return *_messages.emplace(
766  pos, message, _buffers.get(message.data().size()));
767  }
768 
769  auto process_one(
770  const message_context& msg_ctx,
771  const handler_type handler) noexcept -> bool {
772  if(!_messages.empty()) {
773  if(handler(msg_ctx, _messages.back())) {
774  _buffers.eat(_messages.back().release_buffer());
775  _messages.pop_back();
776  return true;
777  }
778  }
779  return false;
780  }
781 
782  auto process_all(
783  const message_context& msg_ctx,
784  const handler_type handler) noexcept -> span_size_t {
785  span_size_t count{0};
786  std::size_t pos = 0;
787  while(pos < _messages.size()) {
788  if(handler(msg_ctx, _messages[pos])) {
789  ++count;
790  _buffers.eat(_messages[pos].release_buffer());
791  _messages.erase(_messages.begin() + signedness_cast(pos));
792  } else {
793  ++pos;
794  }
795  }
796  return count;
797  }
798 
799 private:
800  memory::buffer_pool _buffers{};
801  std::vector<stored_message> _messages;
802 };
