19 /// @brief Class referencing a set of GLSL source code strings.
20 /// @ingroup glsl_utils
21 class glsl_source_ref {
22 public:
23  /// @brief Alias for character string type.
24  using char_type = gl_types::char_type;
25 
26  /// @brief Alias for integer type.
27  using int_type = gl_types::int_type;
28 
29  /// @brief Alias for integer type storing size values.
30  using sizei_type = gl_types::sizei_type;
31 
32  /// @brief Construction from a single C-string and length value.
33  constexpr glsl_source_ref(
34  const char_type* part,
35  const int_type length) noexcept
36  : _part{part}
37  , _length{length}
38  , _count{1}
39  , _parts{&_part}
40  , _lengths{&_length} {}
41 
42  /// @brief Construction from a string_view.
43  constexpr glsl_source_ref(string_view source_str) noexcept
44  : glsl_source_ref{
45  memory::accommodate<const char_type>(source_str).data(),
46  eagine::limit_cast<int_type>(source_str.size())} {}
47 
48  /// @brief Construction from a string literal.
49  template <std::size_t L>
50  constexpr glsl_source_ref(const char (&source_str)[L]) noexcept
51  : glsl_source_ref{
52  reinterpret_cast<const char_type*>(source_str),
53  eagine::limit_cast<int_type>(L - 1)} {}
54 
55  /// @brief Construction from C-string array and lengths array.
56  glsl_source_ref(
57  const span_size_t n,
58  const char_type** ps,
59  const int_type* ls) noexcept
60  : _count{eagine::limit_cast<sizei_type>(n)}
61  , _parts{ps}
62  , _lengths{ls} {
63  EAGINE_ASSERT(_count >= 0);
64  EAGINE_ASSERT(_parts != nullptr);
65  }
66 
67  /// @brief Returns the number of source string fragments.
68  /// @see parts
69  /// @see lengths
70  constexpr auto count() const noexcept -> sizei_type {
71  return _count;
72  }
73 
74  /// @brief Returns pointers to the source string fragments.
75  /// @see count
76  /// @see lengths
77  /// The Length of the returned array is at least the value returned by count().
78  constexpr auto parts() const noexcept -> const char_type** {
79  return _parts;
80  }
81 
82  /// @brief Returns the lengths of the source string fragments.
83  /// @see count
84  /// @see parts
85  /// The Length of the returned array is at least the value returned by count().
86  auto lengths() const noexcept -> const int_type* {
87  return _lengths;
88  }
89 
90 private:
91  const char_type* _part{nullptr};
92  const int_type _length{0};
93  sizei_type _count{0};
94  const char_type** _parts{nullptr};
95  const int_type* _lengths{nullptr};
96 };
