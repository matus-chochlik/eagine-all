18 /// @brief Reallocatable owning byte buffer.
19 /// @ingroup memory
20 /// @see block
21 class buffer {
22 public:
23  /// @brief Buffer size type.
24  using size_type = typename block::size_type;
25 
26  /// @brief Pointer type.
27  using pointer = typename block::pointer;
28 
29  /// @brief Default constructor.
30  constexpr buffer() noexcept = default;
31 
32  /// @brief Constructor with explicit alignment specification.
33  explicit buffer(const span_size_t align) noexcept
34  : _align{align} {}
35 
36  /// @brief Constructor with injected allocator.
37  explicit buffer(shared_byte_allocator alloc) noexcept
38  : _alloc{std::move(alloc)} {}
39 
40  /// @brief Constructor with explicit alignment specification and allocator.
41  buffer(const span_size_t align, shared_byte_allocator alloc) noexcept
42  : _align{align}
43  , _alloc{std::move(alloc)} {}
44 
45  /// @brief Move constructor.
46  buffer(buffer&& temp) noexcept
47  : _align{temp._align}
48  , _size{std::exchange(temp._size, 0)}
49  , _storage{std::move(temp._storage)}
50  , _alloc{std::move(temp._alloc)} {}
51 
52  /// @brief Move assignment operator.
53  auto operator=(buffer&& temp) noexcept -> buffer& {
54  _align = temp._align;
55  _size = std::exchange(temp._size, 0);
56  _storage = std::move(temp._storage);
57  _alloc = std::move(temp._alloc);
58  return *this;
59  }
60 
61  /// @brief Not copyable.
62  buffer(const buffer&) = delete;
63 
64  /// @brief Not copy assignable.
65  auto operator=(const buffer&) = delete;
66 
67  ~buffer() noexcept {
68  free();
69  }
70 
71  /// @brief Returns the memory address of the start of the allocated space.
72  auto addr() const noexcept {
73  return _storage.addr();
74  }
75 
76  /// @brief Returns the pointer to the start of the allocated space.
77  auto data() const noexcept -> pointer {
78  return _storage.data();
79  }
80 
81  /// @brief Returns the size of the buffer in bytes.
82  /// @see empty
83  /// @see capacity
84  constexpr auto size() const noexcept -> span_size_t {
85  return _size;
86  }
87 
88  /// @brief Indicates that the buffer is empty.
89  /// @see size
90  /// @see capacity
91  constexpr auto empty() const noexcept {
92  return size() == 0;
93  }
94 
95  /// @brief Returns the capacity of this buffer.
96  /// @see reserve
97  /// @see resize
98  constexpr auto capacity() const noexcept -> span_size_t {
99  return _storage.size();
100  }
101 
102  /// @brief Pre-allocate the specified number of bytes.
103  /// @see capacity
104  /// @see resize
105  auto reserve(const span_size_t new_size) noexcept -> auto& {
106  if(capacity() < new_size) {
107  _reallocate(new_size);
108  }
109  EAGINE_ASSERT(_is_ok());
110  return *this;
111  }
112 
113  /// @brief Resizes the buffer to the specified number of bytes.
114  /// @see size
115  /// @see ensure
116  /// @see reserve
117  auto resize(const span_size_t new_size) noexcept -> auto& {
118  reserve(new_size);
119  _size = new_size;
120  EAGINE_ASSERT(_is_ok());
121  return *this;
122  }
123 
124  /// @brief Ensure that the buuffer has at least the specified size in bytes.
125  /// @see size
126  /// @see resize
127  /// @see reserve
128  /// @see enlarge_by
129  auto ensure(const span_size_t new_size) noexcept -> auto& {
130  if(size() < new_size) {
131  return resize(new_size);
132  }
133  EAGINE_ASSERT(_is_ok());
134  return *this;
135  }
136 
137  /// @brief Enlarges the buffer by the specified number of bytes.
138  /// @see size
139  /// @see resize
140  /// @see reserve
141  /// @see ensure
142  auto enlarge_by(const span_size_t inc_size) noexcept -> auto& {
143  return resize(size() + inc_size);
144  }
145 
146  /// @brief Clears the buffer.
147  /// @see size
148  /// @see resize
149  /// @see reserve
150  /// @see free
151  /// @post empty()
152  auto clear() noexcept -> auto& {
153  return resize(0);
154  }
155 
156  /// @brief Deallocates the buffer.
157  /// @see size
158  /// @see resize
159  /// @see reserve
160  /// @see clear
161  /// @post empty() && capacity() == 0
162  void free() noexcept {
163  _alloc.deallocate(std::move(_storage), _align);
164  _size = 0;
165  }
166 
167  /// @brief Implicit conversion to block.
168  operator block() noexcept {
169  EAGINE_ASSERT(_is_ok());
170  return {_storage.begin(), _size};
171  }
172 
173  /// @brief Implicit conversion to const_block.
174  operator const_block() const noexcept {
175  EAGINE_ASSERT(_is_ok());
176  return {_storage.begin(), _size};
177  }
178 
179 private:
180  span_size_t _align{alignof(long double)};
181  span_size_t _size{0};
182  owned_block _storage{};
183  shared_byte_allocator _alloc{default_shared_allocator()};
184 
185  auto _is_ok() const noexcept -> bool {
186  return bool(_alloc) && size() <= capacity();
187  }
188 
189  void _reallocate(const span_size_t new_size) noexcept {
190  _alloc.do_reallocate(_storage, new_size, _align);
191  }
192 };
