24 /// @brief Implements constructor of translation transformation matrix.
25 /// @ingroup math
26 /// @see matrix_translation
27 ///
28 /// @note Do not use directly, use matrix_translation
29 template <typename T, bool RM, bool V>
30 class translation<matrix<T, 4, 4, RM, V>> {
31 public:
32  constexpr translation(const vect::data_t<T, 3, V> v) noexcept
33  : _v{v} {}
34 
35  /// @brief Initializes the matrix constructor.
36  /// @param vx is the x-axis translation distance.
37  /// @param vy is the y-axis translation distance.
38  /// @param vz is the z-axis translation distance.
39  constexpr translation(const T vx, const T vy, const T vz) noexcept
40  : _v{vx, vy, vz} {}
41 
42  /// @brief Returns the constructed matrix.
43  constexpr auto operator()() const noexcept {
44  return _make(bool_constant<RM>());
45  }
46 
47  friend constexpr auto reorder_mat_ctr(
48  const translation<matrix<T, 4, 4, RM, V>>& c) noexcept
49  -> translation<matrix<T, 4, 4, !RM, V>> {
50  return {c._v};
51  }
52 
53 private:
54  constexpr auto _make(const std::true_type) const noexcept {
55  return matrix<T, 4, 4, true, V>{
56  {{T(1), T(0), T(0), _v[0]},
57  {T(0), T(1), T(0), _v[1]},
58  {T(0), T(0), T(1), _v[2]},
59  {T(0), T(0), T(0), T(1)}}};
60  }
61 
62  constexpr auto _make(const std::false_type) const noexcept {
63  return matrix<T, 4, 4, false, V>{
64  {{T(1), T(0), T(0), T(0)},
65  {T(0), T(1), T(0), T(0)},
66  {T(0), T(0), T(1), T(0)},
67  {_v[0], _v[1], _v[2], T(1)}}};
68  }
69 
70  vect::data_t<T, 3, V> _v;
71 };

73 // multiply
74 template <typename T, int N, bool RM1, bool RM2, bool V>

77  const translation<matrix<T, N, N, RM2, V>>& b) noexcept
