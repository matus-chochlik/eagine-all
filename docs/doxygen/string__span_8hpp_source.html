24 /// @brief Specialization of memory::basic_span for character string containers.
25 /// @ingroup string_utils
26 template <typename C, typename P = C*, typename S = span_size_t>
27 class basic_string_span : public memory::basic_span<C, P, S> {
28  using base = memory::basic_span<C, P, S>;
29 
30 public:
31  /// @brief Related standard string type.
32  using string_type = std::basic_string<std::remove_const_t<C>>;
33 
34  /// @brief Related standard string view type.
35  using std_view_type = std::basic_string_view<std::remove_const_t<C>>;
36 
37  /// @brief Default constructor.
38  /// @post empty
39  constexpr basic_string_span() noexcept = default;
40 
41  constexpr basic_string_span(P addr, const S length) noexcept
42  : base{addr, length} {}
43 
44 #if EAGINE_CXX_CONSTEVAL
45  static consteval auto _ce_strlen(const char* str) noexcept {
46  span_size_t len = 0U;
47  while(*str) {
48  ++len;
49  ++str;
50  }
51  return len;
52  }
53 
54  template <typename R>
55  consteval basic_string_span(immediate_function_t, R addr) noexcept requires(
56  !std::is_array_v<R> && std::is_convertible_v<R, P> &&
57  std::is_same_v<std::remove_const_t<std::remove_pointer_t<R>>, char>)
58  : base{addr, -static_cast<S>(_ce_strlen(addr))} {}
59 #endif
60  template <typename R>
61  constexpr explicit basic_string_span(R addr) noexcept requires(
62  !std::is_array_v<R> && std::is_convertible_v<R, P> &&
63  std::is_same_v<std::remove_const_t<std::remove_pointer_t<R>>, char>)
64  : base{addr, -limit_cast<S>(std::strlen(addr))} {}
65 
66  /// @brief Construction from C string literal
67  template <std::size_t N>
68  constexpr basic_string_span(C (&array)[N]) noexcept
69  : base{
70  static_cast<P>(array),
71  array[N - 1] == C(0) ? 1 - limit_cast<S>(N) : limit_cast<S>(N)} {}
72 
73  /// @brief Construction from C-array and explicit length value.
74  template <std::size_t N>
75  constexpr basic_string_span(C (&array)[N], const span_size_t n) noexcept
76  : base{static_cast<P>(array), limit_cast<S>(n)} {}
77 
78  /// @brief Construction from related standard string type.
79  constexpr basic_string_span(const string_type& str) noexcept
80  : base{static_cast<P>(str.c_str()), -limit_cast<S>(str.size())} {}
81 
82  /// @brief Construction from compatible container reference.
83  ///
84  /// The container passed as argument should have @c data and @c size
85  /// member functions with the same semantics as std::string does.
86  template <typename Str>
87  constexpr basic_string_span(Str& str) noexcept
88  requires(memory::has_span_data_member_v<Str, C>&&
89  memory::has_span_size_member_v<Str>)
90  : base{static_cast<P>(str.data()), limit_cast<S>(str.size())} {}
91 
92  /// @brief Construction from compatible container const reference.
93  ///
94  /// The container passed as argument should have @c data and @c size
95  /// member functions with the same semantics as std::string does.
96  template <typename Str>
97  constexpr basic_string_span(const Str& str) noexcept
98  requires(memory::has_span_data_member_v<Str, C>&&
99  memory::has_span_size_member_v<Str>)
100  : base{static_cast<P>(str.data()), limit_cast<S>(str.size())} {}
101 
102  using base::data;
103  using base::empty;
104  using base::size;
105 
106  friend constexpr auto operator==(
107  basic_string_span l,
108  basic_string_span r) noexcept -> bool {
109  return are_equal(l, r);
110  }
111 
112  friend constexpr auto operator!=(
113  basic_string_span l,
114  basic_string_span r) noexcept -> bool {
115  return !are_equal(l, r);
116  }
117 
118  /// @brief Named conversion to the corresponding standard string view.
119  constexpr auto std_view() const noexcept -> std_view_type {
120  return {data(), std_size_t(size())};
121  }
122 
123  /// @brief Implicit conversion to the corresponding standard string view.
124  constexpr operator std_view_type() const noexcept {
125  return std_view();
126  }
127 
128  /// @brief Conversion to the corresponding standard string.
129  constexpr auto to_string() const -> string_type {
130  return {data(), std_size_t(size())};
131  }
132 };

186 /// @brief Comparator template for string span - string span comparisons.
187 /// @ingroup string_utils
188 template <typename Spn>
189 struct basic_view_less {
190  using is_transparent = std::true_type;
191 
192  constexpr auto _helper(
193  const int cmp,
194  const span_size_t lsz,
195  const span_size_t rsz) const noexcept -> bool {
196  return cmp < 0 ? true : cmp > 0 ? false : (lsz < rsz);
197  }
198 
199  constexpr auto operator()(const Spn l, const Spn r) const noexcept -> bool {
200  return _helper(
201  std::strncmp(
202  l.data(), r.data(), std_size(std::min(l.size(), r.size()))),
203  l.size(),
204  r.size());
205  }
206 };
207 //------------------------------------------------------------------------------
208 /// @brief Comparator template for standard string - string span comparisons.
209 /// @ingroup string_utils
210 template <typename Str, typename Spn>
211 struct basic_str_view_less : basic_view_less<Spn> {
212  using base = basic_view_less<Spn>;
213 
214  using base::operator();
215 
216  constexpr auto operator()(const Str& l, const Str& r) const noexcept
217  -> bool {
218  return l < r;
219  }
220 
221  constexpr auto operator()(const Str& l, const Spn r) const noexcept
222  -> bool {
223  return (*this)(Spn(l), r);
224  }
225 
226  constexpr auto operator()(const Spn l, const Str& r) const noexcept
227  -> bool {
228  return (*this)(l, Spn(r));
229  }
230 };

236 template <typename C, typename T, typename A, typename Transform>
237 static inline auto make_span_putter(
238  span_size_t& i,
239  std::basic_string<C, T, A>& str,
240  Transform transform) {
241  return [&i, &str, transform](auto value) mutable -> bool {
242  if(i < span_size_t(str.size())) {
243  if(auto transformed = transform(value)) {
244  str[i++] = std::move(extract(transformed));
245  return true;
246  }
247  }
248  return false;
249  };
