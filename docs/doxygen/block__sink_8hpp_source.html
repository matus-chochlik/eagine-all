19 /// @brief Serialization data sink backed by a pre-allocated memory block.
20 /// @ingroup serialization
21 /// @see block_data_source
22 /// @see packed_block_data_sink
23 class block_data_sink : public serializer_data_sink {
24 public:
25  /// @brief Default constructor.
26  /// @post free().empty()
27  constexpr block_data_sink() noexcept = default;
28 
29  /// @brief Constructor setting the backing block.
30  block_data_sink(const memory::block dst) noexcept
31  : _dst{dst} {}
32 
33  /// @brief Resets the backing block.
34  /// @see replace_with
35  void reset(const memory::block dst) {
36  _dst = dst;
37  _done = 0;
38  _save_points.clear();
39  }
40 
41  /// @brief Returns the part of the backing block already written to.
42  /// @see free
43  auto done() const noexcept -> memory::block {
44  return head(_dst, _done);
45  }
46 
47  /// @brief Returns the free part of the backing block.
48  /// @see done
49  /// @see remaining_size
50  auto free() const noexcept -> memory::block {
51  return skip(_dst, _done);
52  }
53 
54  /// @brief Returns the free part of the backing block and marks it as used.
55  /// @see done
56  /// @see remaining_size
57  /// @see free
58  auto mark_used(const span_size_t size) noexcept -> auto& {
59  EAGINE_ASSERT(size <= remaining_size());
60  _done += size;
61  return *this;
62  }
63 
64  /// @brief Returns the size of the free part of the backing block.
65  /// @see free
66  /// @see done
67  auto remaining_size() noexcept -> span_size_t final {
68  return free().size();
69  }
70 
71  using serializer_data_sink::write;
72 
73  auto write(memory::const_block blk) noexcept -> serialization_errors final {
74  auto dst = free();
75  if(dst.size() < blk.size()) {
76  copy(head(blk, dst.size()), dst);
77  _done += dst.size();
78  return {serialization_error_code::incomplete_write};
79  }
80  copy(blk, dst);
81  _done += blk.size();
82  return {};
83  }
84 
85  /// @brief Replaces the content of the backing block with the content of the argument.
86  /// @see reset
87  auto replace_with(const memory::const_block blk) noexcept
88  -> serialization_errors {
89  if(_dst.size() < blk.size()) {
90  return {serialization_error_code::too_much_data};
91  }
92  copy(blk, _dst);
93  _done = blk.size();
94  return {};
95  }
96 
97  auto begin_work() noexcept -> transaction_handle final {
98  _save_points.push_back(_done);
99  return transaction_handle(_save_points.size());
100  }
101 
102  auto commit([[maybe_unused]] const transaction_handle th) noexcept
103  -> serialization_errors final {
104  EAGINE_ASSERT(th == transaction_handle(_save_points.size()));
105  _save_points.pop_back();
106  return {};
107  }
108 
109  void rollback([[maybe_unused]] const transaction_handle th) noexcept final {
110  EAGINE_ASSERT(th == transaction_handle(_save_points.size()));
111  _done = _save_points.back();
112  _save_points.pop_back();
113  }
114 
115  auto finalize() noexcept -> serialization_errors override {
116  return {};
117  }
118 
119 private:
120  memory::block _dst{};
121  span_size_t _done{0};
122  std::vector<span_size_t> _save_points{};
123 };
