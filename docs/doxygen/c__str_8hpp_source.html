19 /// @brief Helper template for getting zero-terminated strings from string spans.
20 /// @ingroup string_utils
21 /// @see c_str
22 ///
23 /// String spans can in many cases internally know if the element past the end
24 /// is zero or not. There are cases where string spans or generally spans are
25 /// not zero terminated. Because of this whenever a zero-terminated C-string
26 /// is used, for example as an argument to a C-API call, instantiations of this
27 /// template should be used the obtain a valid C-string.
28 template <typename C, typename P = C*, typename S = span_size_t, bool isConst = true>
29 class basic_c_str {
30 public:
31  using span_type = basic_string_span<C, P, S>;
32  using string_type = std::basic_string<std::remove_const_t<C>>;
33  using pointer_type = P;
34 
35  constexpr basic_c_str() noexcept = default;
36 
37  constexpr basic_c_str(const span_type s) noexcept
38  : _span{_get_span(s)}
39  , _str{_get_str(s)} {}
40 
41  template <extractable E>
42  constexpr basic_c_str(construct_from_t, const E& e) noexcept
43  requires(has_value_type_v<E, span_type>)
44  : _span{_xtr_span(e)}
45  , _str{_xtr_str(e)} {}
46 
47  /// @brief Return a zero terminated C-string as pointer_type.
48  /// @see view
49  constexpr auto c_str() const noexcept -> pointer_type {
50  return _span.empty() ? _str.c_str() : _span.data();
51  }
52 
53  /// @brief Implicit conversion to character pointer_type.
54  /// @see c_str
55  constexpr operator pointer_type() const noexcept {
56  return c_str();
57  }
58 
59  /// @brief Returns a const view of the string.
60  /// @see c_str()
61  constexpr auto view() const noexcept -> span_type {
62  return _span.empty() ? span_type{_str} : _span;
63  }
64 
65 private:
66  static constexpr auto _get_span(const span_type s) noexcept -> span_type {
67  return s.is_zero_terminated() ? s : span_type{};
68  }
69 
70  static constexpr auto _get_str(const span_type s) noexcept -> string_type {
71  return s.is_zero_terminated() ? string_type{} : s.to_string();
72  }
73 
74  template <extractable E>
75  static constexpr auto _xtr_span(const E& e) noexcept -> span_type
76  requires(has_value_type_v<E, span_type> && !std::is_same_v<E, span_type>) {
77  return has_value(e) ? _get_span(extract(e)) : span_type{};
78  }
79 
80  template <extractable E>
81  static constexpr auto _xtr_str(const E& e) noexcept -> string_type
82  requires(has_value_type_v<E, span_type>) {
83  return has_value(e) ? _get_str(extract(e)) : string_type{};
84  }
85 
86  std::conditional_t<isConst, const span_type, span_type> _span{};
87  std::conditional_t<isConst, const string_type, string_type> _str{};
88 };

101 /// @brief Functions that construct a basic_c_str from a basic_string_span.
102 /// @ingroup string_utils
103 template <typename C, typename P, typename S>
104 static constexpr auto c_str(const memory::basic_span<C, P, S> s)
105  -> basic_c_str<C, P, S> requires(std::is_convertible_v<
106  memory::basic_span<C, P, S>,
107  basic_string_span<C, P, S>>) {
108  return {s};
109 }
