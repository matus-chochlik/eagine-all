19 /// @brief Basic scalar implementation template.
20 /// @ingroup math
21 /// @see tvec
22 /// @note This class is used in vectorizer vector-scalar operations.
23 template <typename T, int N, bool V>
24 struct scalar {
25  using type = scalar;
26 
27  /// @brief The scalar element type.
28  using value_type = T;
29 
30  /// @brief Indicates if the implementation uses SIMD extensions.
31  using is_vectorized = vect::has_vect_data_t<T, N, V>;
32 
33  using data_type =
34  std::conditional_t<is_vectorized::value, vect::data_t<T, N, V>, T>;
35 
36  data_type _v;
37 
38  using scalar_param = const scalar&;
39 
40  static constexpr auto _from(const data_type v) noexcept {
41  return scalar{v};
42  }
43 
44  static constexpr auto _make(const T v, const std::true_type) noexcept {
45  return scalar{vect::fill<T, N, V>::apply(v)};
46  }
47 
48  static constexpr auto _make(const T v, const std::false_type) noexcept {
49  return scalar{v};
50  }
51 
52  /// @brief Creates a scalar with the specified value.
53  static constexpr auto make(const T v) noexcept {
54  return _make(v, is_vectorized());
55  }
56 
57  constexpr auto _get(const std::true_type) const noexcept -> T {
58  return _v[0];
59  }
60 
61  constexpr auto _get(const std::false_type) const noexcept -> T {
62  return _v;
63  }
64 
65  /// @brief Implicit cast to the value type.
66  constexpr operator T() const noexcept {
67  return _get(is_vectorized());
68  }
69 
70  /// @brief Assignment from the value type.
71  auto operator=(const T v) noexcept -> scalar& {
72  return *this = make(v);
73  }
74 };
