22 /// @brief Helper mixin class for message bus services composed of several parts.
23 /// @ingroup msgbus
24 template <typename Base = subscriber>
25 class service_composition
26  : public Base
27  , public connection_user
28  , public service_interface {
29 
30  using This = service_composition;
31 
32 public:
33  /// @brief Construction from a reference to an endpoint.
34  service_composition(endpoint& bus)
35  : Base{bus} {
36  _init();
37  }
38 
39  /// @brief Move constructible.
40  // NOLINTNEXTLINE(hicpp-noexcept-move,performance-noexcept-move-constructor)
41  service_composition(service_composition&& that)
42  : Base{static_cast<Base&&>(that)} {
43  _init();
44  }
45 
46  /// @brief Not copy constructible.
47  service_composition(const service_composition&) = delete;
48 
49  /// @brief Not move assignable.
50  auto operator=(service_composition&&) = delete;
51 
52  /// @brief Not copy assignable.
53  auto operator=(const service_composition&) = delete;
54 
55  ~service_composition() noexcept override {
56  this->retract_subscriptions();
57  this->finish();
58  }
59 
60  /// @brief Adds a connection to the associated endpoint.
61  auto add_connection(std::unique_ptr<connection> conn) noexcept
62  -> bool final {
63  return this->bus_node().add_connection(std::move(conn));
64  }
65 
66  /// @brief Updates the associated endpoint and processes all incoming messages.
67  auto update_and_process_all() noexcept -> work_done final {
68  some_true something_done{};
69  something_done(this->update());
70  something_done(this->process_all());
71  return something_done;
72  }
73 
74 protected:
75  void add_methods() noexcept {
76  Base::add_methods();
77  Base::add_method(
78  this,
79  EAGINE_MSG_MAP(eagiMsgBus, qrySubscrp, This, _handle_sup_query));
80  Base::add_method(
81  this,
82  EAGINE_MSG_MAP(eagiMsgBus, qrySubscrb, This, _handle_sub_query));
83  }
84 
85 private:
86  auto _handle_sup_query(
87  const message_context&,
88  const stored_message& message) noexcept -> bool {
89  this->respond_to_subscription_query(message.source_id);
90  return true;
91  }
92 
93  auto _handle_sub_query(
94  const message_context&,
95  const stored_message& message) noexcept -> bool {
96  message_id sub_msg_id{};
97  if(default_deserialize_message_type(sub_msg_id, message.content())) {
98  this->respond_to_subscription_query(message.source_id, sub_msg_id);
99  }
100  return true;
101  }
102 
103  void _init() noexcept {
104  this->add_methods();
105  this->init();
106  this->announce_subscriptions();
107  }
108 };
