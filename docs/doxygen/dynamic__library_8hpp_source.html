22 /// @brief Wrapper for a handle to dynamically linkable executable module.
23 /// @see shared_executable_module
24 /// @ingroup dyn_lib
25 /// @note Do not use directly, consider using shared executable module.
26 class executable_module {
27 public:
28  /// @brief Default constructor.
29  executable_module() noexcept = default;
30 
31  /// @brief Not move constructible.
32  executable_module(const executable_module&) = delete;
33 
34  /// @brief Not copy constructible.
35  executable_module(executable_module&&) = delete;
36 
37  /// @brief Not move assignable.
38  auto operator=(const executable_module&) = delete;
39 
40  /// @brief Not copy assignable.
41  auto operator=(executable_module&&) = delete;
42 
43  /// @brief Alias for the handle type.
44  using handle_type = void*;
45 
46  executable_module(const string_view filename, const int flags)
47  : _handle{::dlopen(c_str(filename), flags)} {
48  if(!_handle) {
49  _message.assign(::dlerror());
50  }
51  }
52 
53  executable_module(const nothing_t, const int flags)
54  : _handle{::dlopen(nullptr, flags)} {
55  if(!_handle) {
56  _message.assign(::dlerror());
57  }
58  }
59 
60  /// @brief Closes the handle.
61  ~executable_module() noexcept {
62  if(_handle) {
63  ::dlclose(_handle);
64  }
65  }
66 
67  /// @brief Checks if the module is open.
68  /// @see find_symbol
69  auto is_open() const noexcept -> bool {
70  return bool(_handle);
71  }
72 
73  /// @brief Returns a pointer to the exported symbol with the specifed name.
74  /// @see is_open
75  /// @see error_message
76  auto find_symbol(const string_view name) -> optionally_valid<void*> {
77  ::dlerror();
78  void* result = ::dlsym(_handle, c_str(name));
79  if(const auto error{::dlerror()}) {
80  _message.assign(error);
81  return {};
82  }
83  return {result, true};
84  }
85 
86  /// @brief Returns the user-readable error message from the last failed operation.
87  auto error_message() const noexcept -> string_view {
88  return {_message};
89  }
90 
91 private:
92  handle_type _handle{nullptr};
93  std::string _message{};
94 };

96 /// @brief Reference counting handle to dynamically linkable executable module.
97 /// @ingroup dyn_lib
98 class shared_executable_module {
99 public:
100  /// @brief Default constructor.
101  /// @post !is_open()
102  shared_executable_module() noexcept = default;
103 
104  /// @brief Constructor which opens the current executable as a module.
105  /// @see open_self
106  shared_executable_module(const nothing_t)
107  : _module{_do_open(nothing)} {}
108 
109  /// @brief Construction which tries to open executable module with specified filename.
110  /// @see open
111  shared_executable_module(const string_view filename)
112  : _module{_do_open(filename)} {}
113 
114  /// @brief Opens the current executable module. Closes the previous one, if any.
115  /// @see open
116  /// @see is_open
117  auto open_self() -> auto& {
118  _module = _do_open(nothing);
119  return *this;
120  }
121 
122  /// @brief Opens the specified executable module. Closes the previous one, if any.
123  /// @see open_self
124  /// @see is_open
125  auto open(const string_view filename) -> auto& {
126  _module = _do_open(filename);
127  return *this;
128  }
129 
130  /// @brief Indicates if this executable module is open.
131  /// @see open
132  /// @see open_self
133  auto is_open() const noexcept -> bool {
134  return bool(_module) && _module->is_open();
135  }
136 
137  /// @brief Indicates if this executable module is open.
138  /// @see is_open
139  explicit operator bool() const noexcept {
140  return is_open();
141  }
142 
143  /// @brief Returns a human-readable error message from the last failed operation.
144  auto error_message() const noexcept -> string_view {
145  return bool(_module) ? _module->error_message() : string_view();
146  }
147 
148  /// @brief Tests if this executable module exports the specified symbol.
149  /// @see is_open
150  /// @see find
151  auto exports(const string_view name) const noexcept -> bool {
152  if(is_open()) {
153  if(_module->find_symbol(name)) {
154  return true;
155  }
156  }
157  return false;
158  }
159 
160  /// @brief Finds and returns pointer to exported function with the specified name.
161  /// @see is_open
162  /// @see exports
163  template <typename Signature>
164  auto find(const string_view name) const noexcept
165  -> callable_ref<std::remove_pointer_t<Signature>> requires(
166  std::is_function_v<std::remove_pointer_t<Signature>>) {
167  if(is_open()) {
168  if(auto found{_module->find_symbol(name)}) {
169  return {
170  reinterpret_cast<std::remove_pointer_t<Signature>*>(
171  extract(found)),
172  true};
173  }
174  }
175  return {};
176  }
177 
178 private:
179  std::shared_ptr<executable_module> _module{};
180 
181  static auto _do_open(const string_view filename)
182  -> std::shared_ptr<executable_module> {
183  return std::make_shared<executable_module>(filename, RTLD_LAZY);
184  }
185 
186  static auto _do_open(const nothing_t)
187  -> std::shared_ptr<executable_module> {
188  return std::make_shared<executable_module>(nothing, RTLD_LAZY);
189  }
190 };
