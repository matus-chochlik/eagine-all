31 /// @brief Implements constructor of rotation transformation matrix.
32 /// @ingroup math
33 /// @tparam I index of the rotation axis (0 = x, 1 = y, 2 = z).
34 /// @see matrix_rotation_x
35 /// @see matrix_rotation_y
36 /// @see matrix_rotation_z
37 ///
38 /// @note Do not use directly, use matrix_rotation_{x,y,z}
39 template <typename T, bool RM, bool V, int I>
40 class rotation_I<matrix<T, 4, 4, RM, V>, I> {
41 public:
42  /// @brief Initializes the matrix constructor.
43  /// @param v is the angle of rotation.
44  constexpr rotation_I(const radians_t<T> v) noexcept
45  : _v{v} {}
46 
47  /// @brief Returns the constructed matrix.
48  constexpr auto operator()() const {
49  using _axis = int_constant<I>;
50  return _make(cos(_v), sin(_v) * (RM ? 1 : -1), _axis());
51  }
52 
53  friend constexpr auto reorder_mat_ctr(
54  const rotation_I<matrix<T, 4, 4, RM, V>, I>& c) noexcept
55  -> rotation_I<matrix<T, 4, 4, !RM, V>, I> {
56  return {c._v};
57  }
58 
59 private:
60  using _x = int_constant<0>;
61  using _y = int_constant<1>;
62  using _z = int_constant<2>;
63 
64  static constexpr auto _make(const T cx, const T sx, _x) noexcept {
65  return matrix<T, 4, 4, RM, V>{
66  {{T(1), T(0), T(0), T(0)},
67  {T(0), cx, -sx, T(0)},
68  {T(0), sx, cx, T(0)},
69  {T(0), T(0), T(0), T(1)}}};
70  }
71 
72  static constexpr auto _make(const T cx, const T sx, _y) noexcept {
73  return matrix<T, 4, 4, RM, V>{
74  {{cx, T(0), sx, T(0)},
75  {T(0), T(1), T(0), T(0)},
76  {-sx, T(0), cx, T(0)},
77  {T(0), T(0), T(0), T(1)}}};
78  }
79 
80  static constexpr auto _make(const T cx, const T sx, _z) noexcept {
81  return matrix<T, 4, 4, RM, V>{
82  {{cx, -sx, T(0), T(0)},
83  {sx, cx, T(0), T(0)},
84  {T(0), T(0), T(1), T(0)},
85  {T(0), T(0), T(0), T(1)}}};
86  }
87 
88  radians_t<T> _v;
89 };

91 // multiply
92 template <typename T, int N, bool RM1, bool RM2, bool V, int I>
93 static constexpr auto multiply(
94  const rotation_I<matrix<T, N, N, RM1, V>, I>& a,
95  const rotation_I<matrix<T, N, N, RM2, V>, I>& b) noexcept
96  -> rotation_I<matrix<T, N, N, RM1, V>, I> {
97  return {radians_t<T>{T(a._v) + T(b._v)}};
98 }
