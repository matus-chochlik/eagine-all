16 namespace eagine {

18 /// @brief Class representing a value animated between two points, with easing.
19 template <typename T, typename F, typename S = float>
20 class animated_value {
21 public:
22  /// @brief Default constructor.
23  animated_value() = default;
24 
25  /// @brief Initializing to the specified value.
26  animated_value(const T& initial)
27  : _curr{initial}
28  , _next{initial} {}
29 
30  /// @brief Sets the next point to animate into, duration and sigmoid slope.
31  auto set(const T value, const valid_if_positive<F> duration, const S slope)
32  -> auto& {
33  _curr = _next;
34  _next = value;
35  _duration = extract(duration);
36  _phase = F(0);
37  _slope = slope;
38  return *this;
39  }
40 
41  /// @brief Sets the next point to animate into and the duration of the transition.
42  auto set(const T value, const valid_if_positive<F> duration) -> auto& {
43  return set(value, duration, 2);
44  }
45 
46  /// @brief Gets the current value between the current and the next point.
47  auto get() const -> T {
48  return math::blend(
49  _curr,
50  _next,
51  math::sigmoid01(math::clamp(_phase / _duration, 0, 1), _slope));
52  }
53 
54  /// @brief Updates the transition factor by the specified amount.
55  auto update(const F deltaT) -> auto& {
56  _phase += deltaT;
57  return *this;
58  }
59 
60  /// @brief Indicates if the current transition factor is higher than current duration.
61  auto is_done() const noexcept -> bool {
62  return _phase >= _duration;
63  }
64 
65 private:
66  T _curr{};
67  T _next{};
68  F _duration{1};
69  F _phase{_duration};
70  S _slope{2};
71 };
