21 struct element_name : ecs::component<EAGINE_ID_V(Name)> {
22  std::string latin;
23  std::string english;
24 };
25 //------------------------------------------------------------------------------
26 namespace ecs {
27 template <bool Const>
28 struct get_manipulator<element_name, Const> {
29  struct type : basic_manipulator<element_name, Const> {
30  using basic_manipulator<element_name, Const>::basic_manipulator;
31 
32  auto set_names(std::string lat, std::string eng) -> auto& {
33  this->write().latin = std::move(lat);
34  this->write().english = std::move(eng);
35  return *this;
36  }
37 
38  auto get_latin_name() const -> const std::string& {
39  return this->read().latin;
40  }
41 
42  auto get_english_name() const -> const std::string& {
43  const auto& name = this->read().english;
44  if(name.empty()) {
45  return this->read().latin;
46  }
47  return name;
48  }
49 
50  auto has_english_name() const -> bool {
51  return this->is_valid() ? !this->read().english.empty() : false;
52  }
53  };
54 };
55 } // namespace ecs

189 //------------------------------------------------------------------------------
190 class decay_modes : public ecs::component<EAGINE_ID_V(DecayModes)> {
191 public:
192  auto add(string_view symbol) -> decay* {
193  int nv = 0;
194  if(const auto nid{known_decay_modes::get_id(symbol)}) {
195  for(const auto& [id, v, info] : _modes) {
196  if(nid == id) {
197  nv = nv + 1;
198  }
199  }
200  _modes.push_back({nid, nv, {}});
201  return &std::get<2>(_modes.back());
202  }
203  return nullptr;
204  }
205 
206  template <typename Function>
207  void for_each(const Function& func) {
208  for(auto& [id, v, info] : _modes) {
209  (void)(v);
210  if(auto mode_info{known_decay_modes::get_info(id)}) {
211  func(*mode_info, info);
212  }
213  }
214  }
215 
216  template <typename Function>
217  void for_each(const Function& func) const {
218  for(const auto& [id, v, info] : _modes) {
219  (void)(v);
220  if(auto mode_info{known_decay_modes::get_info(id)}) {
221  func(*mode_info, info);
222  }
223  }
224  }
225 
226 private:
227  std::vector<std::tuple<identifier_t, int, decay>> _modes;
228 };
229 //------------------------------------------------------------------------------
230 namespace ecs {
