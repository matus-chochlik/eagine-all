15 /// @brief Class representing a positive or negative numeric sign.
16 /// @ingroup math
17 ///
18 /// Instances of this class can be multiplied to values of type @p T.
19 template <typename T>
20 class sign {
21 public:
22  /// @brief Alias for sign class.
23  using type = sign;
24 
25  /// @brief Alias for the value type @p T.
26  using value_type = T;
27 
28  /// @brief Default constructor (constructs a positive sign).
29  constexpr sign() noexcept = default;
30 
31  /// @brief Construction with explicit specitication of the sign.
32  constexpr explicit sign(const bool pos) noexcept
33  : _positive{pos} {}
34 
35  /// @brief Construction taking the sign from the specified numeric @p value.
36  template <typename X>
37  constexpr explicit sign(const X& value) noexcept
38  requires(!std::is_same_v<X, sign<T>>)
39  : _positive{value >= X(0)} {}
40 
41  /// @brief Assignment taking the sign from the specified numeric @p value.
42  template <typename X>
43  auto operator=(const X& value) noexcept
44  -> auto& requires(!std::is_same_v<X, sign<T>>) {
45  _positive = (value >= X(0));
46  return *this;
47  }
48 
49  /// @brief Constructs a positive sign (plus).
50  /// @see minus
51  static constexpr auto plus() noexcept -> sign {
52  return {true};
53  }
54 
55  /// @brief Constructs a negative sign (minus).
56  /// @see plus
57  static constexpr auto minus() noexcept -> sign {
58  return {false};
59  }
60 
61  /// @brief Returns either 1 if this sign is positive or -1 if negative.
62  operator T() const noexcept {
63  return _positive ? T{1} : T{-1};
64  }
65 
66  /// @brief Flip this sign from positive to negative or vice-versa.
67  /// @see flipped
68  auto flip() noexcept -> auto& {
69  _positive = !_positive;
70  return *this;
71  }
72 
73  /// @brief Returns a new sign opposite to this sign.
74  /// @see flip
75  auto flipped() const noexcept -> sign {
76  return {!_positive};
77  }
78 
79  /// @brief Returns a new sign opposite to this sign.
80  /// @see flipped
81  /// @see flip
82  auto operator-() const noexcept -> sign {
83  return flipped();
84  }
85 
86  /// @brief Returns a new sign opposite to this sign.
87  /// @see flipped
88  /// @see flip
89  auto operator!() const noexcept -> sign {
90  return flipped();
91  }
92 
93 private:
94  bool _positive{true};
95 };
