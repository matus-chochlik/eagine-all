27 /// @brief Class providing access to the contents of a file.
28 /// @see structured_file_content
29 class file_contents {
30 public:
31  /// @brief Default constructor.
32  file_contents() noexcept = default;
33 
34  /// @brief Constructor that opens and loads contents of file at the given path.
35  file_contents(const string_view path);
36 
37  /// @brief Checks if the contents were loaded.
38  /// @see block
39  auto is_loaded() const noexcept -> bool {
40  return bool(_pimpl);
41  }
42 
43  /// @brief Checks if the contents were loaded.
44  /// @see is_loaded
45  explicit operator bool() const noexcept {
46  return is_loaded();
47  }
48 
49  /// @brief Returns the block viewing the loaded file contents.
50  /// @see is_loaded
51  auto block() const noexcept -> memory::const_block {
52  if(_pimpl) [[likely]] {
53  return _pimpl->block();
54  } else {
55  return {};
56  }
57  }
58 
59  /// @brief Implicit conversion to the block viewing the loaded file contents.
60  /// @see block
61  /// @see is_loaded
62  operator memory::const_block() const noexcept {
63  return block();
64  }
65 
66 private:
67  std::shared_ptr<file_contents_intf> _pimpl{};
68 };
