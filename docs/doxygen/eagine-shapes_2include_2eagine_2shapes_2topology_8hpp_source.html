26 /// @brief Class storing information about an edge between two mesh faces.
27 /// @ingroup shapes
28 /// @see mesh_triangle
29 /// @see topology
30 class mesh_edge {
31 public:
32  mesh_edge(
33  mesh_triangle& a,
34  const std::uint8_t aeb,
35  const std::uint8_t aee,
36  mesh_triangle& b,
37  const std::uint8_t beb,
38  const std::uint8_t bee) noexcept
39  : _triangles{{&a, &b}}
40  , _edge_indices{{aeb, aee, beb, bee}} {}
41 
42  /// @brief Returns one of the two adjacent triangle faces.
43  /// @pre i >= 0 && i < 2
44  auto triangle(const span_size_t i) const noexcept -> const mesh_triangle& {
45  EAGINE_ASSERT(i >= 0 && i < 2);
46  return *_triangles[integer(i)];
47  }
48 
49  /// @brief Returns a pair of vertex indices (0,1 or 2) defining the i-th edge.
50  /// @pre i >= 0 && i < 2
51  auto edge_vertices(const span_size_t i) const noexcept
52  -> std::tuple<unsigned, unsigned> {
53  EAGINE_ASSERT(i >= 0 && i < 2);
54  return {
55  _edge_indices[integer(2 * i)], _edge_indices[integer(2 * i + 1)]};
56  }
57 
58 private:
59  std::array<mesh_triangle*, 2> _triangles;
60  std::array<std::uint8_t, 4> _edge_indices;
61 };

88  /// @brief Returns the v-th vertex index.
89  /// @pre v >= 0 && v < 3
90  auto vertex_index(const span_size_t v) const noexcept {
91  EAGINE_ASSERT(v >= 0 && v < 3);

93  }
94 
95  /// @brief Returns the triangle adjacent through the v-th edge.
96  /// @pre v >= 0 && v < 3
97  /// @see opposite_vertex
98  auto adjacent_triangle(const span_size_t v) const noexcept
99  -> const mesh_triangle* {
100  EAGINE_ASSERT(v >= 0 && v < 3);
101  return _adjacent[integer(v)];
102  }
103 
104  /// @brief Returns opposite vertex index (0,1 or 2) in the v-th adjacent triangle.
105  /// @pre v >= 0 && v < 3
106  /// @see adjacent_triangle
107  /// @see opposite_index
108  auto opposite_vertex(const span_size_t v) const noexcept -> unsigned {
109  EAGINE_ASSERT(v >= 0 && v < 3);
110  return _opposite[integer(v)];
111  }
112 
113  /// @brief Returns the in-mesh index of the v-th adjacent vertex.
114  /// @pre v >= 0 && v < 3
115  /// @see opposite_vertex
116  /// @see adjacent_triangle
117  auto opposite_index(const span_size_t v) const noexcept -> unsigned {
118  if(auto tri{adjacent_triangle(v)}) {
119  return extract(tri).vertex_index(opposite_vertex(v));
120  }
121  return _indices[(v + 2) % 3];
122  }
