19 class texture_image_block {
20 private:
21  structured_memory_block<const image_data_header> _header;
22 
23 public:
24  texture_image_block(const memory::const_block blk)
25  : _header(blk) {}
26 
27  texture_image_block(const image_data_header* ptr)
28  : _header(as_bytes(view_one(ptr))) {}
29 
30  auto is_valid() const noexcept -> bool {
31  return _header->magic.is_valid();
32  }
33 
34  auto dimensions() const noexcept -> image_dimensions {
35  return {
36  _header->width, _header->height, _header->depth, _header->channels};
37  }
38 
39  auto format() const noexcept -> image_pixel_format {
40  return {
41  pixel_format(_header->format),
42  pixel_internal_format(_header->internal_format)};
43  }
44 
45  auto format(const pixel_internal_format ifmt) const noexcept
46  -> image_pixel_format {
47  return {pixel_format(_header->format), ifmt};
48  }
49 
50  auto pixel_element_bytes() const noexcept -> span_size_t {
51  return span_size(sizeof(GLubyte));
52  }
53 
54  auto pixel_data() const noexcept -> image_pixel_data {
55  return {
56  pixel_data_type(_header->data_type),
57  as_bytes(_header->pixels),
58  pixel_element_bytes()};
59  }
60 
61  auto spec() const noexcept -> image_spec {
62  return {dimensions(), format(), pixel_data()};
63  }
64 
65  auto spec(const pixel_internal_format ifmt) const noexcept -> image_spec {
66  return {dimensions(), format(ifmt), pixel_data()};
67  }
68 
69  inline operator image_spec() const noexcept {
70  return spec();
71  }
72 };
