18 template <typename F>
19 class memoized;
20 
21 template <typename R, typename... P>
22 class memoized<R(P...)> {
23 public:
24  template <typename Func>
25  memoized(Func&& func) noexcept
26  requires(!std::is_same_v<std::decay_t<Func>, memoized>)
27  : _func(std::forward<Func>(func)) {}
28 
29  template <typename F>
30  auto operator()(P... p, const F& f) -> R {
31  T t(p...);
32  auto i = _memo.find(t);
33  if(i == _memo.end()) {
34  i = _memo.insert(E(t, f(p..., *this))).first;
35  }
36  return i->second;
37  }
38 
39  auto operator()(P... p) -> R {
40  return _func(p..., *this);
41  }
42 
43  void reset(P... p) {
44  _memo.erase(T(p...));
45  }
46 
47  void clear() {
48  _memo.clear();
49  }
50 
51 private:
52  using T = std::tuple<P...>;
53  using E = std::pair<T, R>;
54  std::map<T, R> _memo;
55  callable_ref<R(P..., memoized&)> _func;
56 };
