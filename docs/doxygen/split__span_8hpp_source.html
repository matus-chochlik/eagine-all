14 namespace eagine::memory {
15 //------------------------------------------------------------------------------
16 /// @brief Template for classes representing a memory span split in two parts.
17 /// @ingroup memory
18 /// @tparam ValueType the type of elements assumed in the covered memory range.
19 /// @tparam Pointer the pointer type used to point to the start of the span.
20 /// @tparam SizeType the integer type used to count the elements in the span.
21 /// @see basic_split_block
22 template <
23  typename ValueType,
24  typename Pointer = ValueType*,
25  typename SizeType = span_size_t>
26 class basic_split_span {
27 public:
28  /// @brief Default constructor.
29  constexpr basic_split_span() noexcept = default;
30 
31  /// @brief Initializing constructor.
32  constexpr basic_split_span(
33  const basic_span<ValueType, Pointer, SizeType> spn) noexcept
34  : _span{spn} {}
35 
36  /// @brief Returns the position of split as count of bytes since beginning.
37  /// @see head
38  /// @see tail
39  /// @see advance
40  constexpr auto split_position() const noexcept -> span_size_t {
41  return _split;
42  }
43 
44  /// @brief Returns the part of the block before the split (split not included).
45  /// @see tail
46  /// @see split_position
47  constexpr auto head() const noexcept {
48  return head(_span, _split);
49  }
50 
51  /// @brief Returns the part of the block starting at the split until the end.
52  /// @see head
53  /// @see split_position
54  constexpr auto tail() const noexcept {
55  return skip(_span, _split);
56  }
57 
58  /// @brief Moves the split by the specified amout (number of bytes).
59  /// @see split_position
60  /// @see skip_to_end
61  auto advance(const span_size_t amount) noexcept -> auto& {
62  _split += amount;
63  return *this;
64  }
65 
66  /// @brief Moves the split to the end of the block.
67  /// @see advance
68  /// @see split_position
69  auto skip_to_end() noexcept -> auto& {
70  _split = _span.size();
71  return *this;
72  }
73 
74 private:
75  basic_span<ValueType, Pointer, SizeType> _span{};
76  span_size_t _split{0};
77 };

79 } // namespace eagine::memory
