18 /// @brief Basic RxC matrix implementation template.
19 /// @ingroup math
20 /// @see tmat
21 /// @note This is a base class, typically tmat should be used in client code.
22 template <typename T, int C, int R, bool RM, bool V>
23 struct matrix {
24  using type = matrix;
25 
26  /// @brief Element type.
27  using element_type = T;
28 
29  template <int... U>
30  using _iseq = std::integer_sequence<int, U...>;
31 
32  template <int N>
33  using _make_iseq = std::make_integer_sequence<int, N>;
34 
35  vect::data_t<T, RM ? C : R, V> _v[RM ? R : C];
36 
37  template <typename P, int... I>
38  static auto _from_hlp(const P* dt, span_size_t sz, _iseq<I...>) noexcept
39  -> matrix {
40  return matrix{
41  {vect::from_array < T,
42  RM ? C : R,
43  V > ::apply(dt + I * (RM ? C : R), sz - I * (RM ? C : R))...}};
44  }
45 
46  /// @brief Creates a matrix from data pointer and size.
47  template <typename P>
48  static auto from(const P* dt, const span_size_t sz) noexcept -> matrix {
49  return _from_hlp(dt, sz, _make_iseq < RM ? R : C > ());
50  }
51 
52  template <typename P, int M, int N, bool W, int... I>
53  static auto _from_hlp(const matrix<P, M, N, RM, W>& m, _iseq<I...>) noexcept
54  -> matrix {
55  return matrix{
56  {vect::cast<P, (RM ? M : N), W, T, (RM ? C : R), V>::apply(
57  m._v[I], T(0))...}};
58  }
59 
60  /// @brief Creates a matrix from another matrix type.
61  template <typename P, int M, int N, bool W>
62  static auto from(const matrix<P, M, N, RM, W>& m) noexcept -> matrix
63  requires((C <= M) && (R <= N)) {
64  return _from_hlp(m, _make_iseq < RM ? R : C > ());
65  }
66 
67  /// @brief Subscript operator.
68  auto operator[](const int i) const noexcept
69  -> const vector<T, RM ? C : R, V> {
70  return {_v[i]};
71  }
72 };

282 // transpose_tpl 4x4 matrix
283 template <bool DstRM, bool SrcRM, typename T, bool V>
284 static constexpr auto transpose_tpl(const matrix<T, 4, 4, SrcRM, V>& m) noexcept
285  -> matrix<T, 4, 4, DstRM, V> {
286  return transpose_tpl_hlp<DstRM, T, V>(
287  vect::shuffle2<T, 4, V>::template apply<0, 1, 4, 5>(m._v[0], m._v[1]),
288  vect::shuffle2<T, 4, V>::template apply<2, 3, 6, 7>(m._v[0], m._v[1]),
289  vect::shuffle2<T, 4, V>::template apply<0, 1, 4, 5>(m._v[2], m._v[3]),

291 }
292 //------------------------------------------------------------------------------
293 // transpose_tpl
294 template <bool DstRM, bool SrcRM, typename T, int C, int R, bool V>
295 static constexpr auto transpose_tpl(const matrix<T, C, R, SrcRM, V>& m) noexcept
296  -> matrix<T, (DstRM == SrcRM ? R : C), (DstRM == SrcRM ? C : R), DstRM, V> {
297  const bool S = (DstRM != SrcRM);
298  matrix<T, (S ? C : R), (S ? R : C), DstRM, V> r;
299 
300  for(int i = 0; i < R; ++i) {
301  for(int j = 0; j < C; ++j) {
302  set_rm(r, S ? i : j, S ? j : i, get_rm(m, i, j));
303  }
304  }
305 
306  return r;
307 }

557 
558  for(int i = 0; i < M; ++i) {
559  for(int j = 0; j < N; ++j) {
560  T s = T(0);
561 
562  for(int k = 0; k < K; ++k) {
563  s += get_rm(m1, i, k) * get_rm(m2, k, j);
564  }
565 
566  set_rm(m3, i, j, s);
567  }
568  }
569  return m3;
570 }

581 //------------------------------------------------------------------------------
582 /// @brief Matrix-vector multiplication function.
583 /// @ingroup math
584 template <typename T, int C, int R, bool RM, bool V>
585 static constexpr auto multiply(
586  const matrix<T, C, R, RM, V>& m,
587  const vector<T, C, V>& v) noexcept -> vector<T, R, V> {
588  vector<T, R, V> r{};
589 
590  for(int i = 0; i < R; ++i) {
591  T s = T(0);
592  for(int j = 0; j < C; ++j) {
593  s += get_rm(m, i, j) * v._v[j];
594  }
595  r._v[i] = s;
596  }
597  return r;
598 }

601 /// @ingroup math
602 ///
603 /// Doing multiplication on matrix constructors is typically more efficient
604 /// than doing multiplication on the constructed matrices, because it can
605 /// save row/column reordering operations.

608  requires(

610  are_multiplicable<constructed_matrix_t<MC1>, constructed_matrix_t<MC2>>::
611  value) {
612  return multiply(mc1, mc2);
