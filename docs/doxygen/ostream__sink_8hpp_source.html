20 /// @brief Serialization data sink backed by an output stream.
21 /// @ingroup serialization
22 /// @see istream_data_source
23 class ostream_data_sink : public serializer_data_sink {
24 public:
25  /// @brief Constructor setting the target stream.
26  ostream_data_sink(std::ostream& out) noexcept
27  : _out{out} {}
28 
29  auto remaining_size() noexcept -> span_size_t final {
30  return std::numeric_limits<span_size_t>::max();
31  }
32 
33  using serializer_data_sink::write;
34 
35  auto write(memory::const_block blk) noexcept -> serialization_errors final {
36  write_to_stream(current(), blk);
37  if(current().eof()) {
38  return {serialization_error_code::too_much_data};
39  }
40  if(!current().good()) {
41  return {serialization_error_code::data_sink_error};
42  }
43  return {};
44  }
45 
46  auto begin_work() noexcept -> transaction_handle final {
47  _subs.emplace();
48  return transaction_handle(_subs.size());
49  }
50 
51  auto commit([[maybe_unused]] const transaction_handle th) noexcept
52  -> serialization_errors final {
53  EAGINE_ASSERT(th == transaction_handle(_subs.size()));
54  std::stringstream top{std::move(_subs.top())};
55  _subs.pop();
56  current() << top.rdbuf();
57  return {};
58  }
59 
60  void rollback([[maybe_unused]] const transaction_handle th) noexcept final {
61  EAGINE_ASSERT(th == transaction_handle(_subs.size()));
62  _subs.pop();
63  }
64 
65  auto finalize() noexcept -> serialization_errors final {
66  return {};
67  }
68 
69 private:
70  auto current() noexcept -> std::ostream& {
71  if(_subs.empty()) {
72  return _out;
73  }
74  return _subs.top();
75  }
76 
77  std::ostream& _out;
78  std::stack<std::stringstream> _subs;
79 };
