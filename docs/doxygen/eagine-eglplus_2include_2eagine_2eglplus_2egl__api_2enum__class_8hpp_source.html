18 /// @brief Typed wrapper class for EGL enum constants/values.
19 /// @ingroup egl_api_wrap
20 ///
21 /// This wrapper uses the specified Id to group various EGL constants logically
22 /// belonging together and prevents accidental mismatches and use of EGL constants
23 /// from unrelated groups. For example using a config attribute constant in place
24 /// of context attribute constants. Such mismatches result in compilation errors.
25 template <typename Self, identifier_t Id>
26 using egl_int_class =
27  c_api::enum_class<Self, egl_types::int_type, EAGINE_ID_V(EGL), Id>;
28 
29 template <typename Self, identifier_t Id>
30 using egl_enum_class =
31  c_api::enum_class<Self, egl_types::enum_type, EAGINE_ID_V(EGL), Id>;
32 
33 /// @brief Type erasure alias for EGL enum classes.
34 /// @ingroup egl_api_wrap
35 using egl_any_enum_class = c_api::any_enum_class<EAGINE_ID_V(EGL)>;
36 
37 /// @brief Type erasure alias for EGL enum values.
38 /// @ingroup egl_api_wrap
39 using egl_any_enum_value = c_api::any_enum_value<EAGINE_ID_V(EGL)>;
40 
41 static constexpr auto same_enum_class(
42  const egl_any_enum_class a,
43  const egl_any_enum_class b) noexcept -> bool {
44  return ::eagine::c_api::same_enum_class(a, b);
45 }
