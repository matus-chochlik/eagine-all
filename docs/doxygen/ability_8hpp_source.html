17 /// @brief Service providing information about message types handled by endpoint.
18 /// @ingroup msgbus
19 /// @see service_composition
20 /// @see ability_tester
21 template <typename Base = subscriber>
22 class ability_provider : public Base {
23  using This = ability_provider;
24 
25 public:
26  /// @brief Indicates if the given message type is handled by the endpoint.
27  virtual auto can_handle(const message_id) -> bool = 0;
28 
29 protected:
30  using Base::Base;
31 
32  void add_methods() noexcept {
33  Base::add_methods();
34  Base::add_method(
35  this, EAGINE_MSG_MAP(Ability, query, This, _handle_query));
36  }
37 
38 private:
39  auto _handle_query(
40  const message_context& msg_ctx,
41  const stored_message& message) noexcept -> bool {
42  message_id msg_id{};
43  if(default_deserialize_message_type(msg_id, message.content())) {
44  if(can_handle(msg_id)) {
45  msg_ctx.bus_node().respond_to(
46  message,
47  EAGINE_MSG_ID(Ability, response),
48  {message.content()});
49  }
50  }
51  return true;
52  }
53 };
54 //------------------------------------------------------------------------------
55 /// @brief Service consuming information about message types handled by endpoint.
56 /// @ingroup msgbus
57 /// @see service_composition
58 /// @see ability_provider
59 template <typename Base = subscriber>
60 class ability_tester : public Base {
61 
62  using This = ability_tester;
63 
64 public:
65  /// @brief Sends a query to endpoints if they handle the specified message type.
66  /// @see handler_found
67  void find_handler(const message_id msg_id) noexcept {
68  std::array<byte, 32> temp{};
69  auto serialized{default_serialize(msg_id, cover(temp))};
70  EAGINE_ASSERT(serialized);
71 
72  message_view message{extract(serialized)};
73  this->bus_node().broadcast(EAGINE_MSG_ID(Ability, query), message);
74  }
75 
76  /// @brief Triggered on receipt of response about message handling by endpoint.
77  /// @see find_handler
78  signal<void(const identifier_t target_id, const message_id) noexcept>
79  handler_found;
80 
81 protected:
82  using Base::Base;
83 
84  void add_methods() noexcept {
85  Base::add_methods();
86  Base::add_method(
87  this, EAGINE_MSG_MAP(Ability, response, This, _handle_response));
88  }
89 
90 private:
91  auto _handle_response(
92  const message_context&,
93  const stored_message& message) noexcept -> bool {
94  message_id msg_id{};
95  if(default_deserialize_message_type(msg_id, message.content())) {
96  handler_found(message.source_id, msg_id);
97  }
98  return true;
99  }
100 };
