56 //------------------------------------------------------------------------------
57 /// @brief Class storing multiple reusable memory buffer instances.
58 /// @ingroup memory
59 /// @see buffer
60 class buffer_pool {
61 public:
62  /// @brief Default constructors.
63  buffer_pool() noexcept = default;
64 
65  /// @brief Constructor initializing the maximum number of buffers in the pool.
66  explicit buffer_pool(const std::size_t max) noexcept
67  : _max{max} {}
68 
69  /// @brief Gets a buffer with the specified required size.
70  /// @param req_size The returned buffer will have at least this number of bytes.
71  /// @see eat
72  auto get(const span_size_t req_size = 0) -> memory::buffer {
73  memory::buffer result{};
74 #if !EAGINE_LOW_PROFILE
75  _stats._maxc = std::max(_stats._maxc, _pool.size());
76  ++_stats._gets;
77 #endif
78  if(!_pool.empty()) [[likely]] {
79 #if !EAGINE_LOW_PROFILE
80  ++_stats._hits;
81 #endif
82  result = std::move(_pool.back());
83  _pool.pop_back();
84  }
85  result.resize(req_size);
86  return result;
87  }
88 
89  /// @brief Returns the specified buffer back to the pool for further reuse.
90  /// @see get
91  void eat(memory::buffer used) noexcept {
92  const auto old_cap{std_size(used.capacity())};
93  if(_pool.size() < _max) [[likely]] {
94  try {
95  _pool.emplace_back(std::move(used));
96  } catch(...) {
97  }
98  } else {
99 #if !EAGINE_LOW_PROFILE
100  ++_stats._dscs;
101 #endif
102  }
103 #if !EAGINE_LOW_PROFILE
104  ++_stats._eats;
105  _stats._maxs = std::max(_stats._maxs, old_cap);
106 #endif
107  }
108 
109  auto stats() const noexcept
110  -> optional_reference_wrapper<const buffer_pool_stats> {
111 #if !EAGINE_LOW_PROFILE
112  return {_stats};
113 #else
114  return {};
115 #endif
116  }
117 
118 private:
119  const std::size_t _max{1024};
120  std::vector<memory::buffer> _pool;
121 #if !EAGINE_LOW_PROFILE
122  buffer_pool_stats _stats{};
123 #endif
124 };
