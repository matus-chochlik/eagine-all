201  /// @brief Serializes the specified value and enqueues it for sending in message.
202  /// @see post
203  /// @see post_signed
204  /// @see default_serialize
205  template <typename T>
206  auto post_value(
207  const message_id msg_id,
208  T& value,
209  const message_info& info = {}) noexcept -> bool {
210  if(const auto opt_size = max_data_size()) {
211  const auto max_size = extract(opt_size);
212  return _outgoing.push_if(
213  [this, msg_id, &info, &value, max_size](
214  message_id& dst_msg_id, stored_message& message) {
215  if(message.store_value(value, max_size)) {
216  message.assign(info);
217  dst_msg_id = msg_id;
218  return true;
219  }
220  return false;
221  },
222  max_size);
223  }
224  return false;
225  }
