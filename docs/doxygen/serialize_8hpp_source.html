91 /// @brief Serializes a bus message with the specified serializer backend.
92 /// @ingroup msgbus
93 /// @see serialize_message_header
94 /// @see deserialize_message
95 /// @see default_serialize
96 template <typename Backend>
97 auto serialize_message(
98  const message_id msg_id,
99  const message_view& msg,
100  Backend& backend) noexcept -> serialization_errors
101  requires(std::is_base_of_v<serializer_backend, Backend>) {
102 
103  auto errors = serialize_message_header(msg_id, msg, backend);
104 
105  if(!errors) {
106  if(auto sink{backend.sink()}) {
107  errors |= extract(sink).write(msg.data());
108  } else {
109  errors |= serialization_error_code::backend_error;
110  }
111  }
112 
113  return errors;
114 }

142 /// @brief Deserializes a bus message with the specified deserializer backend.
143 /// @ingroup msgbus
144 /// @see deserialize_message_header
145 /// @see serialize_message
146 /// @see default_deserialize
147 template <typename Backend>
148 auto deserialize_message(
149  identifier& class_id,
150  identifier& method_id,
151  stored_message& msg,
152  Backend& backend) noexcept -> deserialization_errors
153  requires(std::is_base_of_v<deserializer_backend, Backend>) {
154 
155  auto errors = deserialize_message_header(class_id, method_id, msg, backend);
156 
157  if(!errors) {
158  if(auto source{backend.source()}) {
159  msg.fetch_all_from(extract(source));
160  } else {
161  errors |= deserialization_error_code::backend_error;
162  }
163  }
164 
165  return errors;
166 }

285 template <typename Backend, typename Value>
286 inline auto stored_message::do_store_value(
287  const Value& value,
288  const span_size_t max_size) noexcept -> bool {
289  _buffer.resize(max_size);
290  block_data_sink sink(cover(_buffer));
291  Backend backend(sink);
292  auto errors = serialize(value, backend);
293  if(!errors) {
294  set_serializer_id(backend.type_id());
295  _buffer.resize(sink.done().size());
296  return true;
297  }
298  return false;
299 }
300 //------------------------------------------------------------------------------
301 template <typename Value>
302 inline auto stored_message::store_value(
303  const Value& value,
304  const span_size_t max_size) noexcept -> bool {
305  return do_store_value<default_serializer_backend>(value, max_size);
306 }

309 inline auto stored_message::do_fetch_value(Value& value) noexcept -> bool {
310  block_data_source source(view(_buffer));
311  Backend backend(source);
312  auto errors = deserialize(value, backend);
313  return !errors;
314 }
