39  auto entry_backend(const identifier, const console_entry_kind) noexcept

48  auto begin_message(
49  const identifier source,
50  const console_entry_id_t parent_id,
51  const console_entry_id_t,
52  const console_entry_kind kind,
53  const string_view format) noexcept -> bool final {
54  _lockable.lock();
55  assign_to(_message.back(), format);
56  _message.swap();
57  _source = source;
58  _kind = kind;
59  _separate = false;
60  if(const auto pos{_hierarchy.find(parent_id)};
61  pos != _hierarchy.end()) {
62  _depth = std::get<1>(*pos);
63  } else {
64  _depth = 0;
65  }
66  return true;
67  }

145  void to_be_continued(
146  const console_entry_id_t parent_id,
147  const console_entry_id_t id) noexcept final {
148  std::size_t depth{0};
149  if(const auto pos{_hierarchy.find(parent_id)};
150  pos != _hierarchy.end()) {
151  depth = std::get<1>(*pos);
152  }
153  _hierarchy.emplace(id, depth + 1);

160 private:
161  template <typename T>
162  auto _to_str(T value) noexcept -> optionally_valid<string_view> {
163  auto b{_temp.data()};
164  auto e{b + _temp.size()};
165  if(const auto conv{std::to_chars(b, e, value)}; conv.ec == std::errc())
166  [[likely]] {
167  return {head(view(_temp), std::distance(b, conv.ptr)), true};
168  }
169  return {};
170  }

172  void _add(const identifier arg, const string_view value) noexcept {
173  _message.back().clear();
174  substitute_variable_into(
175  _message.back(), _message.front(), arg.name(), value);
176  _message.swap();
177  }

179  void _add(const identifier arg, const std::string& value) noexcept {

181  }

183  template <typename T>
184  void _add(const identifier arg, T value) noexcept {
185  if(const auto str_val{_to_str(value)}) {
186  _add(arg, extract(str_val));

188  }
