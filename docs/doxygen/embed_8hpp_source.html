23 /// @brief Class providing access to a const resource block embedded into the executable.
24 /// @ingroup embedding
25 /// @see embed
26 class embedded_resource {
27 public:
28  /// @brief Default constructor.
29  constexpr embedded_resource() noexcept = default;
30 
31  constexpr embedded_resource(
32  const memory::const_block blk,
33  const string_view src_path,
34  const bool packed = false) noexcept
35  : _res_blk{blk}
36  , _src_path{src_path}
37  , _packed{packed} {}
38 
39  /// @brief Returns the path of the file this resource data comes from.
40  constexpr auto source_path() const noexcept -> string_view {
41  return _src_path;
42  }
43 
44  /// @brief Indicates if the resource is packed and needs to be decompressed.
45  /// @see decompress
46  constexpr auto is_packed() const noexcept -> bool {
47  return _packed;
48  }
49 
50  /// @brief Implicit conversion to const block.
51  constexpr operator memory::const_block() const noexcept {
52  EAGINE_ASSERT(!is_packed());
53  return _res_blk;
54  }
55 
56  /// @brief Unpacks this resource into a buffer using the provided compressor.
57  /// @see is_packed
58  auto unpack(data_compressor& comp, memory::buffer& buf) const
59  -> memory::const_block {
60  if(is_packed()) {
61  return {comp.decompress(_res_blk, buf)};
62  }
63  return copy_into(_res_blk, buf);
64  }
65 
66  /// @brief Unpacks this resource into a buffer using compressor from main context.
67  /// @see is_packed
68  auto unpack(main_ctx& ctx) const -> memory::const_block {
69  return unpack(ctx.compressor(), ctx.scratch_space());
70  }
71 
72  /// @brief Unpacks this resource using compressor from a main context object.
73  /// @see is_packed
74  auto unpack(main_ctx_object& mco) const -> memory::const_block {
75  return unpack(mco.main_context());
76  }
77 
78 private:
79  memory::const_block _res_blk{};
80  string_view _src_path{};
81  bool _packed{false};
82 };

84 /// @brief Converts the embedded resource block to a const span of characters.
85 /// @ingroup embedding
86 /// @relates embedded_resource
87 static inline auto as_chars(const embedded_resource& res) noexcept {
88  return as_chars(memory::const_block{res});
89 }

91 extern auto get_embedded_resource(
92  const identifier res_id,
93  const string_view src_path) noexcept -> embedded_resource;
94 
95 /// @brief Triggers the embedding of data from a file on the specified path.
96 /// @ingroup embedding
97 /// @see embedded_resource
98 ///
99 /// This function only works properly if the build system invokes the embed
100 /// generator, that generates appropriate C++ code that defines byte blocks
101 /// with the content of the specified resource files, compiles them and links
102 /// them into the built executable.
103 /// The embed generator is installed as one of the tools and in cmake-based
104 /// build systems the eagine_embed_packed_target_resources and
105 /// eagine_embed_target_resources cmake functions can be used invoke it.
106 static inline auto embed(
107  const identifier res_id,
108  const string_view src_path) noexcept -> embedded_resource {
109  return get_embedded_resource(res_id, src_path);
110 }
