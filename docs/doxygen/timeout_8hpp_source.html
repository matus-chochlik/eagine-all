18 /// @brief Class that measures elapsed time since instance construction.
19 /// @ingroup time_utils
20 /// @see timeout
21 /// @see referencing_timeout
22 class time_measure {
23  using _clock = std::chrono::steady_clock;
24 
25 public:
26  /// @brief Default constructor. Saves the current time.
27  time_measure() noexcept
28  : _start{_clock::now()} {}
29 
30  /// @brief Returns the time since construction using clock's representation.
31  auto elapsed_time() const noexcept {
32  return _clock::now() - _start;
33  }
34 
35  /// @brief Returns the time since construction in seconds.
36  auto seconds() const noexcept {
37  return std::chrono::duration_cast<std::chrono::duration<float>>(
38  elapsed_time());
39  }
40 
41 private:
42  _clock::time_point _start{};
43 };

54 /// @brief Class representing a timeout since construction or reset.
55 /// @ingroup time_utils
56 /// @see time_measure
57 /// @see referencing_timeout
58 /// @see resetting_timeout
59 class timeout {
60  using _clock = std::chrono::steady_clock;
61 
62 public:
63  /// @brief Alias for the duration type used by this timeout.
64  using duration_type = _clock::duration;
65 
66  /// @brief Default construction. Saves current time. Immediately expires.
67  /// @post is_expired()
68  timeout() noexcept
69  : _timeout{_clock::now()} {}
70 
71  /// @brief Construction from the default and initial timeout duration.
72  /// @see reset
73  /// @see period
74  timeout(
75  const _clock::duration duration,
76  const _clock::duration initial) noexcept
77  : _duration{duration}
78  , _timeout{_clock::now() + initial} {}
79 
80  /// @brief Construction from the default and zero initial timeout duration.
81  /// @see reset
82  /// @see period
83  timeout(const _clock::duration duration, const nothing_t) noexcept
84  : timeout{duration, _clock::duration::zero()} {}
85 
86  /// @brief Construction from the default duration.
87  /// @see reset
88  /// @see period
89  timeout(const _clock::duration duration) noexcept
90  : timeout{duration, duration} {}
91 
92  /// @brief Construction from the default duration.
93  /// @see reset
94  /// @see period
95  template <typename R, typename P>
96  timeout(const std::chrono::duration<R, P> duration) noexcept
97  : timeout{std::chrono::duration_cast<_clock::duration>(duration)} {}
98 
99  /// @brief Resets the timeout using the previously specified duration.
100  auto reset() noexcept -> auto& {
101  _timeout = std::chrono::steady_clock::now() + _duration;
102  return *this;
103  }
104 
105  /// @brief Resets the timeout using the specified duration values.
106  auto reset(
107  const _clock::duration duration,
108  const _clock::duration initial) noexcept -> auto& {
109  _duration = duration;
110  _timeout = std::chrono::steady_clock::now() + initial;
111  return *this;
112  }
113 
114  /// @brief Resets the timeout using the specified duration values.
115  /// @post is_expired()
116  auto reset(const _clock::duration duration, const nothing_t) noexcept
117  -> auto& {
118  return reset(duration, _clock::duration::zero());
119  }
120 
121  /// @brief Resets the timeout using the specified duration value.
122  /// @see is_expired
123  auto reset(const _clock::duration duration) noexcept -> auto& {
124  return reset(duration, duration);
125  }
126 
127  /// @brief Returns the time since reset as clocks duration type.
128  auto elapsed_time() const noexcept -> duration_type {
129  return _clock::now() - _timeout + _duration;
130  }
131 
132  /// @brief Indicates if the timeout is expired.
133  /// @see reset
134  /// @see period
135  auto is_expired() const noexcept -> bool {
136  return _clock::now() >= _timeout;
137  }
138 
139  /// @brief Indicates if the timeout is expired.
140  /// @see is_expired
141  explicit operator bool() const noexcept {
142  return is_expired();
143  }
144 
145  /// @brief Returns the timeout period.
146  /// @see reset
147  /// @see is_expired
148  auto period() const noexcept -> const auto& {
149  return _duration;
150  }
151 
152 private:
153  _clock::duration _duration{};
154  _clock::time_point _timeout{};
155 };

157 /// @brief Class representing a timeout since construction or reset.
158 /// @ingroup time_utils
159 /// @see time_measure
160 /// @see timeout
161 /// @see resetting_timeout
162 class referencing_timeout {
163  using _clock = std::chrono::steady_clock;
164 
165 public:
166  /// @brief Alias for the duration type used by this timeout.
167  using duration_type = _clock::duration;
168 
169  /// @brief Constructor initializing the reference to the timeout interval.
170  referencing_timeout(const duration_type& interval) noexcept
171  : _duration{interval}
172  , _prev_reset{_clock::now()} {}
173 
174  /// @brief Resets the timeout using the referenced specified duration.
175  auto reset() noexcept -> auto& {
176  _prev_reset = std::chrono::steady_clock::now();
177  return *this;
178  }
179 
180  /// @brief Returns the time since reset as clocks duration type.
181  auto elapsed_time() const noexcept -> duration_type {
182  return _clock::now() - _prev_reset;
183  }
184 
185  /// @brief Indicates if the timeout is expired.
186  /// @see reset
187  /// @see period
188  auto is_expired() const noexcept -> bool {
189  return _clock::now() >= _prev_reset + _duration;
190  }
191 
192  /// @brief Indicates if the timeout is expired.
193  /// @see is_expired
194  explicit operator bool() const noexcept {
195  return is_expired();
196  }
197 
198  /// @brief Returns the timeout period.
199  /// @see reset
200  /// @see is_expired
201  auto period() const noexcept -> const auto& {
202  return _duration;
203  }
204 
205 private:
206  const _clock::duration& _duration;
207  _clock::time_point _prev_reset{};
208 };
