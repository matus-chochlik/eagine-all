20 template <typename T>
21 struct future_state {
22  timeout too_late{adjusted_duration(std::chrono::seconds{1})};
23  std::function<void(T)> success_handler{};
24  std::function<void()> timeout_handler{};
25 };
26 //------------------------------------------------------------------------------
27 /// @brief Message bus promise class.
28 /// @ingroup msgbus
29 /// @see future
30 /// @see pending_promises
31 template <typename T>
32 class promise {
33 public:
34  /// @brief Default constructor.
35  promise() noexcept = default;
36 
37  promise(std::shared_ptr<future_state<T>>& state) noexcept
38  : _state{state} {}
39 
40  auto should_be_removed() noexcept -> bool {
41  if(const auto state{_state.lock()}) {
42  if(state->too_late) {
43  if(state->timeout_handler) {
44  state->timeout_handler();
45  _state.reset();
46  }
47  } else {
48  return false;
49  }
50  }
51  return true;
52  }
53 
54  /// @brief Fulfills the promise and the corresponding future.
55  void fulfill(T value) noexcept {
56  if(const auto state{_state.lock()}) {
57  _state.reset();
58  if(state->too_late) {
59  if(state->timeout_handler) {
60  state->timeout_handler();
61  }
62  } else {
63  if(state->success_handler) {
64  state->success_handler(std::move(value));
65  }
66  }
67  }
68  }
69 
70 private:
71  std::weak_ptr<future_state<T>> _state{};
72 };

74 /// @brief Message bus future class.
75 /// @ingroup msgbus
76 /// @see promise
77 /// @see pending_promises
78 template <typename T>
79 class future {
80 public:
81  /// @brief Default constructor.
82  future() = default;
83 
84  /// @brief Constructs empty stateless future.
85  future(const nothing_t) noexcept
86  : _state{} {}
87 
88  /// @brief Checks if the future has state (is associated with a promise).
89  explicit operator bool() const noexcept {
90  return bool(_state);
91  }
92 
93  /// @brief Sets the timeout for this future if there is shared state.
94  template <typename R, typename P>
95  auto set_timeout(const std::chrono::duration<R, P> dur) -> future<T>& {
96  if(_state) {
97  _state->too_late.reset(dur);
98  }
99  return *this;
100  }
101 
102  /// @brief Sets the on-success handler.
103  /// @see then
104  /// @see on_timeout
105  auto on_success(const std::function<void(T)>& handler) noexcept
106  -> future<T>& {
107  if(_state) {
108  _state->success_handler = handler;
109  }
110  return *this;
111  }
112 
113  /// @brief Sets the on-timeout handler.
114  /// @see on_success
115  auto on_timeout(const std::function<void()>& handler) noexcept
116  -> future<T>& {
117  if(_state) {
118  _state->timeout_handler = handler;
119  }
120  return *this;
121  }
122 
123  /// @brief Wraps the given handler object and sets it as the on-success handler.
124  /// @see on_success
125  /// @see otherwise
126  template <typename Handler>
127  auto then(Handler handler) noexcept
128  -> future<T>& requires(std::is_invocable_v<Handler, T>) {
129  if(_state) {
130  _state->success_handler = std::function<void(T)>(
131  [state{_state}, handler{std::move(handler)}](T value) {
132  handler(value);
133  });
134  }
135  return *this;
136  }
137 
138  /// @brief Wraps the given handler object and sets it as the on-timeout handler.
139  /// @see on_timeout
140  /// @see then
141  template <typename Handler>
142  auto otherwise(Handler handler) noexcept
143  -> future<T>& requires(std::is_invocable_v<Handler>) {
144  if(_state) {
145  _state->timeout_handler = std::function<void()>(
146  [state{_state}, handler{std::move(handler)}]() { handler(); });
147  }
148  return *this;
149  }
150 
151  /// @brief Returns the associated promise it there is shared state.
152  auto get_promise() noexcept -> promise<T> {
153  return {_state};
154  }
155 
156 private:
157  std::shared_ptr<future_state<T>> _state{
158  std::make_shared<future_state<T>>()};
159 };

161 /// @brief Class that makes new and tracks existing pending message bus promises.
162 /// @ingroup msgbus
163 /// @see promise
164 /// @see future
165 template <typename T>
166 class pending_promises {
167 public:
168  /// @brief Constructs and returns a new message bus future and its unique id.
169  ///
170  /// The returned future can be used to retrieve the promise.
171  auto make() noexcept -> std::tuple<message_sequence_t, future<T>> {
172  future<T> result{};
173  const auto id{++_id_seq};
174  _promises[id] = result.get_promise();
175  return {id, result};
176  }
177 
178  /// @brief Fulfills the promise/future pair idenified by id with the given value.
179  void fulfill(const message_sequence_t id, T value) noexcept {
180  const auto pos = _promises.find(id);
181  if(pos != _promises.end()) {
182  pos->second.fulfill(std::move(value));
183  _promises.erase(pos);
184  }
185  update();
186  }
187 
188  /// @brief Update the internal state of this promise/future tracker.
189  auto update() noexcept -> bool {
190  return _promises.erase_if(
191  [](auto& p) { return p.second.should_be_removed(); }) > 0;
192  }
193 
194  /// @brief Indicates if there are any unfulfilled pending promises.
195  /// @see has_none
196  auto has_some() const noexcept -> bool {
197  return !_promises.empty();
198  }
199 
200  /// @brief Indicates if there are no pending promises.
201  /// @see has_some
202  auto has_none() const noexcept -> bool {
203  return _promises.empty();
204  }
205 
206 private:
207  message_sequence_t _id_seq{0};
208  flat_map<message_sequence_t, promise<T>> _promises{};
209 };
