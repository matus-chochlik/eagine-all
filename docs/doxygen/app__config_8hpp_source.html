37 /// @brief Class for reading application configuration.
38 /// @ingroup main_context
39 ///
40 /// This class allow to read application configuration values from from
41 /// environment variables, command line arguments and/or configuration files.
42 class application_config : public main_ctx_object {
43 public:
44  application_config(main_ctx_parent parent) noexcept
45  : main_ctx_object{EAGINE_ID(AppConfig), parent} {}
46 
47  /// @brief Do potentially expensive pre-initialization and caching.
48  auto preinitialize() noexcept -> application_config& {
49  [[maybe_unused]] auto unused{_impl()};
50  return *this;
51  }

53  /// @brief Checks is the boolean option identified by @p key is set to true.
54  auto is_set(const string_view key, const string_view tag = {}) noexcept
55  -> bool {
56  if(const auto attr{_find_comp_attr(key, tag)}) {
57  bool flag{false};
58  if(attr.select_value(flag, application_config_tag())) {
59  return flag;
60  }
61  }
62  if(const auto arg{_find_prog_arg(key)}) {
63  bool result{true};
64  arg.parse_next(
65  result, application_config_tag(), log_debug_stream());
66  return result;
67  }
68  if(const auto opt_val{_eval_env_var(key)}) {
69  if(const auto converted{from_string<bool>(extract(opt_val))}) {
70  return extract(converted);
71  }
72  }
73  return false;
74  }
75 
76  /// @brief Fetches the configuration value identified by @p key, into @p dest.
77  template <typename T>
78  auto fetch(
79  const string_view key,
80  T& dest,
81  const string_view tag = {}) noexcept -> bool {
82  if(const auto arg{_find_prog_arg(key)}) {
83  if(arg.parse_next(
84  dest, application_config_tag(), log_error_stream())) {
85  return true;
86  } else {
87  log_error("could not parse configuration value '${value}'")
88  .arg(EAGINE_ID(key), key)
89  .arg(EAGINE_ID(value), arg.get());
90  }
91  }
92  if(const auto opt_val{_eval_env_var(key)}) {
93  if(auto converted{from_string<T>(extract(opt_val))}) {
94  dest = std::move(extract(converted));
95  return true;
96  } else {
97  log_error("could not convert configuration value '${value}'")
98  .arg(EAGINE_ID(key), key)
99  .arg(EAGINE_ID(value), extract(opt_val));
100  }
101  }
102  if(const auto attr{_find_comp_attr(key, tag)}) {
103  if(attr.select_value(dest, application_config_tag())) {
104  return true;
105  } else {
106  log_error("could not fetch configuration value '${key}'")
107  .arg(EAGINE_ID(key), key);
108  }
109  }
110  return false;
111  }
112 
113  /// @brief Fetches the configuration values identified by @p key, into @p dest.
114  template <typename T, typename A>
115  auto fetch(
116  const string_view key,
117  std::vector<T, A>& dest,
118  const string_view tag = {}) noexcept {
119  const auto arg_name{_prog_arg_name(key)};
120  for(const auto arg : _prog_args()) {
121  if(arg.is_tag(arg_name)) {
122  T temp{};
123  if(arg.parse_next(
124  temp, application_config_tag(), log_error_stream())) {
125  dest.emplace_back(std::move(temp));
126  } else {
127  log_error("could not parse configuration value '${value}'")
128  .arg(EAGINE_ID(key), key)
129  .arg(EAGINE_ID(value), arg.get());
130  return false;
131  }
132  }
133  }
134  if(const auto opt_val{_eval_env_var(key)}) {
135  if(auto converted{from_string<T>(extract(opt_val))}) {
136  dest.emplace_back(std::move(extract(converted)));
137  } else {
138  log_error("could not convert configuration value '${value}'")
139  .arg(EAGINE_ID(key), key)
140  .arg(EAGINE_ID(value), extract(opt_val));
141  return false;
142  }
143  }
144  if(const auto attr{_find_comp_attr(key, tag)}) {
145  const auto count = attr.value_count();
146  if(count > 0) {
147  dest.resize(dest.size() + std_size(count));
148  if(!attr.select_values(
149  tail(cover(dest), count), application_config_tag())) {
150  log_error("could not fetch configuration values '${key}'")
151  .arg(EAGINE_ID(key), key);
152  return false;
153  }
154  }
155  }
156  return true;
157  }
158 
159  /// @brief Fetches the configuration value identified by @p key, into @p dest.
160  template <typename T, typename P>
161  auto fetch(
162  const string_view key,
163  valid_if<T, P>& dest,
164  const string_view tag) noexcept -> bool {
165  T temp{};
166  if(fetch(key, temp, tag)) {
167  if(dest.is_valid(temp)) {
168  dest = std::move(temp);
169  return true;
170  } else {
171  log_error("value '${value}' is not valid for '${key}'")
172  .arg(EAGINE_ID(value), temp)
173  .arg(EAGINE_ID(key), key);
174  }
175  }
176  return false;
177  }
178 
179  /// @brief Returns the configuration value or type @p T, identified by @p key.
180  template <typename T>
181  auto get(const string_view key, const type_identity<T> = {}) noexcept
182  -> optionally_valid<T> {
183  T temp{};
184  const auto fetched = fetch(key, temp);
185  return {std::move(temp), fetched};
186  }
187 
188  /// @brief Fetches the configuration value identified by @p key, into @p init.
189  template <typename T>
190  auto init(
191  const string_view key,
192  T& initial,
193  const string_view tag = {}) noexcept -> T {
194  fetch(key, initial, tag);
195  return initial;
196  }
197 
198  /// @brief Links a value loader to this configuration instance.
199  /// @see unlink
200  /// @see reload
201  void link(
202  const string_view key,
203  const string_view tag,
204  application_config_value_loader& loader) noexcept;
205 
206  /// @brief Unlinks a value loader from this configuration instance.
207  /// @see link
208  /// @see reload
209  void unlink(
210  const string_view key,
211  const string_view tag,
212  application_config_value_loader& loader) noexcept;
213 
214  /// @brief Reloads the linked loadable configuration values.
215  /// @see link
216  /// @see unlink
217  void reload() noexcept;
218 
219 private:
220  std::shared_ptr<application_config_impl> _pimpl;
221  auto _impl() noexcept -> application_config_impl*;
222 
223  auto _find_comp_attr(const string_view key, const string_view tag) noexcept
224  -> valtree::compound_attribute;
225 
226  auto _prog_args() noexcept -> const program_args&;
227  auto _prog_arg_name(const string_view key) noexcept -> std::string;
228  auto _find_prog_arg(const string_view key) noexcept -> program_arg;
229  auto _eval_env_var(const string_view key) noexcept
230  -> optionally_valid<string_view>;
231 };
