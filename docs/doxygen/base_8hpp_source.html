23 /// @brief Policy for optionally valid values, indicated by a boolean flag.
24 /// @ingroup valid_if
25 struct valid_flag_policy {

31  : _is_valid{is_valid} {}

33  /// @brief Returns value validity depending on internally stored flag.

50 /// @brief Basic template for conditionally-valid values.
51 /// @ingroup valid_if
52 /// @tparam T type of the stored, conditionally valid value.
53 /// @tparam Policy indicates under what conditions is the stored value valid.
54 template <typename T, typename Policy, typename DoLog, typename... P>
55 class basic_valid_if {

230  /// @brief Calls the specified function if the stored value is invalid.
231  /// @param func the function to be called.
232  /// @param p additional parameters for the policy validity check function.
233  template <typename Func>
234  auto call_if_invalid(Func func, P... p) -> auto& {
235  if(!is_valid(p...)) {
236  func(_do_log, _value, p...);
237  }
238  return *this;
239  }

241  /// @brief Throws an exception if the stored value is invalid.
242  /// @param p additional parameters for the policy validity check function.
243  /// @throws std::runtime_error
244  void throw_if_invalid(P... p) const {
245  if(!is_valid(p...)) {
246  std::stringstream ss;
247  log_invalid(ss, p...);
248  throw std::runtime_error(ss.str());
249  }

297  /// @brief Calls the specified function if the stored value is valid.
298  /// @param func the function to be called.
299  template <typename Func>
300  auto then(const Func& func) const -> basic_valid_if<
301  std::invoke_result_t<Func, T>,
302  valid_flag_policy,
303  typename valid_flag_policy::
304  do_log> requires(!std::is_same_v<std::invoke_result_t<Func, T>, void>) {
305  if(is_valid()) {
306  return {func(this->value_anyway()), true};
307  }
308  return {};
309  }

319  /// @brief Calls a binary transforming function on {value, is_valid()} pair.
320  /// @param func the function to be called.
321  template <typename Func>
322  constexpr auto transformed(Func func, P... p) const noexcept
323  -> basic_valid_if<
324  std::invoke_result_t<Func, T, bool>,
325  valid_flag_policy,
326  typename valid_flag_policy::do_log> {
327  const auto v{is_valid(p...)};
328  auto r{func(_value, v)};
329  if constexpr(extractable<decltype(r)>) {
330  return r;
331  } else {
332  return {std::move(r), v};
333  }
334  }

369 /// @brief Equality comparison of two conditionally valid values.
370 /// @ingroup valid_if
371 template <typename T, typename Po1, typename Po2, typename L1, typename L2>
372 static constexpr auto operator==(
373  const basic_valid_if<T, Po1, L1>& v1,
374  const basic_valid_if<T, Po2, L2>& v2) noexcept -> tribool {
