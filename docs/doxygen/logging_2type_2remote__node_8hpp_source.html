18 static inline auto adapt_entry_arg(
19  const identifier name,
20  const msgbus::remote_node& value) {
21  return [name, value](auto& backend) noexcept {
22  backend.add_unsigned(
23  name, EAGINE_ID(uint64), extract_or(value.id(), 0U));
24 
25  if(const auto opt_id{value.instance_id()}) {
26  backend.add_unsigned(
27  EAGINE_ID(instanceId), EAGINE_ID(uint32), extract(opt_id));
28  }
29 
30  backend.add_string(
31  EAGINE_ID(nodeKind), EAGINE_ID(enum), enumerator_name(value.kind()));
32 
33  backend.add_adapted(
34  EAGINE_ID(isRutrNode), yes_no_maybe(value.is_router_node()));
35  backend.add_adapted(
36  EAGINE_ID(isBrdgNode), yes_no_maybe(value.is_bridge_node()));
37  backend.add_adapted(
38  EAGINE_ID(isPingable), yes_no_maybe(value.is_pingable()));
39  backend.add_adapted(
40  EAGINE_ID(isRespnsve), yes_no_maybe(value.is_responsive()));
41 
42  if(const auto opt_rate{value.ping_success_rate()}) {
43  backend.add_float(
44  EAGINE_ID(pingSucces), EAGINE_ID(Ratio), extract(opt_rate));
45  }
46  if(const auto opt_bld{value.instance().build()}) {
47  backend.add_adapted(EAGINE_ID(buildInfo), extract(opt_bld));
48  }
49 
50  if(const auto opt_name{value.display_name()}) {
51  backend.add_string(
52  EAGINE_ID(dispName), EAGINE_ID(string), extract(opt_name));
53  }
54 
55  if(const auto opt_desc{value.description()}) {
56  backend.add_string(
57  EAGINE_ID(descrption), EAGINE_ID(string), extract(opt_desc));
58  }
59  };
60 }
