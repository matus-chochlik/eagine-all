46 //------------------------------------------------------------------------------
47 class keyboard_distance {
48 public:
49  template <typename Layout>
50  keyboard_distance(const Layout& layout, const float multiplier = 1.F) {
51  layout.for_each_char_coord([&](char lc, auto lp) {
52  layout.for_each_char_coord([&](char rc, auto rp) {
53  if(lc != rc) {
54  _key_dist[std::pair<char, char>{lc, rc}] =
55  multiplier * math::distance(lp, rp);
56  }
57  });
58  });
59  }

62 
63 private:
