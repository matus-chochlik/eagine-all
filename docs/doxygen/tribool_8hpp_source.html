28 /// @brief Weak tri-state boolean value.
29 /// @see tribool
30 /// @see indeterminate
31 class weakbool {
32  using _value_t = _tribool_value_t;
33 
34 public:
35  /// @brief Returns true, if the stored value is not @c false.
36  constexpr explicit operator bool() const noexcept {
37  return _value != _value_t::_false;
38  }
39 
40  /// @brief Returns true, if the stored value is not @c true.
41  constexpr auto operator!() const noexcept {
42  return _value != _value_t::_true;
43  }
44 
45  ///@ brief Checks if the stored value is indeterminate.
46  constexpr auto is(const indeterminate_t) const noexcept {
47  return _value == _value_t::_unknown;
48  }
49 
50 private:
51  _value_t _value;

53  friend class tribool;

59 /// @brief Tri-state boolean value.
60 /// @see weakbool
61 /// @see indeterminate
62 class tribool {
63  using _value_t = _tribool_value_t;
64 
65 public:
66  /// @brief Default constructor.
67  constexpr tribool() noexcept = default;
68 
69  /// @brief Constructions from boolean value.
70  constexpr tribool(const bool value) noexcept
71  : _value{value ? _value_t::_true : _value_t::_false} {}
72 
73  /// @brief Constructions from indeterminate value.
74  constexpr tribool(const indeterminate_t) noexcept
75  : _value{_value_t::_unknown} {}
76 
77  /// @brief Construction with separate true/false, known/unknown arguments.
78  constexpr tribool(const bool value, const bool is_unknown) noexcept
79  : _value(
80  is_unknown ? _value_t::_unknown
81  : value ? _value_t::_true
82  : _value_t::_false) {}
83 
84  /// @brief Returns true, if the stored value is true.
85  constexpr explicit operator bool() const noexcept {
86  return _value == _value_t::_true;
87  }
88 
89  /// @brief Returns true, if the stored value is false.
90  constexpr auto operator!() const noexcept {
91  return _value == _value_t::_false;
92  }
93 
94  /// @brief Returns true, if the stored value is indeterminate.
95  constexpr auto operator*() const noexcept -> bool {
96  return _value == _value_t::_unknown;
97  }
98 
99  /// @brief Converts this value to @c weakbool.
100  constexpr auto operator~() const noexcept -> weakbool {
101  return weakbool{_value};
102  }
103 
104  /// @brief Returns true if the stored value is known and equal to @p value.
105  constexpr auto is(const bool value) const noexcept -> bool {
106  return _value == (value ? _value_t::_true : _value_t::_false);
107  }
108 
109  /// @brief Returns true if the stored value is indeterminate.
110  constexpr auto is(const indeterminate_t) const noexcept -> bool {
111  return *(*this);
112  }
113 
114  /// @brief Equality comparison.
115  friend constexpr auto operator==(const tribool a, const tribool b) noexcept
116  -> tribool {
117  return {a._value == b._value, (*a || *b)};
118  }
119 
120  /// @brief Non-equality comparison.
121  friend constexpr auto operator!=(const tribool a, const tribool b) noexcept
122  -> tribool {
123  return {a._value != b._value, (*a || *b)};
124  }
125 
126 private:
127  _value_t _value{_value_t::_false};
128 };

130 /// @brief Tri-state boolean and operator.
131 constexpr auto operator&&(const tribool a, const tribool b) noexcept {
132  return !a ? tribool{false}
133  : a ? b
134  : !b ? tribool{false}
135  : tribool{indeterminate};
136 }

147 constexpr auto operator||(const tribool a, const tribool b) noexcept {
148  return a ? tribool{true}
