52 template <typename Result, typename Deserializer, typename Source, bool NoExcept>
53 class callback_invoker_base {
54 
55 public:
56  auto fulfill_by(
57  const message_context& msg_ctx,
58  const stored_message& response) noexcept -> bool {
59  Result result{};
60 
61  _source.reset(response.content());
62  Deserializer read_backend(_source);
63 
64  if(response.has_serializer_id(read_backend.type_id())) {
65  const auto errors{deserialize(result, read_backend)};
66  if(!errors) {
67  const result_context res_ctx{
68  msg_ctx, response.source_id, response.sequence_no};
69  _callback(res_ctx, std::move(result));
70  }
71  }
72  return true;
73  }
74 
75 protected:
76  using _callback_t =
77  callable_ref<void(const result_context&, Result&&) noexcept(NoExcept)>;
78  _callback_t _callback{};
79 
80 private:
81  Source _source{};
82 };

152  auto invoke_on(

156  memory::block buffer,
157  Args&&... args) -> bool {

200 template <typename Result, typename Deserializer, typename Source>
201 class invoker_base {
202 public:
203  auto fulfill_by(
204  const message_context&,
205  const stored_message& message) noexcept -> bool {
206  const auto invocation_id = message.sequence_no;
207  std::remove_cv_t<std::remove_reference_t<Result>> result{};
208 
209  _source.reset(message.content());
210  Deserializer read_backend(_source);
211 
212  if(message.has_serializer_id(read_backend.type_id())) {
213  const auto errors{deserialize(result, read_backend)};
214  if(!errors) {
215  _results.fulfill(invocation_id, result);
216  }
217  }
218  return true;
219  }
220 
221  constexpr auto map_fulfill_by(const message_id msg_id) noexcept {
222  return std::tuple<
223  invoker_base*,
224  message_handler_map<EAGINE_MEM_FUNC_T(invoker_base, fulfill_by)>>(
225  this, msg_id);
226  }
227 
228  constexpr auto operator[](const message_id msg_id) noexcept {
229  return map_fulfill_by(msg_id);
230  }
231 
232  auto has_pending() const noexcept -> bool {
233  return _results.has_some();
234  }
235 
236  auto is_done() const noexcept -> bool {
237  return _results.has_none();
238  }
239 
240 protected:
241  pending_promises<Result> _results{};
242 
243 private:
244  Source _source{};
245 };

256 template <
257  typename Result,
258  typename... Params,
259  typename Serializer,
260  typename Deserializer,
261  typename Sink,
262  typename Source,
263  std::size_t MaxDataSize>
264 class invoker<Result(Params...), Serializer, Deserializer, Sink, Source, MaxDataSize>
265  : public invoker_base<Result, Deserializer, Source> {
266 public:
267  auto invoke_on(
268  endpoint& bus,
269  const identifier_t target_id,
270  const message_id msg_id,
271  memory::block buffer,
272  std::add_lvalue_reference_t<std::add_const_t<Params>>... args) noexcept
273  -> future<Result> {
274  const auto [invocation_id, result] = this->_results.make();
275 
276  auto tupl{std::tie(args...)};
277 
278  block_data_sink sink(buffer);
279  Serializer write_backend(sink);
280 
281  const auto errors = serialize(tupl, write_backend);
282  if(!errors) {
283  message_view message{sink.done()};
284  message.set_serializer_id(write_backend.type_id());
285  message.set_target_id(target_id);
286  message.set_sequence_no(invocation_id);
287  bus.post(msg_id, message);
288 
289  return result;
290  }
291  return nothing;
292  }
