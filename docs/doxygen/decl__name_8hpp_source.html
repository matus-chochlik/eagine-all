34 public:
35  static constexpr const span_size_t max_length = decl_name::max_length;
36 
37  auto assign(const string_view str) -> bool {
38  if(str.size() <= max_length) {
39  memory::copy(str, cover(_storage));
40  _length = str.size();
41  return true;
42  }
43  return false;
44  }
45 
46  auto get() noexcept {
47  return head(cover(_storage), _length);
48  }
49 
50  auto get() const noexcept {
51  return head(view(_storage), _length);
52  }
53 
54 private:
55  std::array<char, max_length> _storage{};
56  span_size_t _length{0};
