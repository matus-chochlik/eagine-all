211  typename... QueryClasses,

213  struct query_function<
214  wrapper,
215  QueryResult,
216  mp_list<PreParams...>,
217  mp_list<QueryClasses...>,
218  mp_list<PostParams...>>
219  : adapted_function<
220  wrapper,
221  void(PreParams..., enum_type, PostParams..., QueryResult*)> {
222  using base = adapted_function<
223  wrapper,
224  void(PreParams..., enum_type, PostParams..., QueryResult*)>;
225  using base::base;
226 
227  template <typename Query>
228  constexpr auto operator()(
229  PreParams... pre_params,
230  Query query,
231  PostParams... post_params) const noexcept
232  requires(
233  (true || ... || c_api::is_enum_class_value_v<QueryClasses, Query>)&&(
234  !std::is_array_v<typename Query::tag_type>)) {
235  using RV = typename Query::tag_type;
236  QueryResult result{};
237  return base::operator()(
238  pre_params..., enum_type(query), post_params..., &result)
239  .replaced_with(result)
240  .cast_to(type_identity<RV>{});
241  }
242 
243  template <typename Query>
244  auto operator()(
245  PreParams... pre_params,
246  Query query,
247  PostParams... post_params,
248  span<QueryResult> dest) const noexcept
249  requires(
250  true || ... || c_api::is_enum_class_value_v<QueryClasses, Query>) {
251  if constexpr(std::is_array_v<typename Query::tag_type>) {
252  EAGINE_ASSERT(
253  dest.size() >= std::extent_v<typename Query::tag_type>);
254  } else {
255  EAGINE_ASSERT(dest.size() > 0);
256  }
257  return base::operator()(
258  pre_params..., enum_type(query), post_params..., dest.data());
