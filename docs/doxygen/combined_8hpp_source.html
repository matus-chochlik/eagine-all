87 //------------------------------------------------------------------------------
88 static inline auto combine(std::unique_ptr<generator>&& gen) {
89  std::vector<std::unique_ptr<generator>> v;
90  v.reserve(1);
91  v.emplace_back(std::move(gen));
92  return std::make_unique<combined_gen>(std::move(v));
93 }
94 //------------------------------------------------------------------------------
95 template <std::size_t N>
96 static inline auto combine(std::array<std::unique_ptr<generator>, N>&& gens) {
97  std::vector<std::unique_ptr<generator>> v;
98  v.reserve(N);
99  for(auto& gen : gens) {
100  v.emplace_back(std::move(gen));
101  }
102  return std::make_unique<combined_gen>(std::move(v));
103 }
