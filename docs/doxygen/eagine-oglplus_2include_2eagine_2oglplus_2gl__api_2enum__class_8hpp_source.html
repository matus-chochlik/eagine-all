18 /// @brief Typed wrapper class for GLenum constants/values.
19 /// @ingroup gl_api_wrap
20 /// @see gl_bitfield_class
21 /// @see gl_ubyte_class
22 /// @see gl_bool_class
23 /// @see gl_any_enum_class
24 ///
25 /// This wrapper uses the specified Id to group various GL constants logically
26 /// belonging together and prevents accidental mismatches and use of GL constants
27 /// from unrelated groups. For example using a shader type constant in place
28 /// of texture type constants. Such mismatches result in compilation errors.
29 template <typename Self, identifier_t Id>
30 using gl_enum_class =
31  c_api::enum_class<Self, gl_types::enum_type, EAGINE_ID_V(GL), Id>;
