460 /// @brief Class managing options for an application with video / audio rendering.
461 /// @ingroup application
462 /// @see video_options

464 /// @see application_config

466 /// Instances of this class read the application configuration and provide

468 class launch_options : public main_ctx_object {

472 
473  /// @brief Returns the title of the application.

475 
476  /// @brief Specifies that no video rendering should be used.
477  /// @see require_video
478  /// @see no_audio
479  /// @see no_input
480  auto no_video() noexcept -> auto& {
481  _video_opts.clear();
482  return *this;
483  }
484 
485  /// @brief Requests a new video rendering context and surface.
486  /// @param kind the kind of the requested context.
487  /// @see instance identifier for the requested context.
488  /// @see no_video
489  /// @see require_audio
490  /// @see video_requirements
491  auto require_video(
492  const video_context_kind kind = video_context_kind::opengl,
493  const identifier instance = {}) -> video_options&;
494 
495  /// @brief Returns all current video rendering context options.
496  /// @see require_video

498  auto video_requirements() const noexcept
499  -> const std::map<identifier, video_options>& {
500  return _video_opts;
501  }
502 
503  /// @brief Specifies that no audio playback or recording should be used.
504  /// @see require_audio
505  /// @see no_video
506  /// @see no_input
507  auto no_audio() noexcept -> auto& {
508  _audio_opts.clear();
509  return *this;
510  }
511 
512  /// @brief Requests a new audio playback and recording context.
513  /// @param kind the kind of the requested context.

517  /// @see audio_requirements

533  /// @see no_audio

535  _requires_input = false;
536  return *this;
537  }
538 
539  /// @brief Specifies that user input handling is required.

544  return *this;
545  }
546

551 
552  /// @brief Says if the application ran long enough according to the configuration.
553  template <typename R, typename P>
554  auto enough_run_time(
555  const std::chrono::duration<R, P> run_time) const noexcept -> bool {
556  return _max_run_time && extract(_max_run_time) <= run_time;
557  }
558 
559  /// @brief Says if the application rendered enough frames according to the configuration.
560  auto enough_frames(const span_size_t frame_no) const noexcept -> bool {
561  return _max_frames && extract(_max_frames) <= frame_no;
562  }
563

574  bool _requires_input{false};
