19 /// @brief A single key/value pair for a key/value list.
20 /// @ingroup type_utils
21 /// @see key_value_list
22 template <typename Traits>
23 struct key_value_list_element {
24  /// @brief Alias for the key type.
25  using key_type = typename Traits::key_type;
26  /// @brief Alias for the value type.
27  using value_type = typename Traits::value_type;
28 
29  key_type _key;
30  value_type _value;
31 
32  /// @brief Construction from the key and the value.
33  constexpr key_value_list_element(
34  const key_type key,
35  const value_type value) noexcept
36  : _key{key}
37  , _value{value} {}
38 };

40 /// @brief Template for classes wrapping static key/value typically attribute lists.
41 /// @ingroup type_utils
42 /// @tparam Traits policy class customizing the instantiation.
43 template <typename Traits>
44 class key_value_list {
45 public:
46  using key_type = typename Traits::key_type;
47  using conv_type = typename Traits::conv_type;
48  using value_type = typename Traits::value_type;
49 
50  constexpr key_value_list() noexcept = default;
51 
52  explicit constexpr key_value_list(
53  const key_value_list_element<Traits>& e) noexcept {
54  _values.reserve(3);
55  _values.push_back(static_cast<conv_type>(e._key));
56  _values.push_back(e._value);
57  _values.push_back(Traits::terminator());
58  }
59 
60  explicit constexpr key_value_list(
61  const key_value_list_element<Traits>& l,
62  const key_value_list_element<Traits>& r) noexcept {
63  _values.reserve(5);
64  _values.push_back(static_cast<conv_type>(l._key));
65  _values.push_back(l._value);
66  _values.push_back(static_cast<conv_type>(r._key));
67  _values.push_back(r._value);
68  _values.push_back(Traits::terminator());
69  }
70 
71  /// @brief Returns a pointer to the internal element array.
72  /// @see size
73  auto data() const noexcept -> const value_type* {
74  return _values.data();
75  }
76 
77  /// @brief Returns the size of the element array (including the terminating zero).
78  /// @see data
79  constexpr auto size() const noexcept -> span_size_t {
80  return span_size(_values.size());
81  }
82 
83  /// @brief Appends a key/value pair and returns a new extended list.
84  constexpr auto append(const key_value_list_element<Traits>& key_val)
85  const noexcept -> key_value_list<Traits> {
86  return {*this, key_val};
87  }
88 
89  constexpr auto get() const noexcept {
90  return view(_values);
91  }
92 
93  constexpr operator span<const value_type>() const noexcept {
94  return view(_values);
95  }
96 
97 private:
98  constexpr key_value_list(
99  const key_value_list& base,
100  const key_value_list_element<Traits>& key_val) noexcept {
101  _values.reserve(base._values.size() + 2);
102  for(const auto i : integer_range(_values.size() - 1)) {
103  _values.push_back(base._values[i]);
104  }
105  _values.push_back(static_cast<conv_type>(key_val._key));
106  _values.push_back(key_val._value);
107  _values.push_back(Traits::terminator());
108  }
109 
110  std::vector<value_type> _values{Traits::terminator()};
111 };

113 /// @brief Concatenates two individual key/value elements into a key/value list.
114 /// @ingroup type_utils
115 /// @relates key_value_list
116 /// @see key_value_list_element
117 template <typename Traits>
118 static constexpr auto operator+(
119  const key_value_list_element<Traits>& l,
120  const key_value_list_element<Traits>& r) noexcept -> key_value_list<Traits> {
121  return key_value_list<Traits>(l, r);
122 }
123 
124 /// @brief Adds a key/value pair into a key/value list, returns a new list.
125 /// @ingroup type_utils
126 /// @relates key_value_list
127 /// @see key_value_list_element
128 template <typename Traits>
129 static constexpr auto operator+(
130  const key_value_list<Traits>& l,
131  const key_value_list_element<Traits>& r) noexcept -> key_value_list<Traits> {
132  return l.append(r);
133 }
