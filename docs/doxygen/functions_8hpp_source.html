225 static constexpr auto binomial(const T n, const T k) noexcept -> T {
226  return ((n >= 0) && (k >= 0) && (k <= n))
227  ? (factorial(n) / (factorial(k) * factorial(n - k)))
228  : 0;

231 /// @brief Helper class for bezier curve segment calculations.
232 /// @ingroup math
233 /// @tparam Type the interpolated type.
234 /// @tparam Parameter the curve interpolation parameter type.
235 /// @tparam N the curve order.
236 /// @see bezier_curves
237 template <typename Type, typename Parameter, int N>
238 struct bezier_t {
239 public:
240  /// @brief Interpolate from control points in pack @p p at position @p t.
241  template <typename... Points>
242  constexpr auto operator()(const Parameter t, Points&&... p) const noexcept
243  requires(sizeof...(Points) == N) {
244  return _calc(N - 1, 0, t, std::forward<Points>(p)...);
245  }
246 
247  /// @brief Interpolate from control points in span @p p at position @p t.
248  template <typename P, typename S>
249  auto operator()(const Parameter t, memory::basic_span<const Type, P, S> p)
250  const noexcept {
251  return _calc(N - 1, 0, t, p, std::make_index_sequence<N>());
252  }
253 
254 private:
255  static constexpr auto _coef(
256  const int m,
257  const int i,
258  const Parameter t) noexcept {
259  using std::pow;
260  return pow(t, Parameter(i)) * pow(1 - t, Parameter(m - i)) *
261  Parameter(binomial(m, i));
262  }
263 
264  static constexpr auto _calc(const int, const int, const Parameter) noexcept {
265  return Type{0};
266  }
267 
268  template <typename... P>
269  static constexpr auto _calc(
270  const int m,
271  const int i,
272  const Parameter t,
273  const Type first,
274  const P... rest) noexcept -> Type {
275  return first * _coef(m, i, t) + _calc(m, i + 1, t, rest...);
276  }
277 
278  template <typename P, typename S, std::size_t... I>
279  static constexpr auto _calc(
280  const int m,
281  const int i,
282  const Parameter t,
283  const memory::basic_span<const Type, P, S> p,
284  const std::index_sequence<I...>) noexcept -> Type {
285  return _calc(m, i, t, p[I]...);
286  }
287 };
