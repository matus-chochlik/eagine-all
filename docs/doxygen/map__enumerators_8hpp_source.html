33 //------------------------------------------------------------------------------
34 template <typename T, std::size_t N>
35 using enumerator_map_type = std::array<const name_and_enumerator<T>, N>;
36 //------------------------------------------------------------------------------
37 template <typename T, typename Selector>
38 struct does_have_enumerator_mapping {
39 private:
40  template <
41  typename X,
42  typename = decltype(enumerator_mapping(type_identity<X>{}, Selector{}))>
43  static auto _test(X*) -> std::true_type;
44  static auto _test(...) -> std::false_type;
45 
46 public:
47  // NOLINTNEXTLINE(hicpp-vararg)
48  using type = decltype(_test(static_cast<std::remove_cv_t<T>*>(nullptr)));
49 };
50 //------------------------------------------------------------------------------
51 template <typename T, typename Selector = default_selector_t>
52 using has_enumerator_mapping_t =
53  typename does_have_enumerator_mapping<T, Selector>::type;
54 
55 template <typename T, typename Selector = default_selector_t>
56 constexpr const bool has_enumerator_mapping_v =
57  has_enumerator_mapping_t<T, Selector>::value;
58 //------------------------------------------------------------------------------
59 #if EAGINE_CXX_REFLECTION
60 template <typename Enum, std::size_t... I>
61 consteval auto _make_enumerator_mapping(
62  const std::index_sequence<I...>) noexcept {
63  namespace meta = std::experimental::meta;
64  const auto me = meta::members_of(^Enum, meta::is_enumerator);
65  return enumerator_map_type<Enum, sizeof...(I)>{
66  {{{immediate_function, meta::name_of(_meta_range_at(me, I))},
67  ([:_meta_range_at(me, I):])}...}};
68 }
69 //------------------------------------------------------------------------------
70 template <typename Enum>
71 static constexpr const auto _reflected_enumerator_mapping =
72  _make_enumerator_mapping<Enum>(
73  std::make_index_sequence<size(std::experimental::meta::members_of(
74  ^Enum,
75  std::experimental::meta::is_enumerator))>{});
76 //------------------------------------------------------------------------------
77 template <typename Enum, typename Selector>
78 constexpr auto enumerator_mapping(
79  const type_identity<Enum>,
80  const Selector) noexcept -> const auto& requires(std::is_enum_v<Enum>) {
81  return _reflected_enumerator_mapping<Enum>;
82 }
