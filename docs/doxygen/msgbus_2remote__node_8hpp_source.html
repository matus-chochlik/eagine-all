49  /// @brief The host information has appeared or changed.
50  host_info = 1U << 3U,
51  /// @brief The build information has appeared or changed.
52  build_info = 1U << 4U,
53  /// @brief The application information has appeared or changed.
54  application_info = 1U << 5U,
55  /// @brief The endpoint information has appeared or changed.
56  endpoint_info = 1U << 6U,
57  /// @brief New remotly callable methods have been added.
58  methods_added = 1U << 7U,
59  /// @brief New remotly callable methods have been removed.
60  methods_removed = 1U << 8U,
61  /// @brief Node started responding to pings.
62  started_responding = 1U << 9U,
63  /// @brief Node stopped responding to pings.
64  stopped_responding = 1U << 10U,
65  /// @brief Node ping response rate.
66  response_rate = 1U << 11U,
67  /// @brief The hardware configuration information has appeared or changed.
68  hardware_config = 1U << 12U,
69  /// @brief New sensor values have appeared or changed.
70  sensor_values = 1U << 13U,
71  /// @brief New statistic values have appeared or changed.
72  statistics = 1U << 14U,
73  /// @brief The bus connection information has appeared or changed.

896 public:
897  node_connection() noexcept = default;
898  node_connection(
899  const identifier_t id1,
900  const identifier_t id2,
901  remote_node_tracker tracker) noexcept
902  : _id1{id1}
903  , _id2{id2}
904  , _tracker{std::move(tracker)} {}

906  /// @brief Indicates if this is not-empty and has actual information.
907  explicit operator bool() const noexcept {
908  return bool(_pimpl);
909  }
910 
911  /// @brief Indicates if the connection connects node with the specified id.
912  /// @see opposite_id
913  auto connects(const identifier_t id) const noexcept {
914  return (_id1 == id) || (_id2 == id);
915  }
916 
917  /// @brief Indicates if the connection connects nodes with the specified id.
918  /// @see opposite_id
919  auto connects(const identifier_t id1, const identifier_t id2)
920  const noexcept {
921  return ((_id1 == id1) && (_id2 == id2)) ||
922  ((_id1 == id2) && (_id2 == id1));
923  }

925  /// @brief Returns the id of the node opposite to the node with id in argument.
926  /// @see connects
927  auto opposite_id(const identifier_t id) const noexcept
928  -> valid_if_not_zero<identifier_t> {
929  if(_id1 == id) {
930  return {_id2};
931  }
932  if(_id2 == id) {
933  return {_id1};
934  }
935  return {0U};
936  }

939  auto kind() const noexcept -> connection_kind;
940 
941  /// @brief Returns the message block usage ratio for the connection.
942  auto block_usage_ratio() const noexcept -> valid_if_nonnegative<float>;
943 
944  /// @brief Returns count of bytes per second sent through the connection.
945  auto bytes_per_second() const noexcept -> valid_if_nonnegative<float>;
946

948  std::shared_ptr<node_connection_impl> _pimpl{};
