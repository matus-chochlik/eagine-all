516 static inline auto slice_inside_brackets(
517  basic_span<T, P, S> spn,
518  const B left,
519  const B right) noexcept -> basic_span<T, P, S> {
520 
521  if(const auto found{find_element(spn, left)}) {

785  typename Result,
786  typename T,
787  typename P1,
788  typename S1,
789  typename P2,
790  typename S2,
791  typename BinaryFunction>
792 static auto basic_edit_distance(
793  const basic_span<const T, P1, S1> s1,
794  const basic_span<const T, P2, S2> s2,
795  BinaryFunction get_distance) -> Result {
796  const auto l1 = s1.size();
797  const auto l2 = s2.size();
798 
799  std::vector<Result> column;
800  column.resize(integer(l1 + 1));
801  std::iota(column.begin(), column.end(), Result(0));
802 
803  for(span_size_t x = 1; x <= l2; x++) {
804  column[0] = Result(x);
805  auto prev_diagonal = Result(x) - 1;
806 
807  for(span_size_t y = 1; y <= l1; y++) {
808  const auto curr_diagonal = column[y];
809  const auto prev_min = std::min(column[y] + 1, column[y - 1] + 1);
810  const auto new_dist =
811  prev_diagonal + Result(get_distance(s1[y - 1], s2[x - 1]));
812  column[y] = std::min(prev_min, new_dist);
813  prev_diagonal = curr_diagonal;
814  }
815  }
816  return column[l1];
817 }
