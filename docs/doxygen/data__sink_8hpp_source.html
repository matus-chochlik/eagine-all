20 /// @brief Abstract base class for serialization data sinks.
21 /// @ingroup serialization
22 /// @see data_source
23 /// @see serializer_backend
24 struct serializer_data_sink : abstract<serializer_data_sink> {
25 
26  /// @brief Alias for the operation result type.
27  using result = serialization_errors;
28 
29  /// @brief Returns the remaining available size for data in this sink.
30  virtual auto remaining_size() noexcept -> span_size_t = 0;
31 
32  /// @brief Writes a block of data into this sink.
33  virtual auto write(memory::const_block data) noexcept -> result = 0;
34 
35  /// @brief Writes a single string character into this sink.
36  auto write(const char chr) noexcept -> result {
37  return this->write(as_bytes(view_one(chr)));
38  }
39 
40  /// @brief Writes a string view into this sink.
41  auto write(const string_view str) noexcept -> result {
42  return this->write(as_bytes(str));
43  }
44 
45  /// @brief Writes as much as possible from a split data block.
46  auto write_some(memory::const_split_block data) noexcept
47  -> serialization_result<memory::const_split_block> {
48  const auto before{remaining_size()};
49  const auto errors{write(data.tail())};
50  if(!errors) {
51  return {data.skip_to_end()};
52  }
53  if(errors.has_at_most(serialization_error_code::incomplete_write)) {
54  return {data.advance(before)};
55  }
56  return {data, errors};
57  }
58 
59  /// @brief Alias for type indentifying sink transaction that can be rolled back.
60  using transaction_handle = std::uintptr_t;
61 
62  /// @brief Begins a write transaction on this data sink.
63  /// @see commit
64  /// @see rollback
65  virtual auto begin_work() noexcept -> transaction_handle = 0;
66 
67  /// @brief Commits writes done as a part of transaction identified by argument.
68  /// @see begin_work
69  /// @see rollback
70  virtual auto commit(const transaction_handle) noexcept -> result = 0;
71 
72  /// @brief Rolls-back writes done as a part of transaction identified by argument.
73  /// @see begin_work
74  /// @see commit
75  virtual void rollback(const transaction_handle) noexcept = 0;
76 
77  /// @brief Does additional finalization, like compression after serialization operation.
78  virtual auto finalize() noexcept -> result = 0;
79 };
