276  template <unsigned S>
277  struct rank_info {
278  default_sudoku_board_traits<S> traits;
279  memory::buffer serialize_buffer;

291  void add_board(
292  endpoint& bus,
293  const identifier_t source_id,
294  const message_sequence_t sequence_no,
295  const basic_sudoku_board<S> board) noexcept {
296  if(boards.size() <= 8) [[likely]] {
297  searches.insert(source_id);
298  boards.emplace_back(source_id, sequence_no, std::move(board));
299  } else {
300  bus.log_warning("too many boards in backlog")
301  .arg(EAGINE_ID(rank), S)
302  .arg(EAGINE_ID(count), boards.size());
303  }
304  }
305 
306  auto update(endpoint& bus, const data_compressor& compressor) noexcept
307  -> work_done {
308  const unsigned_constant<S> rank{};
309  some_true something_done;
310 
311  if(boards.size() < 6) {
312  for(auto target_id : searches) {
313  message_view response{};
314  response.set_target_id(target_id);
315  bus.post(sudoku_alive_msg(rank), response);
316  something_done();
317  }
318  }
319  searches.clear();
320 
321  if(!boards.empty()) {
322  const auto target_id = std::get<0>(boards.back());
323  const auto sequence_no = std::get<1>(boards.back());
324  auto board = std::get<2>(boards.back());
325  boards.pop_back();
326 
327  const auto send_board =
328  [&](auto& candidate, const bool is_solved) {
329  serialize_buffer.ensure(
330  default_serialize_buffer_size_for(candidate));
331  const auto serialized{
332  (S >= 4)
333  ? default_serialize_packed(
334  candidate, cover(serialize_buffer), compressor)
335  : default_serialize(
336  candidate, cover(serialize_buffer))};
337  EAGINE_ASSERT(serialized);
338 
339  message_view response{extract(serialized)};
340  response.set_target_id(target_id);
341  response.set_sequence_no(sequence_no);
342  bus.post(sudoku_response_msg(rank, is_solved), response);
343  };
344 
345  bool done{false};
346  const auto process_candidate = [&](auto& candidate) {
347  if(candidate.is_solved()) {
348  send_board(candidate, true);
349  done = true;
350  } else {
351  if(!done) {
352  candidate.for_each_alternative(
353  candidate.find_unsolved(),
354  [&](auto& intermediate) {
355  if(intermediate.is_solved()) {
356  send_board(intermediate, true);
357  done = true;
358  } else {
359  if(!done) {
360  intermediate.for_each_alternative(
361  intermediate.find_unsolved(),
362  [&](auto& nested) {
363  if(nested.is_solved()) {
364  send_board(nested, true);
365  done = true;
366  } else if(!done) {
367  send_board(nested, false);
368  }
369  });
370  }
371  }
372  });
373  }
374  }
375  };
376 
377  board.for_each_alternative(
378  board.find_unsolved(), process_candidate);
379 
380  message_view response{};
381  response.set_target_id(target_id);
382  response.set_sequence_no(sequence_no);
383  bus.post(sudoku_done_msg(rank), response);
384  something_done();
385  }
386  return something_done;
387  }
388  };
