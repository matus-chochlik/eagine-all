20 /// @brief Helper class implementing identifier encoding functionality.
21 /// @ingroup identifiers
22 /// @see identifier
23 /// @note Do not use directly.
24 template <typename CharSet>
25 class identifier_encoding {
26 public:
27  /// @brief Encoded the specified character as N-bit byte.
28  static constexpr auto encode(const char c) noexcept -> std::uint8_t {
29  return _do_encode(c, 0, CharSet::values);
30  }
31 
32  /// @brief Encoded the specified N-bit byte as character.
33  static constexpr auto decode(const std::uint8_t i) noexcept -> char {
34  return _do_decode(i, CharSet::values);
35  }
36 
37  /// @brief Returns the invalid value from the encoding.
38  static constexpr auto invalid() noexcept {
39  return _get_invalid(CharSet::values);
40  }
41 
42  /// @brief Indicates if the encoded value is invalid.
43  static constexpr auto is_invalid(const std::uint8_t c) noexcept {
44  return c >= invalid();
45  }
46 
47  /// @brief Returns the valid character set as a string_view
48  static inline auto chars() -> string_view {
49  return {CharSet::values};
50  }
51 
52 private:
53  template <std::size_t L>
54  static constexpr auto _get_invalid(const char (&)[L]) noexcept {
55  return std::uint8_t(L);
56  }
57 
58  template <std::size_t L>
59  static constexpr auto _do_encode(
60  const char c,
61  const std::uint8_t i,
62  const char (&enc)[L]) noexcept -> std::uint8_t {
63  return ((i < L) && (c != '\0'))
64  ? ((c == enc[i]) ? i : _do_encode(c, i + 1, enc))
65  : invalid();
66  }
67 
68  template <std::size_t L>
69  static constexpr auto _do_decode(
70  const std::uint8_t i,
71  const char (&enc)[L]) noexcept -> char {
72  return (i < invalid()) ? enc[i] : '\0';
73  }
74 };

90 /// @brief Helper template for unpacking of identifier into a character string.
91 /// @ingroup identifiers
92 /// @see indentifier
93 /// @note Do not use directly. Use identifier.
94 template <std::size_t M>
95 class identifier_name {
96 public:
97  template <typename... C>
98  constexpr identifier_name(span_size_t len, C... c) noexcept
99  : _str{c...}
100  , _len{std::uint8_t(len)} {}
101 
102  /// @brief Alias for the string lenght type.
103  using size_type = span_size_t;
104 
105  /// @brief Alias for the string character type.
106  using value_type = char;
107 
108  /// @brief Alias for the iterator type.
109  using iterator = const char*;
110 
111  /// @brief Alias for the const iterator type.
112  using const_iterator = iterator;
113 
114  /// @brief Returns the pointer to the start of the unpacked identifier name.
115  constexpr auto data() const noexcept {
116  return _str.data();
117  }
118 
119  /// @brief Returns the length of the unpacked character string.
120  constexpr auto size() const noexcept {
121  return size_type(_len);
122  }
123 
124  /// @brief Returns an iterator to the start of the unpacked identifier name.
125  constexpr auto begin() const noexcept -> const_iterator {
126  return _str.data();
127  }
128 
129  /// @brief Returns an iterator past the end of the unpacked identifier name.
130  constexpr auto end() const noexcept -> const_iterator {
131  return _str.data() + size();
132  }
133 
134  /// @brief Returns a string view covering the unpacked identifier name.
135  constexpr auto view() const noexcept -> string_view {
136  // negative size indicates that the data is zero terminated
137  return {data(), -size()};
138  }
139 
140  /// @brief Returns the unpacked identifier name as a standard string.
141  auto str() const noexcept -> std::string {
142  return {_str.data(), _len};
143  }
144 
145  /// @brief Assigns the unpacked identifier name into a standard string.
146  /// @returns s
147  auto str(std::string& s) const noexcept -> std::string& {
148  s.assign(_str.data(), _len);
149  return s;
150  }
151 
152 private:
153  fixed_size_string<M> _str{};
154  std::uint8_t _len{0};
155 };

166 /// @brief Basic template for limited length, packed string identifiers.
167 /// @ingroup identifiers
168 /// @see identifier
169 /// @see long_identifier
170 /// @see EAGINE_ID
171 ///
172 /// Packed identifier store short constant strings with limited allowed set
173 /// of characters that are used as object identifiers throughout the project.
174 /// The strings are @c constexpr encoded into unsigned integer values and are
175 /// stored without dynamic memory allocation.
176 /// Comparison operations on identifiers are typically more efficient than
177 /// regular character string comparisons, but note that integer comparisons
178 /// are used instead lexicographical comparisons.
179 template <std::size_t M, std::size_t B, typename CharSet, typename UIntT>
180 class basic_identifier {
181  using _bites_t = biteset<M, B, std::uint8_t>;
182 
183 public:
184  static_assert(
185  (1U << B) > sizeof(CharSet::values),
186  "B-bits are not sufficient to represent CharSet");
187 
188  /// @brief Alias for the encoding type.
189  using encoding = identifier_encoding<CharSet>;
190 
191  /// @brief Alias for the length type.
192  using size_type = span_size_t;
193 
194  /// @brief Alias for the element type.
195  using value_type = char;
196 
197  /// @brief Alias for the unpacked identifier_name type.
198  using name_type = identifier_name<M>;
199 
200  /// @brief Default constructor. Constructs an empty identifier.
201  constexpr basic_identifier() noexcept = default;
202 
203  /// @brief Construction from a C-string literal.
204  template <std::size_t L>
205  explicit constexpr basic_identifier(const char (&init)[L]) noexcept
206  requires(L <= M + 1)
207  : _bites{_make_bites(
208  static_cast<const char*>(init),
209  L,
210  std::make_index_sequence<M>{})} {}
211 
212  /// @brief Construction from a const span of characters.
213  explicit constexpr basic_identifier(const span<const char> init) noexcept
214  : _bites{_make_bites(
215  init.data(),
216  init.size(),
217  std::make_index_sequence<M>{})} {}
218 
219  explicit constexpr basic_identifier(const UIntT init) noexcept
220  : _bites{_bites_t::from_value(init)} {}
221 
222  explicit constexpr basic_identifier(const _bites_t init) noexcept
223  : _bites{std::move(init)} {}
224 
225  /// @brief Returns the maximum length of this identifier type.
226  static constexpr auto max_size() noexcept -> size_type {
227  return M;
228  }
229 
230  /// @brief Returns the size of this identifier.
231  /// @see size()
232  constexpr auto is_empty() const noexcept -> bool {
233  return value() == 0;
234  }
235 
236  constexpr explicit operator bool() const noexcept {
237  return !is_empty();
238  }
239 
240  /// @brief Returns the size of this identifier.
241  /// @see is_empty()
242  constexpr auto size() const noexcept -> size_type {
243  return is_empty() ? 0 : _get_size(0);
244  }
245 
246  /// @brief Subscript operator. Allows to access individual characters.
247  constexpr auto operator[](const size_type idx) const noexcept
248  -> value_type {
249  return encoding::decode(_bites[idx]);
250  }
251 
252  constexpr auto value() const noexcept -> UIntT {
253  return _bites.bytes().template as<UIntT>();
254  }
255 
256  constexpr auto matches(const UIntT what) const noexcept {
257  return value() == what;
258  }
259 
260  constexpr auto matches(string_view what) const noexcept {
261  // TODO: this could be optimized: compare char-by-char and bail early
262  return name() == what;
263  }
264 
265  /// @brief Returns this identifier as unpacked identifier_name.
266  /// @see identifier_name
267  constexpr auto name() const noexcept -> name_type {
268  return _get_name(std::make_index_sequence<M>{});
269  }
270 
271  /// @brief Returns this identifier as unpacked standard string.
272  /// @see identifier_name
273  inline auto str() const -> std::string {
274  return name().str();
275  }
276 
277  /// @brief Equality comparison.
278  friend constexpr auto operator==(
279  const basic_identifier& a,
280  const basic_identifier& b) noexcept {
281  return a._bites == b._bites;
282  }
283 
284  /// @brief Nonequality comparison.
285  friend constexpr auto operator!=(
286  const basic_identifier& a,
287  const basic_identifier& b) noexcept {
288  return a._bites != b._bites;
289  }
290 
291  /// @brief Less-than comparison.
292  /// @note This is not lexicographical string comparison.
293  friend constexpr auto operator<(
294  const basic_identifier& a,
295  const basic_identifier& b) noexcept {
296  return a._bites < b._bites;
297  }
298 
299  /// @brief Less-equal comparison.
300  /// @note This is not lexicographical string comparison.
301  friend constexpr auto operator<=(
302  const basic_identifier& a,
303  const basic_identifier& b) noexcept {
304  return a._bites <= b._bites;
305  }
306 
307  /// @brief Greater-than comparison.
308  /// @note This is not lexicographical string comparison.
309  friend constexpr auto operator>(
310  const basic_identifier& a,
311  const basic_identifier& b) noexcept {
312  return a._bites > b._bites;
313  }
314 
315  /// @brief Greater-equal comparison.
316  /// @note This is not lexicographical string comparison.
317  friend constexpr auto operator>=(
318  const basic_identifier& a,
319  const basic_identifier& b) noexcept {
320  return a._bites >= b._bites;
321  }
322 
323 private:
324  _bites_t _bites;
325 
326  template <std::size_t... I>
327  static constexpr auto _make_bites(
328  const char* init,
329  std::size_t l,
330  std::index_sequence<I...>) noexcept {
331  return biteset<M, B, std::uint8_t>{
332  encoding::encode((I < l) ? init[I] : '\0')...};
333  }
334 
335  template <std::size_t... I>
336  constexpr auto _get_name(std::index_sequence<I...>) const noexcept
337  -> name_type {
338  return name_type{size(), (*this)[size_type(I)]...};
339  }
340 
341  constexpr auto _get_size(std::size_t s) const noexcept -> std::size_t {
342  return (s < M) ? encoding::is_invalid(_bites[size_type(s)])
343  ? s
344  : _get_size(s + 1)
345  : M;
346  }
347 };
