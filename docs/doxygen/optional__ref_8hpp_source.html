19 /// @brief Optional reference to an instance of type @p T.
20 /// @ingroup valid_if
21 /// @see valid_if
22 template <typename T>
23 class optional_reference_wrapper {
24 public:
25  /// @brief Construction from a reference to value of type @p T.
26  optional_reference_wrapper(T& ref) noexcept
27  : _ptr{std::addressof(ref)} {}
28 
29  /// @brief Move constructor.
30  optional_reference_wrapper(optional_reference_wrapper&&) noexcept = default;
31 
32  /// @brief Copy constructor.
33  optional_reference_wrapper(const optional_reference_wrapper&) = default;
34 
35  /// @brief Move assignment operator.
36  auto operator=(optional_reference_wrapper&&) noexcept
37  -> optional_reference_wrapper& = default;
38 
39  /// @brief Copy assignment operator.
40  auto operator=(const optional_reference_wrapper&)
41  -> optional_reference_wrapper& = default;
42 
43  ~optional_reference_wrapper() noexcept = default;
44 
45  /// @brief Construction from nothing.
46  /// @post !is_valid()
47  constexpr optional_reference_wrapper(nothing_t) noexcept {}
48 
49  /// @brief Construction from nullptr.
50  /// @post !is_valid()
51  constexpr optional_reference_wrapper(std::nullptr_t) noexcept {}
52 
53  /// @brief Indicates if this stores a valid reference.
54  auto is_valid() const noexcept -> bool {
55  return _ptr != nullptr;
56  }
57 
58  /// @brief Indicates if this stores a valid reference.
59  /// @see is_valid
60  explicit operator bool() const noexcept {
61  return is_valid();
62  }
63 
64  /// @brief Returns the stored reference.
65  /// @pre is_valid()
66  auto get() const noexcept -> T& {
67  EAGINE_ASSERT(is_valid());
68  return *_ptr;
69  }
70 
71  /// @brief Returns the stored value.
72  /// @pre is_valid()
73  auto value() const noexcept -> const T& {
74  EAGINE_ASSERT(is_valid());
75  return *_ptr;
76  }
77 
78  /// @brief Returns the stored value if valid or @p fallback otherwise.
79  /// @see is_valid
80  template <typename U>
81  auto value_or(U&& fallback) const noexcept -> T
82  requires(std::is_convertible_v<U, T>) {
83  if(is_valid()) {
84  return *_ptr;
85  }
86  return T(std::forward<U>(fallback));
87  }
88 
89  /// @brief Returns the stored reference.
90  /// @see get
91  explicit operator T&() const noexcept {
92  return get();
93  }
94 
95  /// @brief Tri-state equality comparison of the referred instance with a value.
96  friend auto operator==(const optional_reference_wrapper& l, const T& r)
97  -> tribool {
98  if(l.is_valid()) {
99  return l.value() == r;
100  }
101  return indeterminate;
102  }
103 
104  /// @brief Tri-state nonequality comparison of the referred instance with a value.
105  friend auto operator!=(const optional_reference_wrapper& l, const T& r)
106  -> tribool {
107  if(l.is_valid()) {
108  return l.value() != r;
109  }
110  return indeterminate;
111  }
112 
113 private:
114  T* _ptr{nullptr};
115 };
