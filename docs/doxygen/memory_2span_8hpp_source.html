88 /// @brief Non-owning view of a contiguous range of memory with ValueType elements.
89 /// @ingroup type_utils
90 /// @tparam ValueType the type of elements assumed in the covered memory range.
91 /// @tparam Pointer the pointer type used to point to the start of the span.
92 /// @tparam SizeType the integer type used to count the elements in the span.
93 /// @see basic_block
94 /// @see basic_split_span
95 ///
96 /// This template is similar to std::span but allows to specify other pointer
97 /// types besides ValueType*, for example basic_offset_ptr, etc. and to specify
98 /// the size type.
99 template <
100  typename ValueType,
101  typename Pointer = ValueType*,
102  typename SizeType = span_size_t>
103 class basic_span {
104 public:
105  /// @brief The element value type.
106  using value_type = ValueType;
107 
108  /// @brief The element count type.
109  using size_type = SizeType;
110 
111  /// @brief The memory address type.
112  using address_type = basic_address<std::is_const_v<ValueType>>;
113 
114  /// @brief The pointer type.
115  using pointer = Pointer;
116 
117  /// @brief The iterator type.
118  using iterator = Pointer;
119 
120  /// @brief The const iterator type.
121  using const_iterator = Pointer;
122 
123  /// @brief The reverse iterator type
124  using reverse_iterator = std::reverse_iterator<iterator>;
125 
126  /// @brief Construction from pointer and length.
127  constexpr basic_span(
128  pointer addr, // NOLINT(modernize-pass-by-value)
129  const size_type len) noexcept
130  : _size{len}
131  , _addr{addr} {}
132 
133  /// @brief Construction from memory address and length.
134  constexpr basic_span(const address_type addr, const size_type len) noexcept
135  : basic_span{static_cast<pointer>(addr), len} {}
136 
137  /// @brief Construction from a pair of pointers.
138  constexpr basic_span(pointer b, pointer e) noexcept
139  : basic_span{b, b <= e ? e - b : 0} {}
140 
141  /// @brief Construction from a pair of memory addresses.
142  constexpr basic_span(const address_type ba, const address_type be) noexcept
143  : basic_span{
144  static_cast<pointer>(ba),
145  limit_cast<size_type>(
146  (ba <= be) ? (be - ba) / sizeof(value_type) : 0)} {}

148  template <typename T, typename P, typename S>
149  using enable_if_convertible = std::enable_if_t<
150  std::is_convertible_v<T, ValueType> &&
151  std::is_convertible_v<P, Pointer> && std::is_convertible_v<S, SizeType>>;
152 
153  template <typename T, typename P, typename S>
154  using enable_if_different = std::enable_if_t<
155  !std::is_same_v<T, ValueType> || !std::is_same_v<P, Pointer> ||
156  !std::is_same_v<S, SizeType>>;
157 
158  /// @brief Default constructor. Constructs an empty span.
159  /// @post is_empty()
160  /// @post size() == 0
161  constexpr basic_span() noexcept = default;
162

164  constexpr basic_span(const basic_span&) noexcept = default;
165 
166  /// @brief Move constructor.
167  constexpr basic_span(basic_span&&) noexcept = default;

173  auto operator=(basic_span&&) noexcept -> basic_span& = default;

176  /// @brief Converting copy constructor from span of compatible elements.
177  template <

179  typename P,
180  typename S,
181  typename = enable_if_convertible<T, P, S>,
182  typename = enable_if_different<T, P, S>>
183  constexpr basic_span(basic_span<T, P, S> that) noexcept
184  : basic_span{
185  static_cast<pointer>(that.data()),
186  limit_cast<size_type>(that.size())} {}

189  template <

194  typename = enable_if_different<T, P, S>>

248  /// @brief Returns the number of elements in the span.
249  /// @see is_empty
250  constexpr auto size() const noexcept -> size_type {
251  if constexpr(std::is_same_v<std::remove_const_t<ValueType>, char>) {
252  if(_size < 0) [[likely]] {

325  /// @brief Indicates if this span encloses another span.
326  /// @see encloses
327  /// @see overlaps
328  template <typename Ts, typename Ps, typename Ss>
329  auto contains(basic_span<Ts, Ps, Ss> that) const noexcept -> bool {
330  return (addr() <= that.addr()) && (that.end_addr() <= end_addr());
331  }

333  /// @brief Indicates if this span overlaps with another span.
334  /// @see encloses
335  /// @see contains
336  template <typename Ts, typename Ps, typename Ss>
337  auto overlaps(const basic_span<Ts, Ps, Ss>& that) const noexcept -> bool {
338  return encloses(that.addr()) || encloses(that.end_addr()) ||
339  that.encloses(addr()) || that.encloses(end_addr());

346  return EAGINE_CONSTEXPR_ASSERT(index < size(), _addr[index]);

373  return EAGINE_CONSTEXPR_ASSERT(0 < size(), _addr[size() - 1]);

415 private:
416  // if _size is negative it indicates that the span is terminated
417  // by a zero value. if such case the span length is calculated by
418  // getting the absolute value of _size.
419  size_type _size{0};
420  pointer _addr{nullptr};
421 };

659  const memory::basic_span<Tr, Pr, Sr> r) noexcept -> bool {
660  if(are_equal(l.size(), r.size())) {
661  if constexpr(
662  std::is_same_v<std::remove_const_t<Tl>, std::remove_const_t<Tr>> &&
663  std::is_integral_v<std::remove_const_t<Tl>>) {
664  return 0 ==
665  std::memcmp(
666  l.data(), r.data(), sizeof(Tl) * std_size(l.size()));
667  } else {
668  for(const auto i : integer_range(span_size(l.size()))) {
669  if(!are_equal(l[i], r[i])) {
670  return false;
671  }
672  }
673  return true;
674  }
675  }
676  return false;
677  }
678 };
