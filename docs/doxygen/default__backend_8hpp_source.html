22 struct default_progress_info {
23  std::string title;
24  span_size_t total_steps{0};
25  span_size_t prev_steps{0};
26  span_size_t curr_steps{0};
27  span_size_t increment{0};
28  activity_progress_id_t parent_id{0};
29  log_event_severity log_level{log_event_severity::info};
30 };
31 //------------------------------------------------------------------------------
32 /// @brief Default implementation of activity progress backend implementations.
33 /// @ingroup progress
34 class default_progress_tracker_backend : public progress_tracker_backend {
35 public:
36  default_progress_tracker_backend(main_ctx_getters& ctx)
37  : _log{EAGINE_ID(Progress), ctx.log()} {}
38 
39  auto register_observer(progress_observer& observer) noexcept -> bool final {
40  if(!_observer) {
41  _observer = &observer;
42  return true;
43  }
44  return false;
45  }
46 
47  void unregister_observer(progress_observer& observer) noexcept final {
48  if(_observer == &observer) {
49  _observer = nullptr;
50  }
51  }
52 
53  auto begin_activity(
54  const activity_progress_id_t parent_id,
55  const string_view title,
56  span_size_t total_steps) -> activity_progress_id_t final {
57  auto id_and_info =
58  [&]() -> std::tuple<activity_progress_id_t, default_progress_info&> {
59  const std::lock_guard<std::mutex> lock{_mutex};
60  while(true) {
61  if(++_id_sequence != 0) {
62  if(_activities.find(_id_sequence) == _activities.end()) {
63  break;
64  }
65  }
66  }
67  auto& result = _activities[_id_sequence];
68  const auto activity_id = _encode(result);
69  _index[activity_id] = _id_sequence;
70  if(auto observer{_observer.load()}) {
71  extract(observer).activity_begun(
72  parent_id, activity_id, title, total_steps);
73  }
74  return {activity_id, result};
75  }();
76  const auto activity_id = std::get<0>(id_and_info);
77  auto& info = std::get<1>(id_and_info);
78  info.title = to_string(title);
79  info.total_steps = total_steps;
80  info.increment = span_size_t(float(total_steps) * 0.001F);
81  info.parent_id = parent_id;
82  return activity_id;
83  }
84 
85  auto update_progress(
86  const activity_progress_id_t activity_id,
87  span_size_t current) noexcept -> bool final {
88  _call_back();
89  if(activity_id) {
90  _update(activity_id, _decode(activity_id), current);
91  }
92  return _keep_going;
93  }
94 
95  auto advance_progress(
96  const activity_progress_id_t activity_id,
97  span_size_t increment) noexcept -> bool final {
98  _call_back();
99  if(activity_id) {
100  auto& info = _decode(activity_id);
101  _update(activity_id, info, info.curr_steps + increment);
102  }
103  return _keep_going;
104  }
105 
106  void finish_activity(
107  const activity_progress_id_t activity_id) noexcept final {
108  if(activity_id) {
109  _do_log(_decode(activity_id));
110  const std::lock_guard<std::mutex> lock{_mutex};
111  const auto ipos = _index.find(activity_id);
112  if(ipos != _index.end()) {
113  if(auto observer{_observer.load()}) {
114  const auto apos = _activities.find(ipos->second);
115  if(apos != _activities.end()) {
116  auto& activity = apos->second;
117  extract(observer).activity_finished(
118  activity.parent_id,
119  activity_id,
120  activity.title,
121  activity.total_steps);
122  }
123  }
124  _activities.erase(ipos->second);
125  _index.erase(ipos);
126  }
127  }
128  }
129 
130  void set_update_callback(
131  const callable_ref<bool() noexcept> callback,
132  const std::chrono::milliseconds min_interval) final {
133  _callback = callback;
134  _min_interval = min_interval;
135  }
136 
137  void reset_update_callback() noexcept final {
138  _callback = {};
139  }
140 
141 private:
142  static auto _encode(const default_progress_info& info) noexcept
143  -> activity_progress_id_t {
144  return reinterpret_cast<activity_progress_id_t>(&info);
145  }
146 
147  static auto _decode(const activity_progress_id_t activity_id) noexcept
148  -> default_progress_info& {
149  /// NOLINTNEXTLINE(performance-no-int-to-ptr)
150  return *reinterpret_cast<default_progress_info*>(activity_id);
151  }
152 
153  void _call_back() noexcept {
154  const auto now = _clock.now();
155  if(_last_call < now) {
156  _last_call = now + _min_interval;
157  if(_callback) {
158  _keep_going = _callback() && _keep_going;
159  }
160  }
161  }
162 
163  void _update(
164  const activity_progress_id_t activity_id,
165  default_progress_info& info,
166  span_size_t current) {
167  info.curr_steps = current;
168  if(info.curr_steps - info.prev_steps >= info.increment) {
169  info.prev_steps = info.curr_steps;
170  _do_log(info);
171  }
172  if(auto observer{_observer.load()}) {
173  extract(observer).activity_updated(
174  info.parent_id, activity_id, info.curr_steps, info.total_steps);
175  }
176  }
177 
178  void _do_log(const default_progress_info& info) const {
179  _log.log(info.log_level, info.title)
180  .arg(EAGINE_ID(current), info.curr_steps)
181  .arg(EAGINE_ID(total), info.total_steps)
182  .arg(
183  EAGINE_ID(progress),
184  EAGINE_ID(Progress),
185  0.F,
186  float(info.curr_steps),
187  float(info.total_steps));
188  }
189 
190  std::mutex _mutex;
191  logger _log;
192  callable_ref<bool() noexcept> _callback{};
193  std::atomic<progress_observer*> _observer{nullptr};
194  std::chrono::milliseconds _min_interval{500};
195  std::chrono::steady_clock::time_point _last_call{};
196  std::chrono::steady_clock _clock;
197  activity_progress_id_t _id_sequence{0};
198  flat_map<activity_progress_id_t, activity_progress_id_t> _index;
199  std::map<activity_progress_id_t, default_progress_info> _activities;
200  std::atomic<bool> _keep_going{true};
201 };

203 } // namespace eagine
