21 /// @brief Service responding to pings from the pinger counterpart.
22 /// @ingroup msgbus
23 /// @see service_composition
24 /// @see pinger
25 template <typename Base = subscriber>
26 class pingable : public Base {
27  using This = pingable;
28 
29 public:
30  /// @brief Decides if a ping request should be responded.
31  virtual auto respond_to_ping(
32  [[maybe_unused]] const identifier_t pinger_id,
33  const message_sequence_t,
34  const verification_bits) noexcept -> bool {
35  return true;
36  }
37 
38 protected:
39  using Base::Base;
40 
41  void add_methods() noexcept {
42  Base::add_methods();
43  Base::add_method(
44  this, EAGINE_MSG_MAP(eagiMsgBus, ping, This, _handle_ping));
45  }
46 
47 private:
48  auto _handle_ping(
49  const message_context&,
50  const stored_message& message) noexcept -> bool {
51  if(respond_to_ping(
52  message.source_id,
53  message.sequence_no,
54  this->verify_bits(message))) {
55  this->bus_node().respond_to(message, EAGINE_MSGBUS_ID(pong), {});
56  }
57  return true;
58  }

61 /// @brief Service sending to pings from the pingable counterparts.
62 /// @ingroup msgbus
63 /// @see service_composition
64 /// @see pingable
65 template <typename Base = subscriber>
66 class pinger
67  : public Base
68  , protected std::chrono::steady_clock {
69 
70  using This = pinger;
71 
72  std::vector<std::tuple<identifier_t, message_sequence_t, timeout>>
73  _pending{};
74 
75 public:
76  /// @brief Returns the ping message type id.
77  static constexpr auto ping_msg_id() noexcept {
78  return EAGINE_MSGBUS_ID(ping);
79  }
80 
81  /// @brief Broadcasts a query searching for pingable message bus nodes.
82  void query_pingables() noexcept {
83  this->bus_node().query_subscribers_of(ping_msg_id());
84  }
85 
86  /// @brief Sends a pings request and tracks it for the specified maximum time.
87  /// @see ping_responded
88  /// @see ping_timeouted
89  /// @see has_pending_pings
90  void ping(
91  const identifier_t pingable_id,
92  const std::chrono::milliseconds max_time) noexcept {
93  message_view message{};
94  auto msg_id{EAGINE_MSGBUS_ID(ping)};
95  message.set_target_id(pingable_id);
96  message.set_priority(message_priority::low);
97  this->bus_node().set_next_sequence_id(msg_id, message);
98  this->bus_node().post(msg_id, message);
99  _pending.emplace_back(message.target_id, message.sequence_no, max_time);
100  }
101 
102  /// @brief Sends a pings request and tracks it for a default time period.
103  /// @see ping_responded
104  /// @see ping_timeouted
105  /// @see has_pending_pings
106  void ping(const identifier_t pingable_id) noexcept {
107  ping(
108  pingable_id,
109  adjusted_duration(
110  std::chrono::milliseconds{5000}, memory_access_rate::low));
111  }
112 
113  auto update() noexcept -> work_done {
114  some_true something_done{};
115  something_done(Base::update());
116 
117  something_done(
118  std::erase_if(_pending, [this](auto& entry) {
119  auto& [pingable_id, sequence_no, ping_time] = entry;
120  if(ping_time.is_expired()) {
121  ping_timeouted(
122  pingable_id,
123  sequence_no,
124  std::chrono::duration_cast<std::chrono::microseconds>(
125  ping_time.elapsed_time()));
126  return true;
127  }
128  return false;
129  }) > 0);
130  return something_done;
131  }
132 
133  /// @brief Indicates if there are yet unresponded pending ping requests.
134  /// @see ping_responded
135  /// @see ping_timeouted
136  auto has_pending_pings() const noexcept -> bool {
137  return !_pending.empty();
138  }
139 
140  /// @brief Triggered on receipt of ping response.
141  /// @see ping
142  /// @see ping_timeouted
143  /// @see has_pending_pings
144  signal<void(
145  const identifier_t pingable_id,
146  const message_sequence_t sequence_no,
147  const std::chrono::microseconds age,
148  const verification_bits) noexcept>
149  ping_responded;
150 
151  /// @brief Triggered on timeout of ping response.
152  /// @see ping
153  /// @see ping_responded
154  /// @see has_pending_pings
155  signal<void(
156  const identifier_t pingable_id,
157  const message_sequence_t sequence_no,
158  const std::chrono::microseconds age) noexcept>
159  ping_timeouted;
160 
161 protected:
162  using Base::Base;
163 
164  void add_methods() noexcept {
165  Base::add_methods();
166  Base::add_method(
167  this, EAGINE_MSG_MAP(eagiMsgBus, pong, This, _handle_pong));
168  }
169 
170 private:
171  auto _handle_pong(
172  const message_context&,
173  const stored_message& message) noexcept -> bool {
174  std::erase_if(_pending, [this, &message](auto& entry) {
175  auto& [pingable_id, sequence_no, ping_time] = entry;
176  const bool is_response = (message.source_id == pingable_id) &&
177  (message.sequence_no == sequence_no);
178  if(is_response) {
179  ping_responded(
180  message.source_id,
181  message.sequence_no,
182  std::chrono::duration_cast<std::chrono::microseconds>(
183  ping_time.elapsed_time()),
184  this->verify_bits(message));
185  return true;
186  }
187  return false;
188  });
189  return true;
190  }
191 };
