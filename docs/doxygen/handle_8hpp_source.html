19 /// @brief Non-owning wrapper for C-API opaque handle types.
20 /// @ingroup c_api_wrap
21 /// @tparam Tag type distinguisihing various handler with the same underlying type.
22 /// @tparam Handle the underlying handle type.
23 /// @tparam invalid the invalid value for this particular handle type.
24 /// @see basic_owned_handle
25 template <typename Tag, typename Handle, Handle invalid = ~Handle(0)>
26 class basic_handle {
27 public:
28  /// @brief Alias for the tag type.
29  using tag_type = Tag;
30 
31  /// @brief Alias for the underlying handle type.
32  using handle_type = Handle;
33 
34  ~basic_handle() noexcept = default;
35 
36  /// @brief Default constructor.
37  /// @post !is_valid()
38  constexpr basic_handle() noexcept = default;
39 
40  /// @brief Move constructor.
41  constexpr basic_handle(basic_handle&& tmp) noexcept
42  : _name{tmp._release()} {}
43 
44  /// @brief Move assignment operator.
45  auto operator=(basic_handle&& tmp) noexcept -> auto& {
46  _name = tmp._release();
47  return *this;
48  }
49 
50  /// @brief Copy constructor.
51  constexpr basic_handle(const basic_handle&) noexcept = default;
52 
53  /// @brief Copy assignment operator.
54  auto operator=(const basic_handle&) noexcept -> basic_handle& = default;
55 
56  /// @brief Construction from the underlying handle type.
57  explicit constexpr basic_handle(Handle name) noexcept
58  : _name{name} {}
59 
60  /// @brief Indicates if this instance contains a valid handle.
61  constexpr auto is_valid() const noexcept -> bool {
62  return _name != invalid;
63  }
64 
65  /// @brief Indicates if this instance contains a valid handle.
66  /// @see is_valid
67  explicit constexpr operator bool() const noexcept {
68  return is_valid();
69  }
70 
71  /// @brief Explicit conversion to the underlying handle type.
72  explicit constexpr operator Handle() const noexcept {
73  return _name;
74  }
75 
76  /// @brief Explicit conversion to a pointer to the underlying handle type.
77  explicit constexpr operator Handle*() noexcept {
78  return &_name;
79  }

81  struct transform {
82  constexpr auto operator()(Handle hndl) const noexcept {
83  return basic_handle<Tag, Handle, invalid>{hndl};
84  }
85  };
86 
87 protected:
88  auto _release() noexcept {
89  return std::exchange(_name, invalid);
90  }
91 
92 private:
93  Handle _name{invalid};
94 };

96 /// @brief Owning wrapper for C-API opaque handle types.
97 /// @ingroup c_api_wrap
98 /// @tparam Tag type distinguisihing various handler with the same underlying type.
99 /// @tparam Handle the underlying handle type.
100 /// @tparam invalid the invalid value for this particular handle type.
101 /// @see basic_handle
102 template <typename Tag, typename Handle, Handle invalid = ~Handle(0)>
103 class basic_owned_handle : public basic_handle<Tag, Handle, invalid> {
104  using base = basic_handle<Tag, Handle, invalid>;
105 
106 public:
107  ~basic_owned_handle() noexcept = default;
108 
109  /// @brief Default constructor.
110  /// @post !is_valid()
111  constexpr basic_owned_handle() noexcept = default;
112 
113  /// @brief Move constructor.
114  constexpr basic_owned_handle(basic_owned_handle&& tmp) noexcept
115  : base{tmp.release()} {}
116 
117  /// @brief Move assignment operator.
118  auto operator=(basic_owned_handle&& tmp) noexcept -> auto& {
119  *static_cast<base*>(this) = static_cast<base&&>(tmp);
120  return *this;
121  }
122 
123  /// @brief Copy constructor.
124  basic_owned_handle(const basic_owned_handle&) = delete;
125 
126  /// @brief Copy assignment operator.
127  auto operator=(const basic_owned_handle&) = delete;
128 
129  /// @brief Constructor adopting a non-owning handle wrapper.
130  explicit constexpr basic_owned_handle(base adopted) noexcept
131  : base{adopted} {}
132 
133  /// @brief Constructor adopting an underlying handle value.
134  explicit constexpr basic_owned_handle(Handle name) noexcept
135  : base{name} {}
136 
137  /// @brief Releases the underlying handle value.
138  /// @post !is_valid()
139  auto release() noexcept -> Handle {
140  return this->_release();
141  }
142 };
