22 /// @brief Shape vertex attribute kind enumeration.
23 /// @ingroup shapes
24 /// @see vertex_attrib_name_and_kind
25 enum class vertex_attrib_kind : std::uint16_t {
26  /// @brief The object id attributes (typically unique integer).
27  object_id = 1U << 0U,
28  /// @brief Vertex position.
29  position = 1U << 1U,
30  /// @brief Vertex normal vector.
31  normal = 1U << 2U,
32  /// @brief Vertex tangential vector.
33  tangent = 1U << 3U,
34  /// @brief Vertex bi-tangential vector.
35  bitangent = 1U << 4U,
36  /// @brief Shape pivot point.
37  pivot = 1U << 5U,
38  /// @brief Pivot of vertex pivot point.
39  pivot_pivot = 1U << 6U,
40  /// @brief Vertex pivot point.
41  vertex_pivot = 1U << 7U,
42  /// @brief Normalized coordinate within shape bounding box.
43  box_coord = 1U << 8U,
44  /// @brief Generic face coordinate.
45  face_coord = 1U << 9U,
46  /// @brief UV-texture wrapping coordinate.
47  wrap_coord = 1U << 10U,
48  /// @brief Vertex color value.
49  color = 1U << 11U,
50  /// @brief Generic vertex weight value.
51  weight = 1U << 12U,
52  /// @brief Vertex (ambient) light occlusion value.
53  occlusion = 1U << 13U,
54  /// @brief Face polygon id value (multiple faces can belong to the same polygon)
55  polygon_id = 1U << 14U,
56  /// @brief Face material id value.
57  material_id = 1U << 15U
58  // also fix all_vertex_attrib_kinds
59 };

108 //------------------------------------------------------------------------------
109 /// @brief Class designating an vertex attribute variant in a shape generator.
110 /// @ingroup shapes
111 /// @see vertex_attrib_variants
112 class vertex_attrib_variant {
113 public:
114  /// @brief Default constructor.
115  constexpr vertex_attrib_variant() noexcept = default;
116 
117  /// @brief Construction from vertex attribute kind enumerator.
118  constexpr vertex_attrib_variant(const vertex_attrib_kind a) noexcept
119  : _attrib{a} {}
120 
121  /// @brief Construction from vertex attribute kind enumerator and index.
122  constexpr vertex_attrib_variant(
123  const vertex_attrib_kind a,
124  const span_size_t v) noexcept
125  : _attrib{a}
126  , _index{std::int16_t(v)} {}
127 
128  /// @brief Construction from vertex attribute kind and another attribute variant.
129  constexpr vertex_attrib_variant(
130  const vertex_attrib_kind a,
131  const vertex_attrib_variant vav) noexcept
132  : _attrib{a}
133  , _index{vav._index} {}
134 
135  /// @brief Return the attribute kind.
136  constexpr auto attribute() const noexcept -> vertex_attrib_kind {
137  return _attrib;
138  }
139 
140  /// @brief Check if the stored index is valid.
141  constexpr auto has_valid_index() const noexcept {
142  return _index >= 0;
143  }
144 
145  /// @brief Check if the stored index is valid.
146  /// @see has_valid_index
147  constexpr explicit operator bool() const noexcept {
148  return has_valid_index();
149  }
150 
151  /// @brief Returns the index of the attribute variant.
152  constexpr auto index() const noexcept -> span_size_t {
153  return _index;
154  }
155 
156  constexpr operator std::array<const vertex_attrib_variant, 1>()
157  const noexcept {
158  return {{*this}};
159  }
160 
161  constexpr auto as_tuple() const noexcept {
162  return std::make_tuple(_attrib, _index);
163  }
164 
165  /// @brief Equality comparison.
166  friend constexpr auto operator==(
167  const vertex_attrib_variant l,
168  const vertex_attrib_variant r) noexcept {
169  return l.as_tuple() == r.as_tuple();
170  }
171 
172  /// @brief Nonequality comparison.
173  friend constexpr auto operator!=(
174  const vertex_attrib_variant l,
175  const vertex_attrib_variant r) noexcept {
176  return l.as_tuple() != r.as_tuple();
177  }
178 
179  /// @brief Less-than comparison.
180  friend constexpr auto operator<(
181  const vertex_attrib_variant l,
182  const vertex_attrib_variant r) noexcept {
183  return l.as_tuple() < r.as_tuple();
184  }
185 
186  /// @brief Variant and kind equality comparison.
187  friend constexpr auto operator==(
188  const vertex_attrib_variant l,
189  const vertex_attrib_kind r) noexcept {
190  return l._attrib == r;
191  }
192 
193  /// @brief Variant and kind nonequality comparison.
194  friend constexpr auto operator!=(
195  const vertex_attrib_variant l,
196  const vertex_attrib_kind r) noexcept {
197  return l._attrib != r;
198  }
199 
200 private:
201  vertex_attrib_kind _attrib{};
202  std::int16_t _index{0};
203 };

257 //------------------------------------------------------------------------------
258 /// @brief Extracts vertex attribute kind bits from and vertex attribute variants array.
259 /// @ingroup shapes
260 /// @see vertex_attrib_kinds
261 template <std::size_t N>
262 static inline auto get_attrib_kinds(
263  const vertex_attrib_variants<N>& vaals) noexcept -> vertex_attrib_kinds {
264  vertex_attrib_kinds res;
265 
266  for(const vertex_attrib_variant& vaal : vaals) {
267  res = res | vaal.attribute();
268  }
269 
270  return res;
271 }
272 //------------------------------------------------------------------------------
273 /// @brief Gets the default number of values per vertex for an attribute kind.
274 /// @ingroup shapes
275 static inline auto attrib_values_per_vertex(
276  const vertex_attrib_kind attr) noexcept -> span_size_t {
277  switch(attr) {
278  case vertex_attrib_kind::color:
279  return 4;
280  case vertex_attrib_kind::position:
281  case vertex_attrib_kind::normal:
282  case vertex_attrib_kind::tangent:
283  case vertex_attrib_kind::bitangent:
284  case vertex_attrib_kind::pivot:
285  case vertex_attrib_kind::pivot_pivot:
286  case vertex_attrib_kind::vertex_pivot:
287  case vertex_attrib_kind::face_coord:
288  case vertex_attrib_kind::box_coord:
289  return 3;
290  case vertex_attrib_kind::wrap_coord:
291  return 2;
292  case vertex_attrib_kind::weight:
293  case vertex_attrib_kind::occlusion:
294  case vertex_attrib_kind::object_id:
295  case vertex_attrib_kind::polygon_id:
296  case vertex_attrib_kind::material_id:
297  return 1;
298  }
299  return 0;
300 }
301 //------------------------------------------------------------------------------
302 /// @brief Gets the default number of values per vertex for an attribute variant.
303 /// @ingroup shapes
304 static inline auto attrib_values_per_vertex(
305  const vertex_attrib_variant vav) noexcept {
306  return attrib_values_per_vertex(vav.attribute());
307 }
