20 /// @brief Class storing two identifier values representing class/method pair.
21 /// @ingroup identifiers
22 /// @see static_message_id
23 /// @see identifier_t
24 /// @see identifier
25 struct message_id : std::tuple<identifier_t, identifier_t> {
26  using base = std::tuple<identifier_t, identifier_t>;
27 
28  /// @brief Default constructor.
29  /// @post !is_valid()
30  constexpr message_id() noexcept = default;
31 
32  /// @brief Construction from two identifier values.
33  constexpr message_id(const identifier_t c, const identifier_t m) noexcept
34  : base{c, m} {}
35 
36  /// @brief Construction from two identifier objects.
37  constexpr message_id(const identifier c, const identifier m) noexcept
38  : base{c.value(), m.value()} {}
39 
40  /// @brief Construction from a tuple of two identifier objects.
41  constexpr message_id(const std::tuple<identifier, identifier> t) noexcept
42  : message_id{std::get<0>(t), std::get<1>(t)} {}
43 
44  /// @brief Construction from static_message_id value.
45  template <identifier_t ClassId, identifier_t MethodId>
46  constexpr message_id(const static_message_id<ClassId, MethodId>&) noexcept
47  : base{ClassId, MethodId} {}
48 
49  /// @brief Returns the class identifier value.
50  constexpr auto class_id() const noexcept -> identifier_t {
51  return std::get<0>(*this);
52  }
53 
54  /// @brief Returns the class identifier.
55  constexpr auto class_() const noexcept -> identifier {
56  return identifier{class_id()};
57  }
58 
59  /// @brief Returns the method identifier value.
60  constexpr auto method_id() const noexcept -> identifier_t {
61  return std::get<1>(*this);
62  }
63 
64  /// @brief Returns the method identifier.
65  constexpr auto method() const noexcept -> identifier {
66  return identifier{method_id()};
67  }
68 
69  /// @brief Checks if the stored identifier values are non-zero.
70  constexpr auto is_valid() const noexcept {
71  return (class_id() != 0U) && (method_id() != 0U);
72  }
73 
74  /// @brief Returns the class and method identifiers in a tuple.
75  /// See class_
76  /// See method
77  constexpr auto id_tuple() const noexcept
78  -> std::tuple<identifier, identifier> {
79  return {class_(), method()};
80  }
81 
82  /// @brief Checks if the class identifier matches the argument.
83  /// @see has_method
84  constexpr auto has_class(const identifier id) const noexcept {
85  return class_id() == id.value();
86  }
87 
88  /// @brief Checks if the method identifier matches the argument.
89  /// @see has_class
90  constexpr auto has_method(const identifier id) const noexcept {
91  return method_id() == id.value();
92  }
93 };
94 //------------------------------------------------------------------------------
95 /// @brief Template with two identifier parameters representing class/method pair.
96 /// @ingroup identifiers
97 /// @see message_id
98 ///
99 /// This class encodes an identifier pair in its template parameters.
100 /// It is implicitly convertible to message_id.
101 template <identifier_t ClassId, identifier_t MethodId>
102 struct static_message_id {
103  using type = static_message_id;
104 
105  /// @brief Returns the class identifier value.
106  static constexpr auto class_id() noexcept -> identifier_t {
107  return ClassId;
108  }
109 
110  /// @brief Returns the class identifier.
111  static constexpr auto class_() noexcept -> identifier {
112  return identifier{class_id()};
113  }
114 
115  /// @brief Returns the method identifier value.
116  static constexpr auto method_id() noexcept -> identifier_t {
117  return MethodId;
118  }
119 
120  /// @brief Returns the method identifier.
121  static constexpr auto method() noexcept -> identifier {
122  return identifier{method_id()};
123  }
124 };
125 //------------------------------------------------------------------------------
126 /// @brief Equality comparison between message_id and static_message_id.
127 /// @ingroup identifiers
128 template <identifier_t ClassId, identifier_t MethodId>
129 inline auto operator==(
130  const message_id l,
131  const static_message_id<ClassId, MethodId> r) noexcept {
132  return l == message_id{r};
133 }
