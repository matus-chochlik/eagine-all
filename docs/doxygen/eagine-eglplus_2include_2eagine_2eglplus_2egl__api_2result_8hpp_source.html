43 //------------------------------------------------------------------------------
44 /// @brief Class storing information about an EGL function call result.
45 /// @ingroup egl_api_wrap
46 /// @see egl_result
47 /// @see egl_no_result_info
48 class egl_result_info {
49 public:
50  /// @brief Indicates if the call finished without error.
51  explicit constexpr operator bool() const noexcept {
52  return egl_types::error_code_no_error(_error_code);
53  }
54 
55  constexpr auto error_code(const egl_types::int_type ec) noexcept -> auto& {
56  _error_code = ec;
57  return *this;
58  }
59 
60  /// @brief Returns a message associated with the result.
61  auto message() const noexcept -> string_view {
62 #ifdef EGL_BAD_SURFACE
63  if(_error_code == EGL_BAD_SURFACE) {
64  return {"bad surface"};
65  }
66 #endif
67 #ifdef EGL_BAD_CURRENT_SURFACE
68  if(_error_code == EGL_BAD_CURRENT_SURFACE) {
69  return {"bad current surface"};
70  }
71 #endif
72 #ifdef EGL_BAD_DISPLAY
73  if(_error_code == EGL_BAD_DISPLAY) {
74  return {"bad display"};
75  }
76 #endif
77 #ifdef EGL_CONTEXT_LOST
78  if(_error_code == EGL_CONTEXT_LOST) {
79  return {"EGL context lost"};
80  }
81 #endif
82 #ifdef EGL_BAD_PARAMETER
83  if(_error_code == EGL_BAD_PARAMETER) {
84  return {"invalid parameter"};
85  }
86 #endif
87 #ifdef EGL_BAD_ATTRIBUTE
88  if(_error_code == EGL_BAD_ATTRIBUTE) {
89  return {"unrecognized attribute or attribute value"};
90  }
91 #endif
92 #ifdef EGL_BAD_CONFIG
93  if(_error_code == EGL_BAD_CONFIG) {
94  return {"invalid configuration"};
95  }
96 #endif
97 #ifdef EGL_BAD_CONTEXT
98  if(_error_code == EGL_BAD_CONTEXT) {
99  return {"invalid context"};
100  }
101 #endif
102 #ifdef EGL_BAD_NATIVE_PIXMAP
103  if(_error_code == EGL_BAD_NATIVE_PIXMAP) {
104  return {"bad native pixmap"};
105  }
106 #endif
107 #ifdef EGL_BAD_NATIVE_WINDOW
108  if(_error_code == EGL_BAD_NATIVE_WINDOW) {
109  return {"bad native window"};
110  }
111 #endif
112 #ifdef EGL_BAD_MATCH
113  if(_error_code == EGL_BAD_MATCH) {
114  return {"inconsistent attributes"};
115  }
116 #endif
117 #ifdef EGL_BAD_ACCESS
118  if(_error_code == EGL_BAD_ACCESS) {
119  return {"cannot access requested resource"};
120  }
121 #endif
122 #ifdef EGL_NOT_INITIALIZED
123  if(_error_code == EGL_NOT_INITIALIZED) {
124  return {"EGL not initialized"};
125  }
126 #endif
127 #ifdef EGL_BAD_STREAM_KHR
128  if(_error_code == EGL_BAD_STREAM_KHR) {
129  return {"bad EGL stream"};
130  }
131 #endif
132 #ifdef EGL_BAD_OUTPUT_LAYER_EXT
133  if(_error_code == EGL_BAD_OUTPUT_LAYER_EXT) {
134  return {"bad EGL output layer"};
135  }
136 #endif
137 #ifdef EGL_BAD_OUTPUT_PORT_EXT
138  if(_error_code == EGL_BAD_OUTPUT_PORT_EXT) {
139  return {"bad EGL output port"};
140  }
141 #endif
142 #ifdef EGL_BAD_STATE_KHR
143  if(_error_code == EGL_BAD_STATE_KHR) {
144  return {"bad EGL state"};
145  }
146 #endif
147 #ifdef EGL_SUCCESS
148  if(_error_code == EGL_SUCCESS) {
149  return {"no error"};
150  }
151 #endif
152 #ifdef EGL_BAD_ALLOC
153  if(_error_code == EGL_BAD_ALLOC) {
154  return {"out of memory"};
155  }
156 #endif
157  return {"unknown error"};
158  }
159 
160  constexpr auto set_unknown_error() noexcept -> auto& {
161  if(_error_code != EGL_SUCCESS) {
162  _error_code = egl_types::int_type(~0U);
163  }
164  return *this;
165  }
166 
167 private:
168  egl_types::int_type _error_code{EGL_SUCCESS};
169 };
