23 /// @brief Implements constructor of scale transformation matrix.
24 /// @ingroup math
25 /// @see matrix_scale
26 ///
27 /// @note Do not use directly, use matrix_scale
28 template <typename T, bool RM, bool V>
29 class scale<matrix<T, 4, 4, RM, V>> {
30 public:
31  constexpr scale(const vect::data_t<T, 3, V> v) noexcept
32  : _v{v} {}
33 
34  /// @brief Initializes the matrix constructor.
35  /// @param vx is the x-axis scale factor.
36  /// @param vy is the y-axis scale factor.
37  /// @param vz is the z-axis scale factor.
38  constexpr scale(const T vx, const T vy, const T vz) noexcept
39  : _v{vx, vy, vz} {}
40 
41  /// @brief Returns the constructed matrix.
42  constexpr auto operator()() const noexcept {
43  return matrix<T, 4, 4, RM, V>{
44  {{_v[0], T(0), T(0), T(0)},
45  {T(0), _v[1], T(0), T(0)},
46  {T(0), T(0), _v[2], T(0)},
47  {T(0), T(0), T(0), T(1)}}};
48  }
49 
50  friend constexpr auto reorder_mat_ctr(
51  const scale<matrix<T, 4, 4, RM, V>>& c) noexcept
52  -> scale<matrix<T, 4, 4, !RM, V>> {
53  return {c._v};
54  }
55 
56 private:
57  vect::data_t<T, 3, V> _v;
58 };
