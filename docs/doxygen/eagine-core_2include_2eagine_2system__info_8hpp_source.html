124  /// @brief Returns the currently user power supply kind.
125  auto power_supply() noexcept -> power_supply_kind {
126  for(const auto i : integer_range(ac_supply_count())) {
127  if(ac_supply_online(i)) {
128  return power_supply_kind::ac_supply;
129  }
130  }
131  for(const auto i : integer_range(battery_count())) {
132  if(battery_capacity(i) > 0.F) {
133  return power_supply_kind::battery;
134  }
135  }
136  return power_supply_kind::unknown;
137  }
