19 template <typename Derived>
20 struct valid_if_filesystem_policy {
21  auto self() const noexcept -> const Derived& {
22  return *static_cast<const Derived*>(this);
23  }

93 // in writable directory
94 struct valid_if_in_writable_directory_policy
95  : valid_if_filesystem_policy<valid_if_in_writable_directory_policy> {
96  auto is_valid(std::filesystem::path path) const noexcept {
97  path.remove_filename();
98  const auto test = [](const auto& p) {
99  return is_directory(p) && (status(p).permissions() &
100  (std::filesystem::perms::owner_write |
101  std::filesystem::perms::group_write)) !=
102  std::filesystem::perms::none;
103  };
104  return test(path) || (is_symlink(path) && test(read_symlink(path)));
105  }
