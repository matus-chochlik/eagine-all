53 static inline auto from_string(
54  const string_view src,
55  const type_identity<bool>,
56  const selector<V>) noexcept -> optionally_valid<bool> {
57 
58  const string_view true_strs[] = {{"true"}, {"True"}, {"1"}, {"t"}, {"T"}};
59  if(find_element(view(true_strs), src)) {
60  return {true, true};

62 
63  const string_view false_strs[] = {
64  {"false"}, {"False"}, {"0"}, {"f"}, {"F"}};
65  if(find_element(view(false_strs), src)) {
66  return {false, true};
67  }
68 
69  return {};

101 template <typename T, typename N>
102 constexpr auto multiply_and_convert_if_fits(const N n, string_view t) noexcept
103  -> optionally_valid<T> {
104  if(t.empty()) {
105  return convert_if_fits<T>(n);
106  } else if(t.size() == 1) {
107  if(t.back() == 'k') {
108  return convert_if_fits<T>(n * 1000);
109  }
110  if(t.back() == 'M') {
111  return convert_if_fits<T>(n * 1000000);
112  }
113  if(t.back() == 'G') {
114  return convert_if_fits<T>(n * 1000000000);
115  }
116  } else if(t.size() == 2) {
117  if(t.back() == 'i') {
118  if(t.front() == 'K') {
119  return convert_if_fits<T>(n * 1024);
120  }
121  if(t.front() == 'M') {
122  return convert_if_fits<T>(n * 1024 * 1024);
123  }
124  if(t.front() == 'G') {
125  return convert_if_fits<T>(n * 1024 * 1024 * 1024);
126  }
127  }
128  }
129  return {};
130 }

132 template <typename T, typename N>
133 auto convert_from_string_with(
134  N (*converter)(const char*, char**),
135  const string_view src,
136  const type_identity<T> tid) noexcept -> optionally_valid<T> {
137  char* end = nullptr; // NOLINT(hicpp-vararg)
138  const auto cstr{c_str(src)};
139  errno = 0;
140  const N result{converter(cstr, &end)};
141  if((errno != ERANGE) && (end != cstr) && (end != nullptr)) {
142  if(auto converted{multiply_and_convert_if_fits<T>(
143  result, skip_to(cstr.view(), end))}) {
144  return converted;
145  }
146  }
147 
148  return parse_from_string(src, tid);
149 }
150 //------------------------------------------------------------------------------
151 template <typename T, typename N>
152 auto convert_from_string_with(
153  N (*converter)(const char*, char**, int),
154  const int base,
155  const string_view src,
156  const type_identity<T> tid) noexcept -> optionally_valid<T> {
157  char* end = nullptr; // NOLINT(hicpp-vararg)
158  const auto cstr{c_str(src)};
159  errno = 0;
160  const N result = converter(cstr, &end, base);
161  if((errno != ERANGE) && (end != cstr) && (end != nullptr)) {
162  if(auto converted{multiply_and_convert_if_fits<T>(
163  result, skip_to(cstr.view(), end))}) {
164  return converted;
165  }
166  }
167  return parse_from_string(src, tid);
168 }

242 template <identifier_t V>
243 static inline auto from_string(
244  const string_view src,
245  const type_identity<byte>,
246  const selector<V> sel) noexcept -> optionally_valid<byte> {
247  if(starts_with(src, string_view("0x"))) {
248  if(const auto opt_val{
249  from_string(skip(src, 2), type_identity<unsigned>(), sel, 16)}) {
250  return {static_cast<byte>(extract(opt_val)), bool(opt_val <= 255U)};
251  }
252  }
253  if(starts_with(src, string_view("0"))) {
254  if(const auto opt_val{
255  from_string(skip(src, 1), type_identity<unsigned>(), sel, 8)}) {
256  return {static_cast<byte>(extract(opt_val)), bool(opt_val <= 255U)};
257  }
258  }
259  if(const auto opt_val{
260  from_string(src, type_identity<unsigned>(), sel, 10)}) {
261  return {static_cast<byte>(extract(opt_val)), bool(opt_val <= 255U)};
262  }
263  return {};
264 }

298 template <typename Rep, typename Period, identifier_t V>
299 static inline auto convert_from_string(
300  const string_view src,
301  const type_identity<std::chrono::duration<Rep, Period>>,
302  const selector<V> sel,
303  const string_view symbol) noexcept
304  -> optionally_valid<std::chrono::duration<Rep, Period>> {
305  if(memory::ends_with(src, symbol)) {
306  if(
307  const auto opt_val =
308  from_string(snip(src, symbol.size()), type_identity<Rep>(), sel)) {
309  return {std::chrono::duration<Rep, Period>(extract(opt_val)), true};
310  }
311  }
312  return {};
313 }

316 static inline auto from_string(
317  const string_view str,
318  const type_identity<std::chrono::duration<Rep, Period>>,
319  const selector<V> sel) noexcept
320  -> optionally_valid<std::chrono::duration<Rep, Period>> {
321  using dur_t = std::chrono::duration<Rep, Period>;
322 
323  if(const auto d{convert_from_string(

325  type_identity<std::chrono::duration<Rep, std::ratio<1>>>(),
326  sel,
327  "s")}) {
328  return {std::chrono::duration_cast<dur_t>(extract(d)), true};
329  }
330  if(const auto d{convert_from_string(
331  str,
332  type_identity<std::chrono::duration<Rep, std::milli>>(),
