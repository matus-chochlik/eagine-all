140 /// @brief Registers the activity progress observer.
141 /// @ingroup main_context
142 /// @see main_ctx_setters
143 inline auto register_progress_observer(
144  main_ctx_getters& ctx,
145  progress_observer& observer) noexcept -> bool {
146  auto setters{ctx.setters()};
147  EAGINE_ASSERT(setters);
148  return extract(setters).register_observer(observer);
149 }
