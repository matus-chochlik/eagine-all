20 //------------------------------------------------------------------------------
21 /// @brief Abstract base class for deserialization data sources.
22 /// @ingroup serialization
23 /// @see data_sink
24 /// @see deserializer_backend
25 struct deserializer_data_source : interface<deserializer_data_source> {
26 
27  /// @brief Returns a block covering the specified amount of data of the top.
28  /// @see pop
29  /// @see scan_until
30  /// @see scan_for
31  virtual auto top(span_size_t size) noexcept -> memory::const_block = 0;
32 
33  /// @brief Returns the specified amount of data of the top of the source.
34  /// @see top
35  virtual void pop(span_size_t size) noexcept = 0;
36 
37  /// @brief Returns the position of the first byte where predicate is true.
38  /// @see scan_for
39  /// @param predicate the function indicating where to stop the scan.
40  /// @param max the maximum number of bytes from the top to scan.
41  /// @param step how much data should be fetched per scan iteration.
42  template <typename Function>
43  auto scan_until(
44  Function predicate,
45  const span_size_t max,
46  const span_size_t step = 256) noexcept
47  -> valid_if_nonnegative<span_size_t> {
48  EAGINE_ASSERT(max > 0);
49  EAGINE_ASSERT(step > 0);
50  const auto inc{step};
51  span_size_t start{0};
52  span_size_t total{inc};
53  while(const auto blk = top(total)) {
54  if(const auto found = find_element_if(skip(blk, start), predicate)) {
55  return {start + extract(found)};
56  }
57  if(blk.size() < total) {
58  break;
59  }
60  if(max < total) {
61  return max;
62  }
63  start += inc;
64  total += inc;
65  }
66  return {-1};
67  }
68 
69  /// @brief Returns the position of the first occurrence of the specified byte.
70  /// @see scan_until
71  /// @param what the searched value.
72  /// @param max the maximum number of bytes from the top to scan.
73  /// @param step how much data should be fetched per scan iteration.
74  auto scan_for(
75  const byte what,
76  const span_size_t max,
77  const span_size_t step = 256) noexcept
78  -> valid_if_nonnegative<span_size_t> {
79  EAGINE_ASSERT(max > 0);
80  EAGINE_ASSERT(step > 0);
81  return scan_until(
82  [what](const byte b) { return b == what; }, max, step);
83  }
84 
85  /// @brief Fetches all the remaining data into a buffer.
86  void fetch_all(memory::buffer& dst, const span_size_t step = 256) noexcept {
87  EAGINE_ASSERT(step > 0);
88  span_size_t offs{dst.size()};
89  while(const auto blk{top(step)}) {
90  dst.enlarge_by(blk.size());
91  copy(blk, skip(cover(dst), offs));
92  offs += blk.size();
93  pop(blk.size());
94  }
95  }
96 };
