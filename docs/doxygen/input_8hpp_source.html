19 /// @brief Application input value kind bits enumeration.
20 /// @ingroup application
21 /// @see input_value_kinds
22 enum class input_value_kind : unsigned {
23  /// @brief Relative input value change.
24  relative = 1U << 0U,
25  /// @brief Absolute input value normalized to <-1, 1>.
26  absolute_norm = 1U << 1U,
27  /// @brief Absolute input value without bounds.
28  absolute_free = 1U << 2U
29 };
30 
31 /// @brief Application input value kind bitfield.
32 /// @ingroup application
33 /// @see input_value_kinds
34 using input_value_kinds = bitfield<input_value_kind>;
35 
36 static inline auto operator|(
37  const input_value_kind l,
38  const input_value_kind r) noexcept -> input_value_kinds {
39  return {l, r};
40 }
41 
42 static inline auto all_input_value_kinds() noexcept -> input_value_kinds {
43  return input_value_kind::relative | input_value_kind::absolute_norm |
44  input_value_kind::absolute_free;
45 }

73 //------------------------------------------------------------------------------
74 /// @brief Class that managers user input state.
75 /// @ingroup application
76 class input_setup {
77 public:
78  /// @brief Constructor with input kinds specification.
79  auto value_kinds(const input_value_kinds init) noexcept -> auto& {
80  _value_kinds = init;
81  return *this;
82  }
83 
84  /// @brief Specifies that relative input values are tracked.
85  /// @see absolute_norm
86  /// @see absolute_free
87  /// @see any_value_kind
88  auto relative() noexcept -> auto& {
89  _value_kinds |= input_value_kind::relative;
90  return *this;
91  }
92 
93  /// @brief Specifies that absolute normalized input values are tracked.
94  /// @see relative
95  /// @see absolute_free
96  /// @see any_value_kind
97  auto absolute_norm() noexcept -> auto& {
98  _value_kinds |= input_value_kind::absolute_norm;
99  return *this;
100  }
101 
102  /// @brief Specifies that absolute unbounded input values are tracked.
103  /// @see relative
104  /// @see absolute_norm
105  /// @see any_value_kind
106  auto absolute_free() noexcept -> auto& {
107  _value_kinds |= input_value_kind::absolute_free;
108  return *this;
109  }
110 
111  /// @brief Specifies that input values of any kind are tracked.
112  /// @see relative
113  /// @see absolute_norm
114  /// @see absolute_free
115  auto any_value_kind() noexcept -> auto& {
116  _value_kinds = all_input_value_kinds();
117  return *this;
118  }
119 
120  /// @brief Indicates if values of the specified kind are tracked.
121  auto has(const input_value_kind kind) const noexcept -> bool {
122  return _value_kinds.has(kind);
123  }
124 
125  /// @brief Specifies that this is a trigger-like input.
126  /// @see absolute_norm
127  /// @see absolute_free
128  auto trigger() noexcept -> auto& {
129  return absolute_free().absolute_norm();
130  }
131 
132  /// @brief Specifies that the raw value from the input source should be inverted.
133  auto invert(const bool init = true) noexcept -> auto& {
134  _invert = init;
135  return *this;
136  }
137 
138  /// @brief Specifies the multiplier for the raw value from the input source.
139  auto multiply(const float mult) noexcept -> auto& {
140  _multiplier = mult;
141  return *this;
142  }
143 
144  /// @brief Returns the raw input value multiplier and the inverse setting.
145  auto multiplier() const noexcept {
146  return _invert ? -_multiplier : _multiplier;
147  }
148 
149  /// @brief Registers a flag that "switches" the input on and off.
150  /// @see is_applicable
151  auto only_if(bool& flag) noexcept -> auto& {
152  _only_if = &flag;
153  return *this;
154  }
155 
156  /// @brief Checks if an input value is currently applicable.
157  /// @see only_if
158  auto is_applicable() const noexcept {
159  return !_only_if || *_only_if;
160  }
161 
162 private:
163  float _multiplier{1.F};
164  bool* _only_if{nullptr};
165  input_value_kinds _value_kinds{};
166  bool _invert{false};
167 };
168 //------------------------------------------------------------------------------
169 /// @brief Class representing a user input.
170 /// @ingroup application
171 class input : public input_value<float> {
172 public:
173  template <typename T>
174  input(
175  const input_value<T>& value,
176  const input_info&,
177  const input_setup& setup) noexcept
178  : input_value<float>{transform(
179  [&](auto elem) { return setup.multiplier() * float(elem); },
180  value)} {}
181 
182  /// @brief Indicates if the current value is non-zero.
183  explicit operator bool() const noexcept {
184  return !are_equal(this->get(), 0.F);
185  }
186 };
