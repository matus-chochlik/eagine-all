41  template <typename T>
42  auto do_write(const span<const T> values, span_size_t& done) noexcept
43  -> result {
44  done = 0;
45  result errors{};
46  for(const auto& val : values) {
47  errors |= _write_one(val, type_identity<T>{});
48  errors |= do_sink(';');
49  if(errors) {
50  break;
51  }
52  ++done;
53  }
54 
55  return errors;
56  }

244  template <typename T>
245  auto do_read(span<T> values, span_size_t& done) noexcept -> result {
246  done = 0;
247  result errors{};

320 private:
321  auto _read_one(bool& value, const char delimiter) noexcept -> result {
322  result temp{};
323  if(this->consume("true", temp)) {
324  value = true;
325  } else if(this->consume("false", temp)) {
326  value = false;
327  } else {
328  return temp;
329  }
330  return require(delimiter);
331  }
332 
333  auto _read_one(char& value, const char delimiter) noexcept -> result {
334  result errors = require('\'');
335  if(!errors) {
336  if(const auto opt_char{this->top_char()}) {
337  value = extract(opt_char);
338  pop(1);
339  } else {
340  errors |= error_code::not_enough_data;
341  }
342  if(!errors) {
343  const char t[3] = {'\'', delimiter, '\0'};
344  errors |= require(string_view(t));
345  }
346  }
347  return errors;
348  }

350  template <typename T, std::size_t L>
351  auto _sscanf_one(
352  T& value,
353  const char delimiter,
354  const char (&fmt)[L]) noexcept -> result {
355  result errors{};
356  if(const auto src{this->string_before(delimiter, 128)}) {
357  auto fmtstr = static_cast<const char*>(fmt);
358  // TODO: to_chars from_chars when available
359  // NOLINTNEXTLINE(hicpp-vararg)
360  if(std::sscanf(src.data(), fmtstr, &value) == 1) {
361  pop(src.size() + 1);
362  } else {
363  errors |= error_code::invalid_format;
364  }
365  } else {
366  errors |= error_code::not_enough_data;
367  }
368  return errors;
369  }

449  if(const auto src{this->string_before(delimiter, max)}) {

458  auto _read_one(std::string& value, const char delimiter) noexcept
459  -> result {
460  result errors = require('"');
461  if(!errors) {
462  span_size_t len{0};
463  errors |= _read_one(len, '|');
464  if(!errors) {
465  auto str = this->top_string(len);
466  if(str.size() < len) {
467  errors |= error_code::not_enough_data;
468  } else {
469  assign_to(value, str);
470  pop(str.size());
471  errors |= require('"');
472  errors |= require(delimiter);
473  }
474  }
475  }
476  return errors;
477  }
