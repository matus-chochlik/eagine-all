19 /// @brief Extension of orbiting camera wrapper.
20 /// @ingroup application
21 class orbiting_camera : public oglplus::orbiting_camera {
22 
23 public:
24  using base = oglplus::orbiting_camera;
25  using base::matrix;
26 
27  /// @brief Construction from a reference to a video context.
28  auto matrix(video_context& vc) const noexcept {
29  return base::matrix(vc.surface_aspect());
30  }
31 
32  /// @brief Inddicates if the camera has changed and resets the flag.
33  /// @see mark_changed
34  auto has_changed() noexcept {
35  return std::exchange(_changed, false);
36  }
37 
38  /// @brief Inddicates that the camera has changed.
39  /// @see has_changed
40  auto mark_changed() noexcept -> orbiting_camera& {
41  _changed = true;
42  return *this;
43  }
44 
45  /// @brief Does a generic orbit update with given increment.
46  auto update_orbit(const float inc) noexcept -> orbiting_camera&;
47 
48  /// @brief Does a generic azimuth update with given increment.
49  auto update_turns(const float inc) noexcept -> orbiting_camera&;
50 
51  /// @brief Does a generic elevation update with given increment.
52  auto update_pitch(const float inc) noexcept -> orbiting_camera&;
53 
54  /// @brief Sets the maximum value for the pitch (elevation) angle.
55  /// @see set_pitch_min
56  auto set_pitch_max(const radians_t<float> max) noexcept -> auto& {
57  _pitch_max = max;
58  return *this;
59  }
60 
61  /// @brief Sets the minimum value for the pitch (elevation) angle.
62  /// @see set_pitch_max
63  auto set_pitch_min(const radians_t<float> min) noexcept -> auto& {
64  _pitch_min = min;
65  return *this;
66  }
67 
68  /// @brief Does a generic combined update when the user does not provide input.
69  /// @see update_orbit

109  dampening_input_id(), EAGINE_THIS_MEM_FUNC_REF(_handle_dampening)};

112  constexpr auto altitude_change_input_id() const noexcept -> message_id {

114  }
