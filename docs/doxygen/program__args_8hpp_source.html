179  /// @brief Tries to parse this argument's value into @p dest.
180  /// @returns True if the parse is successful, false otherwise.
181  template <typename T, identifier_t V>
182  auto parse(T& dest, const selector<V> sel, std::ostream& parse_log) const
183  -> bool {
184  if(is_valid()) {
185  T temp = dest;

216  auto missing_handler(std::ostream& errorlog) {
217  return [&errorlog](const string_view arg_tag) {

222  auto invalid_handler(std::ostream& errorlog) {
223  return [&errorlog](
224  const string_view arg_tag,

232  template <typename T, typename MissingFunc, typename InvalidFunc>
233  auto do_consume_next(
234  T& dest,
235  MissingFunc handle_missing,
236  InvalidFunc handle_invalid) -> bool {
237  if(next()) {
238  std::stringstream parse_log;
239  if(parse_next(dest, parse_log)) {
240  *this = next();
241  return true;
242  } else {
243  handle_invalid(
244  get(), next().get(), string_view(parse_log.str()));
245  }
246  } else {
247  handle_missing(get());
248  }
249  return false;
250  }

252  /// @brief Tries to parse the following argument's value into @p dest.
253  /// @returns True if the parse is successful, false otherwise.
254  template <typename T>
255  auto consume_next(T& dest, std::ostream& errorlog) -> bool {
256  auto if_missing = missing_handler(errorlog);
257  auto if_invalid = invalid_handler(errorlog);
258  return do_consume_next(dest, if_missing, if_invalid);
259  }

261  template <typename T, typename MissingFunc, typename InvalidFunc>
262  auto do_consume_next(
263  T& dest,
264  const span<const T> choices,
265  MissingFunc handle_missing,
266  InvalidFunc handle_invalid) {
267  valid_if_in_list<T, span<const T>> temp(T(), choices);
268  if(do_consume_next(temp, handle_missing, handle_invalid)) {
269  dest = temp.value();
270  return true;
271  }
272  return false;

301  template <typename T, typename MissingFunc, typename InvalidFunc>
302  auto do_consume_next(
303  T& dest,
304  const span<const string_view> symbols,
305  const span<const T> translations,
306  MissingFunc handle_missing,
307  InvalidFunc handle_invalid) {
308  EAGINE_ASSERT(symbols.size() <= translations.size());
309 
310  string_view parsed;
311  if(do_consume_next(parsed, symbols, handle_missing, handle_invalid)) {
312  for(const auto i : integer_range(symbols.size())) {
313  if(are_equal(parsed, symbols[i])) {
314  dest = translations[i];
315  return true;
316  }
317  }
318  }
319  return false;
320  }

373  auto _do_parse(T& dest, const selector<V> sel, const std::ostream&)

396  template <typename T, typename P, typename L, identifier_t V>
397  auto _do_parse(
398  valid_if<T, P, L>& dest,
399  const selector<V> sel,
400  std::ostream& parse_log) const -> bool {
401  typename valid_if<T, P, L>::value_type value{};
402  if(parse(value, sel, parse_log)) {
403  if(dest.is_valid(value)) {
404  dest = std::move(value);
405  return true;
406  } else {
407  dest.log_invalid(parse_log, value);
408  }
409  } else {
410  parse_log << "'" << get() << "' "
411  << "is not a valid `" << type_name<T>() << "` value";
412  }
413  return false;
414  }
415 
416  template <typename T, typename A, identifier_t V>
417  auto _do_parse(
418  std::vector<T, A>& dest,
419  const selector<V> sel,
420  std::ostream& parse_log) const -> bool {
421  T value{};
422  if(parse(value, sel, parse_log)) {
423  dest.push_back(std::move(value));
424  return true;
425  }
426  return false;

698  /// @brief Finds and returns the argument with the specified value.
699  auto find(const string_view what) const noexcept -> program_arg {
700  int i = 1;
701  while(i < _argc) {
702  if((_argv != nullptr) && (_argv[i] != nullptr)) {
703  if(are_equal(string_view(_argv[i]), what)) {
704  break;
705  }
706  }
707  ++i;
708  }
709  return {i, _argc, _argv};
