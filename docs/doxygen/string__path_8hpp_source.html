203  /// @brief Appends a new element with the specified name to the end.

292 private:
293  span_size_t _size{0};
294  std::string _str{};
295 
296  template <typename Str>
297  void _init(const span<Str> names) noexcept {
298  span_size_t len = 2 * names.size();
299  for(const auto& name : names) {
300  len += span_size(name.size());
301  }
302  _str.reserve(std_size(len));
303 
304  for(const auto& name : names) {
305  push_back(name);
306  }
307  EAGINE_ASSERT(span_size(_str.size()) == len);
308  }
309 
310  static inline auto _fix(string_view str) noexcept -> string_view {
311  while(str.size() > 0) {
312  if(str[str.size() - 1] == '\0') {
313  str = string_view(str.data(), str.size() - 1);
314  } else {
315  break;
316  }
317  }
318  return str;
319  }
320 
321  template <typename... Str>
322  static inline auto _pack_names(const Str&... n) noexcept
323  -> std::array<string_view, sizeof...(Str)> {
324  return {{_fix(n)...}};
325  }
326 };
