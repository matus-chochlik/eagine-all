49 template <typename F>
50 using decomposed_t =
51  std::tuple<decompose_fraction_t<F>, decompose_exponent_t<F>>;
52 //------------------------------------------------------------------------------
53 template <typename F>
54 constexpr auto max_fraction(type_identity<F> = {}) noexcept
55  -> decompose_fraction_t<F> requires(std::is_floating_point_v<F>) {
56  using T = decompose_fraction_t<F>;
57  return 1 + std::numeric_limits<T>::max() / T(2);
58 }
59 //------------------------------------------------------------------------------
60 template <typename F>
61 constexpr inline auto decompose(const F f, const type_identity<F> = {}) noexcept
62  -> decomposed_t<F> {
63  switch(std::fpclassify(f)) {
64  case FP_ZERO:
65  return {0, std::signbit(f) ? 1 : 0};
66  case FP_INFINITE:
67  return {0, std::signbit(f) ? 3 : 2};
68  case FP_NAN:
69  return {0, std::signbit(f) ? 5 : 4};
70  default:
71  break;
72  }
73  int exp = 0;
74  const auto one = F(1);
75  const auto two = F(2);
76  const auto fr2 = std::fabs(std::frexp(f, &exp) * two);
77  EAGINE_ASSERT(fr2 >= one && fr2 < two);
78  return {
79  (static_cast<decompose_fraction_t<F>>((fr2 - one) * max_fraction<F>())
80  << 1U) |
81  ((f < 0) ? 0x1U : 0x0U),
82  limit_cast<decompose_exponent_t<F>>(exp)};
83 }
84 //------------------------------------------------------------------------------
85 template <typename F>
86 constexpr inline auto compose(
87  const decomposed_t<F>& f,
88  const type_identity<F> = {}) noexcept -> F {
89  const auto fre = std::get<0>(f);
90  if(!fre) [[unlikely]] {
91  switch(std::get<1>(f)) {
92  case 0:
93  return static_cast<F>(0);
94  case 1:
95  return -static_cast<F>(0);
96  case 2:
97  return std::numeric_limits<F>::infinity();
98  case 3:
99  return -std::numeric_limits<F>::infinity();
100  case 4:
101  return std::numeric_limits<F>::quiet_NaN();
102  case 5:
103  return -std::numeric_limits<F>::quiet_NaN();
104  default:
105  break;
106  }
107  }
108  const auto one = F(1);
109  const auto two = F(2);
110  const auto half = one / two;
111  const auto sig = ((fre & 0x1U) == 0x1U) ? -one : one;
112  const auto frc = F(fre >> 1U) / max_fraction<F>() / two;
113  return sig * std::ldexp(frc + half, limit_cast<int>(std::get<1>(f)));
114 }
