17 namespace eagine::msgbus {
18 //------------------------------------------------------------------------------
19 /// @brief Structure containing basic information about a message bus endpoint.
20 struct subscriber_info {
21  /// @brief The endpoint id.
22  identifier_t endpoint_id{0U};
23  /// @brief The endpoint's instance (process) id.
24  process_instance_id_t instance_id{0U};
25 
26  /// @brief The type storing distance in number of hops to the endpoint.
27  using hop_count_t = std::int8_t;
28  /// @brief The distance in number of bus node hops to the endpoint.
29  /// @see max_hops
30  hop_count_t hop_count{0};
31 
32  /// @brief Returns the maximum possible value for hop_count.
33  /// @see hop_count
34  static constexpr auto max_hops() noexcept -> hop_count_t {
35  return std::numeric_limits<hop_count_t>::max();
36  }
37 };
38 //------------------------------------------------------------------------------
39 /// @brief Service discovering information about endpoint status and subscriptions.
40 /// @ingroup msgbus
41 /// @see service_composition
42 /// @see subscriber_info
43 template <typename Base = subscriber>
44 class subscriber_discovery : public Base {
45  using This = subscriber_discovery;
46 
47 public:
48  /// @brief Triggered on receipt of notification that an endpoint is alive.
49  signal<void(const subscriber_info&) noexcept> reported_alive;
50 
51  /// @brief Triggered on receipt of info that endpoint subscribes to message.
52  signal<void(const subscriber_info&, const message_id) noexcept> subscribed;
53 
54  /// @brief Triggered on receipt of info that endpoint unsubscribes from message.
55  signal<void(const subscriber_info&, const message_id) noexcept> unsubscribed;
56 
57  /// @brief Triggered on receipt of info that endpoint doesn't handle message type.
58  signal<void(const subscriber_info&, const message_id) noexcept>
59  not_subscribed;
60 
61 protected:
62  using Base::Base;
63 
64  void add_methods() noexcept {
65  Base::add_methods();
66  Base::add_method(
67  this, EAGINE_MSG_MAP(eagiMsgBus, stillAlive, This, _handle_alive));
68  Base::add_method(
69  this,
70  EAGINE_MSG_MAP(eagiMsgBus, subscribTo, This, _handle_subscribed));
71  Base::add_method(
72  this,
73  EAGINE_MSG_MAP(eagiMsgBus, unsubFrom, This, _handle_unsubscribed));
74  Base::add_method(
75  this,
76  EAGINE_MSG_MAP(eagiMsgBus, notSubTo, This, _handle_not_subscribed));
77  }
78 
79 private:
80  auto _handle_alive(
81  const message_context&,
82  const stored_message& message) noexcept -> bool {
83  subscriber_info info{};
84  info.endpoint_id = message.source_id;
85  info.instance_id = message.sequence_no;
86  info.hop_count = message.hop_count;
87  reported_alive(info);
88  return true;
89  }
90 
91  auto _handle_subscribed(
92  const message_context&,
93  const stored_message& message) noexcept -> bool {
94  message_id sub_msg_id{};
95  if(default_deserialize_message_type(sub_msg_id, message.content())) {
96  subscriber_info info{};
97  info.endpoint_id = message.source_id;
98  info.instance_id = message.sequence_no;
99  info.hop_count = message.hop_count;
100  subscribed(info, sub_msg_id);
101  }
102  return true;
103  }
104 
105  auto _handle_unsubscribed(
106  const message_context&,
107  const stored_message& message) noexcept -> bool {
108  message_id sub_msg_id{};
109  if(default_deserialize_message_type(sub_msg_id, message.content())) {
110  subscriber_info info{};
111  info.endpoint_id = message.source_id;
112  info.instance_id = message.sequence_no;
113  info.hop_count = message.hop_count;
114  unsubscribed(info, sub_msg_id);
115  }
116  return true;
117  }
118 
119  auto _handle_not_subscribed(
120  const message_context&,
121  const stored_message& message) noexcept -> bool {
122  message_id sub_msg_id{};
123  if(default_deserialize_message_type(sub_msg_id, message.content())) {
124  subscriber_info info{};
125  info.endpoint_id = message.source_id;
126  info.instance_id = message.sequence_no;
127  info.hop_count = message.hop_count;
128  not_subscribed(info, sub_msg_id);
129  }
130  return true;
131  }
132 };
