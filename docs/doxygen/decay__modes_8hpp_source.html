116 //------------------------------------------------------------------------------
117 template <decay_part... P>
118 struct decay_mode_traits<decay_mode_t<P...>> {
119 public:
120  static auto info() noexcept -> const auto& {
121  static const decay_mode_info i{
122  _make_symbol(),
123  (0 + ... + mode_info(decay_mode_t<P>{}).proton_count_diff),
124  (0 + ... + mode_info(decay_mode_t<P>{}).neutron_count_diff),
125  is_fission_v<P...>};
126  return i;
127  }
128 
129 private:
130  template <decay_part H>
131  static void _append_symbol(decay_mode_t<H> m, std::string& s) {
132  s.append(mode_info(m).symbol);
133  }
134 
135  template <decay_part H, decay_part N, decay_part... T>
136  static void _append_symbol(decay_mode_t<H, N, T...>, std::string& s) {
137  s.append(mode_info(decay_mode_t<H>{}).symbol);
138  s.append(",");
139  _append_symbol(decay_mode_t<N, T...>{}, s);
140  }
141 
142  static auto _make_symbol() noexcept -> std::string {
143  std::string s;
144  _append_symbol(decay_mode_t<P...>{}, s);
145  return s;
146  }
147 };

219 struct known_decay_modes {
220  using m = decay_part;
221  using list = mp_list<
222  decay_mode_t<m::alpha>,
223  decay_mode_t<m::proton_emi>,
224  decay_mode_t<m::neutron_emi>,
225  decay_mode_t<m::electron_cap>,
226  decay_mode_t<m::electron_cap, m::electron_cap>,
227  decay_mode_t<m::electron_cap, m::fission>,
228  decay_mode_t<m::beta_m>,
229  decay_mode_t<m::beta_m, m::beta_m>,
230  decay_mode_t<m::beta_m, m::alpha>,
231  decay_mode_t<m::beta_m, m::neutron_emi>,
232  decay_mode_t<m::beta_m, m::neutron_emi, m::neutron_emi>,
233  decay_mode_t<m::beta_p>,
234  decay_mode_t<m::beta_p, m::beta_p>,
235  decay_mode_t<m::beta_p, m::alpha>,
236  decay_mode_t<m::fission>,
237  decay_mode_t<m::transition>>;

239  template <decay_part... M>
240  static auto get_id(decay_mode_t<M...> = {}) noexcept {
241  return decay_mode_id<decay_mode_t<M...>>::value;
242  }
243 
244  static auto get_id(const string_view symbol) -> identifier_t {
245  return _get_id(symbol, list{});
246  }
247 
248  static auto get_info(const identifier_t mode_id) -> const decay_mode_info* {
249  return _get_info(mode_id, list{});
250  }
251 
252  static auto proton_count_diff(const identifier_t mode_id) noexcept -> int {
253  if(const auto i{get_info(mode_id)}) {
254  return extract(i).proton_count_diff;
255  }
256  return 0;
257  }
258 
259  static auto neutron_count_diff(identifier_t mode_id) noexcept -> int {
260  if(const auto i{get_info(mode_id)}) {
261  return extract(i).neutron_count_diff;
262  }
263  return 0;
264  }
265 
266 private:
267  static auto _get_info(identifier_t, mp_list<>) noexcept
268  -> const decay_mode_info* {
269  return nullptr;
270  }
271 
272  template <typename H, typename... T>
273  static auto _get_info(const identifier_t id, const mp_list<H, T...>)
274  -> const decay_mode_info* {
275  if(decay_mode_id<H>::value == id) {
276  return &mode_info(H{});
277  }
278  return _get_info(id, mp_list<T...>{});
279  }
280 
281  static auto _get_id(const string_view, const mp_list<>) noexcept
282  -> identifier_t {
283  return 0;
284  }
285 
286  template <typename H, typename... T>
287  static auto _get_id(const string_view symbol, const mp_list<H, T...>)
288  -> identifier_t {
289  if(are_equal(string_view(mode_info(H{}).symbol), symbol)) {
290  return get_id(H{});
291  }
292  return _get_id(symbol, mp_list<T...>{});
293  }
