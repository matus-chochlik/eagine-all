315 /// @brief Enum class for constants or enumerators (typically from a C-API).
316 /// @tparam Self the actual derived class based on this template.
317 /// @tparam T the type of the constant or enumerator value.
318 /// @tparam LibId an identifier of the "library" or API that this enum belongs to.
319 /// @tparam Id a type identifier of this enumeration (unique within an API).
320 /// @ingroup c_api_wrap
321 /// @see any_enum_class
322 /// @see enum_value
323 /// @see any_enum_value
324 /// @see opt_enum_value
325 /// @see no_enum_value
326 /// @see is_enum_class_v
327 /// @see enum_class_span
328 /// @see enum_class_view
329 /// @see enum_class_array
330 ///
331 /// Instantiations of this template can be used to represent a class of
332 /// enumerated values from a C-API that logically belong together. All values
333 /// from that logical class can be stored in enum_class and they are typically
334 /// declared as enum_value or opt_enum_value. The conversions between enum_class
335 /// and enum_value do static type checking to ensure that constants from
336 /// unrelated enum classes cannot be assigned.
337 template <typename Self, typename T, identifier_t LibId, identifier_t Id>
338 struct enum_class {
339  using type = enum_class;
340 
341  /// @brief The constant or enumerator value type.
342  using value_type = T;
343 
344  static constexpr const identifier_t lib_id = LibId;
345  static constexpr const identifier_t id = Id;
346 
347  value_type _value{};
348 
349  /// @brief Default constructor.
350  enum_class() = default;
351 
352  /// @brief Construction from a related enum_value.
353  template <typename Classes, typename Tag>
354  constexpr enum_class(const enum_value<T, Classes, Tag> ev) noexcept
355  requires(mp_contains_v<Classes, Self>)
356  : _value{ev.value} {}
357 
358  /// @brief Construction from a related opt_enum_value.
359  template <typename Classes, typename Tag>
360  constexpr enum_class(const opt_enum_value<T, Classes, Tag> ev) noexcept
361  requires(mp_contains_v<Classes, Self>)
362  : _value{ev.value} {
363  EAGINE_ASSERT(ev.is_valid);
364  }
365 
366  /// @brief Construction from a no_enum_value.
367  constexpr enum_class(const no_enum_value<T>) noexcept {
368  EAGINE_UNREACHABLE();
369  }
370 
371  /// @brief Construction from an any_enum_value.
372  constexpr enum_class(const any_enum_value<LibId>& aev) noexcept
373  : _value{static_cast<T>(aev._value)} {
374  EAGINE_ASSERT(aev._type_id == Id);
375  }
376 
377  /// @brief Explicit initialization from argument of value type.
378  explicit constexpr enum_class(const value_type value) noexcept
379  : _value{value} {}
380 
381  constexpr operator Self() const noexcept {
382  return Self{_value};
383  }
384 
385  /// @brief Explicit conversion to value type.
386  explicit constexpr operator value_type() const noexcept {
387  return _value;
388  }
389 
390  template <typename V>
391  explicit constexpr operator V() const noexcept requires(
392  !std::is_same_v<bool, V> && std::is_convertible_v<value_type, V>) {
393  return limit_cast<T>(_value);
394  }
395 
396  /// @brief Equality comparison.
397  friend constexpr auto operator==(
398  const enum_class a,
399  const enum_class b) noexcept {
400  return a._value == b._value;
401  }
402 
403  /// @brief Nonequality comparison.
404  friend constexpr auto operator!=(
405  const enum_class a,
406  const enum_class b) noexcept {
407  return a._value != b._value;
408  }
409 
410  struct transform {
411  constexpr auto operator()(const T value) noexcept {
412  return enum_class<Self, T, LibId, Id>{value};
413  }
414  };
415 };
416 template <
417  typename Dst,
418  typename Self,
419  typename Src,
420  identifier_t LibId,
421  identifier_t Id>
422 static constexpr auto limit_cast(enum_class<Self, Src, LibId, Id> val) noexcept
423  -> Dst requires(std::is_convertible_v<Src, Dst>) {
424  return limit_cast<Dst>(val._value);
425 }

433 struct any_enum_class {
434 
435  identifier_t _type_id{~identifier_t(0)};
436 
437  /// @brief Default constructor.
438  constexpr any_enum_class() noexcept = default;
439 
440  /// @brief Construction from enum_class from the same "library" or API.
441  template <typename Self, typename T, identifier_t Id>
442  constexpr any_enum_class(const enum_class<Self, T, LibId, Id>&) noexcept
443  : _type_id{Id} {
444  static_assert(std::is_base_of_v<enum_class<Self, T, LibId, Id>, Self>);

446 
447  /// @brief Construction from any_enum_value from the same "library" or API.
448  constexpr any_enum_class(const any_enum_value<LibId>& aev) noexcept
449  : _type_id{aev._type_id} {}
450

452  explicit constexpr operator bool() const noexcept {
453  return _type_id != ~identifier_t(0);
454  }
455 
456  /// @brief Equality comparison.
457  friend auto operator==(
458  const any_enum_class& a,
459  const any_enum_class& b) noexcept {
460  return a._type_id == b._type_id;
461  }
462 
463  /// @brief Nonequality comparison.
464  friend auto operator!=(
465  const any_enum_class& a,
466  const any_enum_class& b) noexcept {
467  return a._type_id != b._type_id;
468  }

471 /// @brief Type erasure for instantiations of enum_value from a specified library.
472 /// @tparam LibId unique identifier of a "library" or API the enums belong to.
473 /// @ingroup c_api_wrap
474 /// @see enum_value
475 /// @see any_enum_class
476 template <identifier_t LibId>
477 struct any_enum_value {
478  long _value{0};
479  identifier_t _type_id{~identifier_t(0)};
480 
481  /// @brief Default constructor.
482  constexpr any_enum_value() noexcept = default;
483 
484  /// @brief Construction from enum_class from the same "library" or API.
485  template <typename Self, typename T, identifier_t Id>
486  constexpr any_enum_value(const enum_class<Self, T, LibId, Id> v) noexcept
487  : _value{long(v._value)}
488  , _type_id{Id} {
489  static_assert(std::is_base_of_v<enum_class<Self, T, LibId, Id>, Self>);
490  }
491 
492  /// @brief Indicates if this value has a valid enumeration class.
493  explicit constexpr operator bool() const noexcept {
494  return _type_id != ~identifier_t(0);
495  }
496 
497  /// @brief Equality comparison.
498  friend auto operator==(
499  const any_enum_value& a,
500  const any_enum_value& b) noexcept {
501  return (a._value == b._value) && (a._type_id == b._type_id);
502  }
503 
504  /// @brief Nonequality comparison.
505  friend auto operator!=(
506  const any_enum_value& a,
507  const any_enum_value& b) noexcept {
508  return (a._value != b._value) || (a._type_id != b._type_id);
509  }
510 };
